00061   BoolExpr::BoolExpr(const BoolExpr& l, NodeType t, const BoolExpr& r)
00062     : n(new Node) {
00063     unsigned int ls = ((l.n->t == t) || l.n->t == BT_VAR) ? l.n->same : 0;
00064     unsigned int rs = ((r.n->t == t) || r.n->t == BT_VAR) ? r.n->same : 0;
00065     n->same = ls+rs;
00066     n->t    = t;
00067     n->l    = l.n;
00068     n->l->use++;
