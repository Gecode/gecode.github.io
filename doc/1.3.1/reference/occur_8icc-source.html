00187   forceinline int
00188   lookupValue(T& a, int v){
00189     int idx = -1;
00190 
00191     int l  = 0;
00192     int r  = a.size() - 1;
00193 
00194     if (r == 0) {
00195       if (a[0].card() == v) {
00196         return 0;
00197       } else {
00198         return -1;
00199       }
00200     }
00201 
00202     while ( l < r ) {
00203       if ( a[l].card() == v) {
00204         idx = l;
00205         break;
00206       }
00207       if ( a[r].card() == v) {
00208         idx = r;
00209         break;
00210       }
00211       int p  = (l + r) / 2;
00212       if ( v == a[p].card()) {
00213         idx = p;
00214         break;
00215       } else {
00216         if ( v < a[p].card()) {
00217           r = p;
00218         } else {
00219           l = p;
00220         }
00221       }
00222       if (l == r - 1) {
00223         break;
00224       }
00225     }
00226 
00227     return idx;
00228   }
