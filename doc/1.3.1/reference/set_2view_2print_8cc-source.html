00032  * Printing a bound
00033  *
00034  */
00035 template <class I>
00036 static void
00037 printBound(std::ostream& os, I& r) {
00038   os << '{';
00039   while (r()) {
00040     if (r.min() == r.max()) {
00041       os << r.min();
00042     } else if (r.min()+1 == r.max()) {
00043       os << r.min() << " " << r.max();
00044     } else {
00045       os << r.min() << "#" << r.max();
00046     }
00047     ++r;
00048     if (!r()) break;
00049     os << ' ';
00050   }
00051   os << '}';
00052 }
