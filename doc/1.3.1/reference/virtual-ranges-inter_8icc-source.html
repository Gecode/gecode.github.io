00120   Inter::operator++(void) {
00121     if (!(*i)() || !(*j)()) goto done;
00122     do {
00123       while ((*i)() && (i->max() < j->min())) ++(*i);
00124       if (!(*i)()) goto done;
00125       while ((*j)() && (j->max() < i->min())) ++(*j);
00126       if (!(*j)()) goto done;
00127     } while (i->max() < j->min());
00128     // Now the intervals overlap: consume the smaller interval
00129     ma = std::min(i->max(),j->max());
00130     mi = std::max(i->min(),j->min());
00131     if (i->max() < j->max()) ++(*i); else ++(*j);
00132     return;
00133   done:
00134     finish();
00135   }

00167   NaryInter::operator++(void) {
00168     // The next interval to be returned must have a hole
00169     // between it and the previously returned range.
00170     mi = ma+2;
00171     ma = is[0]->max();
00172     // Intersect with all other intervals
00173   restart:
00174     for (int i = n; i--;) {
00175       // Skip intervals that are too small
00176       while ((*(is[i]))() && (is[i]->max() < mi))
00177         ++(*(is[i]));
00178       if (!(*(is[i]))())
00179         goto done;
00180 
00181       if (is[i]->min() > ma) {
00182         mi=is[i]->min();
00183         ma=is[i]->max();
00184         goto restart;
00185       }
00186       // Now the intervals overlap
00187       if (mi < is[i]->min())
00188         mi = is[i]->min();
00189       if (ma > is[i]->max()) {
00190         ma = is[i]->max();
00191       }
00192     }
