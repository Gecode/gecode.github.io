00066   IntVarArray x;
00067 public:
00069   AllInterval(const SizeOptions& opt) :
00070     x(this, opt.size(), 0, opt.size() - 1) {
00071     const int n = x.size();
00072 
00073     IntVarArgs d(n-1);
00074 
00075     // Set up variables for distance
00076     for (int i=0; i<n-1; i++)
00077       d[i] = abs(this, minus(this,x[i+1],x[i],opt.icl()),opt.icl());
00078 
00079     // Constrain them to be between 1 and n-1
00080     dom(this, d, 1, n-1);
00081 
00082     distinct(this, x, opt.icl());
00083     distinct(this, d, opt.icl());
00084 
00085     // Break mirror symmetry
00086     rel(this, x[0], IRT_LE, x[1]);
00087     // Break symmetry of dual solution
00088     rel(this, d[0], IRT_GR, d[n-2]);
00089 
00090     branch(this, x, INT_VAR_SIZE_MIN, INT_VAL_SPLIT_MIN);
00091   }

00094     : Example(share, e) {
00095     x.update(this, share, e.x);
00096   }

00099   copy(bool share) {

00101   }
