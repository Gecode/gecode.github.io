00134   // Propagate domain information from x to y
00135   template <class View>
00136   ExecStatus
00137   prop_dom(Space* home, int n, DomInfo<View>* x, DomInfo<View>* y,
00138            ProcessStack& ya) {
00139     for (int i = n; i--; )
00140       // Only views with not yet propagated missing values
00141       if (x[i].dodom()) {
00142         // Iterate the values in the complement of x[i]
00143         ViewRanges<View>
00144           xir(x[i].view);
00145         Iter::Ranges::ComplVal<ViewRanges<View> >
00146           xirc(x[i].min,x[i].max,xir);
00147         Iter::Ranges::ToValues<Iter::Ranges::ComplVal<ViewRanges<View> > >
00148           jv(xirc);
00149         while (jv()) {
00150           // j is not in the domain of x[i], so prune i from y[j]
00151           int j = jv.val();
00152           ModEvent me = y[j].view.nq(home,i);
00153           if (me_failed(me))
00154             return ES_FAILED;
00155           if (me_modified(me)) {
00156             if (me == ME_INT_VAL) {
00157               // Record that y[j] has been assigned and must be propagated
00158               ya.push(j);
00159             } else {
00160               // Obvious as x[i] is different from j
00161               y[j].removed(i);
00162             }
00163           }
00164           ++jv;
00165         }
00166         // Update which values have been propagated and what are the new bounds
00167         x[i].done();
00168       }
00169     return ES_OK;
00170   }

00252           // Propagate assigned views for x
00253           GECODE_ES_CHECK((prop_val<View,DomInfo<View> >
00254                            (home,n,x,y,n_na,xa,ya)));
00255           // Propagate assigned views for y
00256           GECODE_ES_CHECK((prop_val<View,DomInfo<View> >
00257                            (home,n,y,x,n_na,ya,xa)));
00258           assert(ya.empty());

00260         return ES_FIX_PARTIAL(this,View::med(ME_INT_DOM));
00261       }
00262     }

00265     // This propagates from y to x and possibly records xs that
00266     // got assigned
00267     GECODE_ES_CHECK(prop_dom(home,n,y,x,xa));
00268 
00269     // Process assigned views for x
00270     GECODE_ES_CHECK((prop_val<View,DomInfo<View> >(home,n,x,y,n_na,xa,ya)));
00271 
00272     // Perform domain consistent propagation for distinct on the x views
00273     if (dc.available()) {
00274       GECODE_ES_CHECK(dc.sync());
00275     } else {
00276       GECODE_AUTOARRAY(View,xv,n);
00277       for (int i=n; i--; )
00278         xv[i] = x[i].view;
00279       GECODE_ES_CHECK(dc.init(home,n,&xv[0]));
00280     }

00282     GECODE_ES_CHECK(dc.propagate(home,assigned));
00283     if (assigned) {
00284       // This has assigned some more views in x which goes unnoticed
00285       // (that is, not recorded in xa). This must be checked and propagated
00286       // to the y views, however the distinctness on x is already
00287       // propagated.
00288       for (int i=n; i--; )
00289         if (x[i].doval()) {
00290           int j = x[i].view.val();
00291           // Assign the y variable to i (or test if already assigned!)
00292           ModEvent me = y[j].view.eq(home,i);
00293           if (me_failed(me))
00294             return ES_FAILED;
00295           if (me_modified(me)) {
00296             // Record that y[j] has been assigned and must be propagated
00297             assert(me == ME_INT_VAL);
00298             // Otherwise the modification event would not be ME_INT_VAL
00299             ya.push(j);
00300           }
00301           x[i].assigned(); n_na--;
00302         }
00303     }
00304 
00305     // Process changed views for x
00306     // This propagates from x to y and possibly records ys that
00307     // got assigned
00308     GECODE_ES_CHECK(prop_dom(home,n,x,y,ya));
00309 
00310     // Process assigned view again (some might have been found above)
00311     while (!xa.empty() || !ya.empty()) {
00312       // Process assigned views for x
00313       GECODE_ES_CHECK((prop_val<View,DomInfo<View> >(home,n,x,y,n_na,xa,ya)));
00314       // Process assigned views for y
00315       GECODE_ES_CHECK((prop_val<View,DomInfo<View> >(home,n,y,x,n_na,ya,xa)));
00316     };

00318     if (shared) {

00321           return ES_NOFIX;
