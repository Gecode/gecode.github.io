00087   template <class View>
00088   ExecStatus
00089   Dom<View>::propagate(Space* home, ModEventDelta med) {
00090     if (View::me(med) == ME_INT_VAL) {
00091       GECODE_ES_CHECK((Distinct::prop_val<View,true>(home,y)));
00092       ExecStatus escv = connected(home);
00093       if (escv != ES_FIX)
00094         return escv;
00095       if (y.size() < 2)
00096         return ES_SUBSUMED(this,home);
00097       return ES_FIX_PARTIAL(this,View::med(ME_INT_DOM));
00098     }
00099 
00100     if (dc.available()) {
00101       GECODE_ES_CHECK(dc.sync());
00102     } else {
00103       GECODE_ES_CHECK(dc.init(home,y.size(),&y[0]));
00104     }
00105     bool assigned;
00106     GECODE_ES_CHECK(dc.propagate(home,assigned));
00107 
00108     ExecStatus esc = connected(home);
00109     if (esc != ES_FIX)
00110       return esc;
00111 
00112     // Elminiate assigned views from y, as they have been assigned
00113     // and propagated by domain consistent propagation. This is required
00114     // as we need to know how many assigned views actually exist.
00115     if (assigned)
00116       for (int i=y.size(); i--; )
00117         if (y[i].assigned())
00118           y.move_lst(i);
00119 
00120     return path(home);
00121   }

00123   template <class View>
00124   ExecStatus
00125   Dom<View>::post(Space* home, ViewArray<View>& x) {
00126     int n = x.size();
00127     if (n == 1) {
00128       GECODE_ME_CHECK(x[0].eq(home,0));
00129     } else if (n == 2) {
00130       GECODE_ME_CHECK(x[0].eq(home,1));
00131       GECODE_ME_CHECK(x[1].eq(home,0));
00132     } else {
00133       for (int i=n; i--; ) {
00134         GECODE_ME_CHECK(x[i].gq(home,0));
00135         GECODE_ME_CHECK(x[i].le(home,n));
00136         GECODE_ME_CHECK(x[i].nq(home,i));
00137       }
00138       (void) new (home) Dom<View>(home,x);
00139     }
00140     return ES_OK;
00141   }
