00091     ExecStatus esc = connected(home);
00092     if (esc != ES_FIX)

00094     /*

00103   template <class View>
00104   ExecStatus
00105   Val<View>::post(Space* home, ViewArray<View>& x) {
00106     int n = x.size();
00107     if (n == 1) {
00108       GECODE_ME_CHECK(x[0].eq(home,0));
00109     } else if (n == 2) {
00110       GECODE_ME_CHECK(x[0].eq(home,1));
00111       GECODE_ME_CHECK(x[1].eq(home,0));
00112     } else {
00113       for (int i=n; i--; ) {
00114         GECODE_ME_CHECK(x[i].gq(home,0));
00115         GECODE_ME_CHECK(x[i].le(home,n));
00116         GECODE_ME_CHECK(x[i].nq(home,i));
00117       }
00118       (void) new (home) Val<View>(home,x);
00119     }
00120     return ES_OK;
00121   }
