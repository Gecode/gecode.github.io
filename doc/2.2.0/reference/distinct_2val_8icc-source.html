00044   template <class View, bool complete>
00045   ExecStatus
00046   prop_val(Space* home, ViewArray<View>& x) {
00047     assert(x.size() > 1);
00048     int n = x.size();
00049 
00050     GECODE_AUTOARRAY(int, stack, n);
00051     int* c_v = &stack[0];
00052     // c_n is the current number of values on stack
00053     int c_n = 0;
00054 
00055     // Collect all assigned variables on stack
00056     for (int i = n; i--; )
00057       if (x[i].assigned()) {
00058         c_v[c_n++]=x[i].val(); x[i]=x[--n];
00059       }
00060 
00061     // The number of trips
00062     int t = 0;
00063     do {
00064       t++;
00065       if (!complete && (t > 16)) {
00066         // Give up after sixteen iterations, but the values must be
00067         // propagated first
00068         // Maybe we are lucky in that this iteration does the trick...
00069         ExecStatus es = ES_FIX;
00070         while (c_n > 0) {
00071           int v = c_v[--c_n];
00072           // Check whether value is on stack only once
00073           for (int i = c_n; i--; )
00074             if (c_v[i] == v)
00075               goto failed;
00076           // Tell and do not collect new values
00077           for (int i = n; i--; ) {
00078             ModEvent me = x[i].nq(home,v);
00079             if (me_failed(me))
00080               goto failed;
00081             if (me == ME_INT_VAL)
00082               es = ES_NOFIX;
00083           }
00084         }
00085         x.size(n);
00086         return es;
00087       }
00088       if (c_n > 31) {
00089         // Many values, use full domain operation
00090         IntSet d(&c_v[0],c_n);
00091         // Check for failure
00092         unsigned int s = 0;
00093         for (int i = d.size(); i--; )
00094           s += d.width(i);
00095         // If the size s of d is different from the number of values,
00096         // a value must have appeared multiply: failure
00097         if (s != static_cast<unsigned int>(c_n))
00098           goto failed;
00099         // We do not need the values on the stack any longer, reset
00100         c_n = 0;
00101         // Tell and collect new values
00102         for (int i = n; i--; )
00103           if ((d.min() <= x[i].max()) && (d.max() >= x[i].min())) {
00104             IntSetRanges dr(d);
00105             ModEvent me = x[i].minus_r(home,dr,false);
00106             if (me_failed(me))
00107               goto failed;
00108             if (me == ME_INT_VAL) {
00109               c_v[c_n++]=x[i].val(); x[i]=x[--n];
00110             }
00111           }
00112       } else {
00113         // Values for next iteration
00114         int* n_v = &c_v[c_n];
00115         // Stack top for the next iteration
00116         int n_n = 0;
00117         while (c_n > 0) {
00118           int v = c_v[--c_n];
00119           // Check whether value is not on current stack
00120           for (int i = c_n; i--; )
00121             if (c_v[i] == v)
00122               goto failed;
00123           // Check whether value is not on next stack
00124           for (int i = n_n; i--; )
00125             if (n_v[i] == v)
00126               goto failed;
00127           // Tell and collect new values
00128           for (int i = n; i--; ) {
00129             ModEvent me = x[i].nq(home,v);
00130             if (me_failed(me))
00131               goto failed;
00132             if (me == ME_INT_VAL) {
00133               n_v[n_n++]=x[i].val(); x[i]=x[--n];
00134             }
00135           }
00136         }
00137         c_v = n_v; c_n = n_n;
00138       }
00139     } while (c_n > 0);
00140     x.size(n);
00141     return ES_FIX;
00142   failed:
00143     x.size(0);
00144     return ES_FAILED;
00145   }
