00044   void
00045   branch(Space* home, const CpltSetVarArgs& xa,
00046          CpltSetVarBranch vars, CpltSetValBranch vals) {
00047     if (home->failed()) 
00048       return;
00049     ViewArray<CpltSetView> x(home,xa);
00050     switch (vars) {
00051     case CPLTSET_VAR_NONE:
00052       Branch::create<Branch::ByNone>(home,x,vals); break;
00053     case CPLTSET_VAR_MIN_CARD:
00054       Branch::create<Branch::ByMinCard>(home,x,vals); break;
00055     case CPLTSET_VAR_MAX_CARD:
00056       Branch::create<Branch::ByMaxCard>(home,x,vals); break;
00057     case CPLTSET_VAR_MIN_UNKNOWN_ELEM:
00058       Branch::create<Branch::ByMinUnknown>(home,x,vals); break;
00059     case CPLTSET_VAR_MAX_UNKNOWN_ELEM:
00060       Branch::create<Branch::ByMaxUnknown>(home,x,vals); break;
00061     default:
00062       throw CpltSet::UnknownBranching("CpltSet::branch");
00063     }
00064   }
