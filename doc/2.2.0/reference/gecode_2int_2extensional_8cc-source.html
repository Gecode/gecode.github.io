00070               IntConLevel, PropKind pk) {

00072     if (!t.finalized())

00077

00079     ViewArray<IntView> xv(home,x);
00080 
00081     // All variables in the correct domain

00083       GECODE_ME_FAIL(home, xv[i].gq(home, t.min()));
