00574   VarArray<Var>::resize(Space* home, int m) {
00575     int newsize;
00576     if (m<used) {
00577       newsize = m;
00578     } else if (m<n) {
00579       used = m; n = m;
00580       return;

00582       newsize = std::max(m, (3*n)/2);

00588     if (used != n)

00591       home->reuse(oldx, n);
00592     n = newsize; used = m;
