00077   ExecStatus
00078   LinkSingle::propagate(Space* home, ModEventDelta) {
00079     if (x0.zero()) {
00080       GECODE_ME_CHECK(x1.eq(home,0));
00081     } else if (x0.one()) {
00082       GECODE_ME_CHECK(x1.eq(home,1));
00083     } else {
00084       assert(x0.none() && x1.assigned());
00085       if (x1.val() == 0) {
00086         GECODE_ME_CHECK(x0.zero_none(home));
00087       } else {
00088         assert(x1.val() == 1);
00089         GECODE_ME_CHECK(x0.one_none(home));
00090       }
00091     }
00092     return ES_SUBSUMED(this,sizeof(*this));
