00230   forceinline int
00231   lookupValue(T& a, int v){
00232     int idx = -1;
00233 
00234     int l  = 0;
00235     int r  = a.size() - 1;
00236 
00237     if (r == 0) {
00238       if (a[0].card() == v) {
00239         return 0;
00240       } else {
00241         return -1;
00242       }
00243     }
00244 
00245     while ( l < r ) {
00246       if ( a[l].card() == v) {
00247         idx = l;
00248         break;
00249       }
00250       if ( a[r].card() == v) {
00251         idx = r;
00252         break;
00253       }
00254       int p  = (l + r) / 2;
00255       if ( v == a[p].card()) {
00256         idx = p;
00257         break;
00258       } else {
00259         if ( v < a[p].card()) {
00260           r = p;
00261         } else {
00262           l = p;
00263         }
00264       }
00265       if (l == r - 1) {
00266         break;
00267       }
00268     }
00269 
00270     return idx;
00271   }
