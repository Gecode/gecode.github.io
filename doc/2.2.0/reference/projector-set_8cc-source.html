00050   void
00051   ProjectorSet::scope(Support::DynamicArray<int>& s,
00052                       unsigned int size) const {
00053     // Clear out s
00054     for (int i=std::max((unsigned)_arity+1, size); i--;)
00055       s[i]=Set::PC_SET_ANY + 1;
00056 
00057     // Collect scope from individual projectors
00058     for (int i=_count; i--; ) {
00059       _ps[i].scope(s);
00060     }
00061   }

00063   ExecStatus
00064   ProjectorSet::check(Space* home, ViewArray<Set::SetView>& x) {
00065     ExecStatus es = __ES_SUBSUMED;
00066     for (int i=0; i<_count; i++) {
00067       ExecStatus es_new = _ps[i].check(home, x);
00068       switch (es_new) {
00069       case ES_FAILED:
00070         return ES_FAILED;
00071       case __ES_SUBSUMED:
00072         break;
00073       default:
00074         es = es_new;
00075         break;
00076       }
00077     }
00078     return es;
00079   }
