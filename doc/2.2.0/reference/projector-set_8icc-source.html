00056   template <bool negated>
00057   inline ExecStatus
00058   ProjectorSet::propagate(Space* home, ViewArray<Set::SetView>& x) {
00059     int failedCount = 0;
00060     for (int i=_count; i--; ) {
00061       ExecStatus es = _ps[i].propagate<negated>(home, x);
00062       switch (es) {
00063       case ES_FAILED:
00064         if (!negated)
00065           return ES_FAILED;
00066         failedCount++;
00067         break;
00068       case __ES_SUBSUMED:
00069         return __ES_SUBSUMED;
00070       default:
00071         break;
00072       }
00073     }
00074     return failedCount==_count ? ES_FAILED : ES_NOFIX;
00075   }
