00045   template <class T>
00046   Restart<T>::Restart(T* s, const Search::Options& o) :
00047     Gecode::DFS<T>(s,o),
00048     root(s->status() == SS_FAILED ? NULL : s->clone()), best(NULL) {
00049     // Failure and propagation must not be counted, has happened already
00050     // in the DFS constructor
00051   }

00053   template <class T>
00054   forceinline
00055   Restart<T>::~Restart(void) {
00056     delete best;
00057     delete root;
00058   }

00060   template <class T>
00061   forceinline T*
00062   Restart<T>::next(void) {
00063     if (best != NULL) {
00064       dynamic_cast<T*>(root)->constrain(dynamic_cast<T*>(best));
00065       this->e.clone++;
00066       if (root->status(this->e.propagate) == SS_FAILED) {
00067         this->e.reset();
00068       } else {
00069         this->e.reset(root->clone());
00070       }
00071     }
00072     Space* b = this->e.explore();
00073     delete best;
00074     best = (b != NULL) ? b->clone() : NULL;
00075     return dynamic_cast<T*>(b);
00076   }

00086   template <class T>
00087   T*
00088   restart(T* s, const Search::Options& o) {
00089     Restart<T> b(s,o);
00090     T* l = NULL;
00091     while (T* n = b.next()) {
00092       delete l; l = n;
00093     }
00094     return l;
00095   }
