00046 class StressMin : public Example {
00047 protected:
00049   const int n;
00051   IntVarArray x;
00052 public:
00054   StressMin(const SizeOptions& opt)
00055     : n(opt.size()), x(this,n,0,2*n-1) {
00056     for (int i=1; i<n; i++) {
00057       IntVarArgs y(i);
00058       for (int j=0; j<i; j++)
00059         y[j]=x[j];
00060       IntVar m(this,0,2*n);
00061       min(this, y, m);
00062       rel(this, m, IRT_GR, x[i]);
00063     }
00064     branch(this, x, INT_VAR_NONE, INT_VAL_SPLIT_MAX);
00065   }
00066 
00068   StressMin(bool share, StressMin& s) : Example(share,s), n(s.n) {
00069     x.update(this, share, s.x);
00070   }
