00217               I& iter) {

00220     for (Iter::Ranges::ToValues<I> v(iter); v(); ++v) {
00221       // Skip all elements below the current
00222       while (elements[i]<v.val()) i++;
00223       assert(elements[i] == v.val());

00225       case ALL_COST:
00226         sum += weights[i];

00228       case POS_COST:
00229         if (weights[i] > 0) sum += weights[i];
