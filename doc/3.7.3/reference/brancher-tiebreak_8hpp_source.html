00546     switch (tb[0]->select(home,x)) {
00547     case VSS_BEST:
00548       {

00550         for (int i=1; i<n; i++)

00556       for (int i=1; i<n; i++)
00557         (void) tb[i]->init(home,x);

00559     case VSS_WORSE:

00561     case VSS_TIE:
