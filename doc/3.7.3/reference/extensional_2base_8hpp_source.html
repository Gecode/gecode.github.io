00104   forceinline Tuple
00105   Base<View,subscribe>::last_next(int i, int n) {
00106     assert(last(i,n) != NULL);
00107     assert(last(i,n)[i] == n+ts()->min);
00108     int pos = (i*static_cast<int>(ts()->domsize)) + n;
00109     ++(last_data[pos]);
