00041   forceinline
00042   IterPruneVal<View>::IterPruneVal(ViewNode<View>* y)
00043     : x(y), e(y->val_edges()) {
00044     while ((e != NULL) && e->used(x))
00045       e = e->next_edge();
00046   }
00047   template<class View>
00048   forceinline bool
00049   IterPruneVal<View>::operator ()(void) const {
00050     return e != NULL;

00052   template<class View>
00053   forceinline void
00054   IterPruneVal<View>::operator ++(void) {
00055     assert(!e->used(x));
00056     do {
00057       e = e->next_edge();
00058     } while ((e != NULL) && e->used(x));
00059   }

00061   forceinline int
00062   IterPruneVal<View>::val(void) const {
