00048   forceinline ExecStatus
00049   LinkMulti::post(Home home, ViewArray<BoolView>& x, IntView y, int o) {
00050     int n=x.size();
00051     GECODE_ME_CHECK(y.gq(home,o));
00052     GECODE_ME_CHECK(y.lq(home,o+n-1));
00053     assert(n > 0);
00054     if (n == 1) {
00055       GECODE_ME_CHECK(x[0].one(home));
00056       assert(y.val() == o);
00057     } else if (y.assigned()) {
00058       int j=y.val()-o;
00059       GECODE_ME_CHECK(x[j].one(home));
00060       for (int i=0; i<j; i++)
00061         GECODE_ME_CHECK(x[i].zero(home));
00062       for (int i=j+1; i<n; i++)
00063         GECODE_ME_CHECK(x[i].zero(home));
00064     } else {
00065       for (int i=n; i--; )
00066         if (x[i].one()) {
00067           for (int j=0; j<i; j++)
00068             GECODE_ME_CHECK(x[j].zero(home));
00069           for (int j=i+1; j<n; j++)
00070             GECODE_ME_CHECK(x[j].zero(home));
00071           GECODE_ME_CHECK(y.eq(home,o+i));
00072           return ES_OK;
00073         } else if (x[i].zero()) {
00074           GECODE_ME_CHECK(y.nq(home,o+i));
00075         }
00076       (void) new (home) LinkMulti(home,x,y,o);
00077     }
00078     return ES_OK;
00079   }
