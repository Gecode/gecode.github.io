00281   inline int
00282   MinusView::med(void) const {
00283     if (x.range())
00284       return (min()+max())/2 - ((min()+max())%2 < 0 ? 1 : 0);
00285 
00286     unsigned int i = x.size() / 2;
00287     if (size() % 2 == 0)
00288       i--;
00289     ViewRanges<MinusView> r(*this);
00290     while (i >= r.width()) {
00291       i -= r.width();
00292       ++r;
00293     }
00294     return r.min() + static_cast<int>(i);
