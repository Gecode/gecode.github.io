00048   template<class View>
00049   ExecStatus
00050   ReRange<View>::post(Home home, View x, int min, int max, BoolView b) {
00051     if (min == max) {
00052       return Rel::ReEqDomInt<View,BoolView>::post(home,x,min,b);
00053     } else if ((min > max) || (max < x.min()) || (min > x.max())) {
00054       GECODE_ME_CHECK(b.zero(home));
00055     } else if ((min <= x.min()) && (x.max() <= max)) {
00056       GECODE_ME_CHECK(b.one(home));
00057     } else if (b.one()) {
00058       GECODE_ME_CHECK(x.gq(home,min));
00059       GECODE_ME_CHECK(x.lq(home,max));
00060     } else if (b.zero()) {
00061       Iter::Ranges::Singleton r(min,max);
00062       GECODE_ME_CHECK(x.minus_r(home,r,false));
00063     } else {
00064       (void) new (home) ReRange<View>(home,x,min,max,b);
00065     }
00066     return ES_OK;
00067   }

00083   ExecStatus
00084   ReRange<View>::propagate(Space& home, const ModEventDelta&) {
00085     if (b.one()) {
00086       GECODE_ME_CHECK(x0.gq(home,min));
00087       GECODE_ME_CHECK(x0.lq(home,max));
00088     } else if (b.zero()) {
00089       Iter::Ranges::Singleton r(min,max);
00090       GECODE_ME_CHECK(x0.minus_r(home,r,false));
00091     } else if ((x0.max() <= max) && (x0.min() >= min)) {
00092       GECODE_ME_CHECK(b.one_none(home));
00093     } else if ((x0.max() < min) || (x0.min() > max)) {
00094       GECODE_ME_CHECK(b.zero_none(home));
00095     } else {
00096       return ES_FIX;

00098     return home.ES_SUBSUMED(*this);
