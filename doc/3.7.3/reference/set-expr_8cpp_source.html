00475   SetExpr::NNF::nnf(Region& r, Node* n, bool neg) {

00478       {
00479         NNF* x = new (r) NNF;
00480         x->t = n->t; x->neg = neg; x->u.a.x = n;
00481         if (neg) {

00646           }
00647         } else if (t==SNLE_MIN && (irt==IRT_GR || irt==IRT_GQ)) {
