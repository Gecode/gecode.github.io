00091     if (size > 0) {

00093       IntSetRanges dr(dom);

00095         int min = dr.min(); int max = dr.max();
00096         ranges[i] = min;
00097         ranges[i+1] = max;
00098         domSize += static_cast<unsigned int>(max-min+1);
00099       }
00100     } else {
00101       ranges = NULL;
