00348   forceinline void
00349   Engine::terminate(void) {
00350     // Grab the wait mutex for termination
00351     _m_wait_terminate.acquire();
00352     // Release all threads
00353     release(C_TERMINATE);
00354     // Wait until all threads have acknowledged termination request
00355     _e_term_ack.wait();
00356     // Release waiting threads
00357     _m_wait_terminate.release();    
00358     // Wait until all threads have in fact terminated
00359     _e_terminate.wait();
00360     // Now all threads are terminated!

00392   /*
00393    * Worker: finding and stealing working
00394    */
00395   forceinline Space*
00396   Engine::Worker::steal(unsigned long int& d) {
00397     /*
00398      * Make a quick check whether the worker might have work
00399      *
00400      * If that is not true any longer, the worker will be asked
00401      * again eventually.
00402      */
00403     if (!path.steal())
00404       return NULL;
00405     m.acquire();
00406     Space* s = path.steal(*this,d);
00407     m.release();
00408     // Tell that there will be one more busy worker
00409     if (s != NULL) 
00410       engine().busy();
00411     return s;
00412   }
