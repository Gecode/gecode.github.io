00064   inline IntType 
00065   PowOps::pow(IntType x) const {
00066     int m = n;
00067     IntType p = 1;
00068     do {
00069       if (even(m)) {
00070         x *= x; m >>= 1;
00071       } else {
00072         p *= x; m--;
00073       }
00074     } while (m > 0);
00075     return p;
00076   }

00118   inline int 
00119   PowOps::fnroot(int x) const {
00120     if (x < 2)
00121       return x;
00122     /*
00123      * We look for l such that: l^n <= x < (l+1)^n
00124      */
00125     long long int l = 1;
00126     long long int u = x;
00127     do {
00128       long long int m = (l + u) >> 1;
00129       if (powgr(m,x)) u=m; else l=m;
00130     } while (l+1 < u);
00131     assert((pow(l) <= x) && (x < pow(l+1)));
00132     return static_cast<int>(l);
00133   }
