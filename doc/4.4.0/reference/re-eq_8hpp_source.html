00094   template<class View0, class View1, class CtrlView, ReifyMode rm>
00095   ExecStatus
00096   ReEq<View0,View1,CtrlView,rm>::propagate(Space& home,
00097                                            const ModEventDelta&) {
00098     if (b.one()) {
00099       if (rm == RM_PMI)
00100         return home.ES_SUBSUMED(*this);
00101       GECODE_REWRITE(*this,(Eq<View0,View1>::post(home(*this),x0,x1)));
00102     }
00103     if (b.zero()) {
00104       if (rm == RM_IMP)
00105         return home.ES_SUBSUMED(*this);        
00106       GECODE_REWRITE(*this,(Distinct<View0,View1>::post(home(*this),x0,x1)));
00107     }
00108 
00109     if (x0.assigned() && x1.assigned()) {
00110       // directly test x0==x1
00111       GlbRanges<View0> x0lb(x0);
00112       GlbRanges<View1> x1lb(x1);
00113       bool x0eqx1 = true;
00114       for (; x0lb() && x1lb(); ++x0lb, ++x1lb) {
00115         if (x0lb.min() != x1lb.min() ||
00116             x0lb.max() != x1lb.max()) {
00117           x0eqx1 = false;
00118           break;
00119         }
00120       }
00121       if (x0eqx1 && !x0lb() && !x1lb()) {
00122         if (rm != RM_IMP)
00123           GECODE_ME_CHECK(b.one_none(home));
00124         return home.ES_SUBSUMED(*this);
00125       } else {
00126         if (rm != RM_PMI)
00127           GECODE_ME_CHECK(b.zero_none(home));
00128         return home.ES_SUBSUMED(*this);
00129       }
00130     }
00131 
00132     // check whether cardinalities still allow equality
00133     if (x0.cardMin() > x1.cardMax() ||
00134         x1.cardMin() > x0.cardMax()) {
00135       if (rm != RM_PMI)
00136         GECODE_ME_CHECK(b.zero_none(home));
00137       return home.ES_SUBSUMED(*this);
00138     }
00139 
00140     // check glb(x0) subset lub(x1)
00141     GlbRanges<View0> x0lb(x0);
00142     LubRanges<View1> x1ub(x1);
00143     Iter::Ranges::Diff<GlbRanges<View0>, LubRanges<View1> > diff1(x0lb, x1ub);
00144     if ( diff1() ) {
00145       if (rm != RM_PMI)
00146         GECODE_ME_CHECK(b.zero_none(home));
00147       return home.ES_SUBSUMED(*this);
00148     }
00149 
00150     // check glb(x1) subset lub(x0)
00151     GlbRanges<View1> x1lb(x1);
00152     LubRanges<View0> x0ub(x0);
00153     Iter::Ranges::Diff<GlbRanges<View1>, LubRanges<View0> > diff2(x1lb, x0ub);
00154     if ( diff2() ) {
00155       if (rm != RM_PMI)
00156         GECODE_ME_CHECK(b.zero_none(home));
00157       return home.ES_SUBSUMED(*this);
00158     }
