00080   forceinline void
00081   ViewBase<VX,VY,VZ>::count(Space& home) {
00082     int n = x.size();
00083     for (int i=n; i--; )
00084       switch (holds(x[i],y)) {
00085       case RT_FALSE:
00086         x[i].cancel(home,*this,PC_INT_DOM); x[i]=x[--n];
00087         break;
00088       case RT_TRUE:
00089         x[i].cancel(home,*this,PC_INT_DOM); x[i]=x[--n];
00090         c--;
00091         break;
00092       case RT_MAYBE:
00093         break;
00094       default:
00095         GECODE_NEVER;
00096       }
00097     x.size(n);
00098   }
