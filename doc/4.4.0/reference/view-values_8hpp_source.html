00075   forceinline int
00076   PosValuesChoice::val(unsigned int a) const {
00077     PosMin* l = &pm[0];
00078     PosMin* r = &pm[n-1];
00079     while (true) {
00080       PosMin* m = l + (r-l)/2;
00081       if (a < m->pos) {
00082         r=m-1;
00083       } else if (a >= (m+1)->pos) {
00084         l=m+1;
00085       } else {
00086         return m->min + static_cast<int>(a - m->pos);
00087       }
00088     }
00089     GECODE_NEVER;
00090     return 0;
00091   }
