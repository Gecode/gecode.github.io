00096     if (y.assigned())
00097       return EqBoolInt<XV>::post(home,x,y.val()+c);
00098     int n = x.size();
00099     for (int i = n; i--; )
00100       if (x[i].one()) {
00101         x[i]=x[--n]; c--;
00102       } else if (x[i].zero()) {
00103         x[i]=x[--n];
00104       }
00105     x.size(n);
00106     GECODE_ME_CHECK(y.lq(home,n-c));
00107     GECODE_ME_CHECK(y.gq(home,-c));
00108     if (n == 0)
00109       return ES_OK;
00110     if (y.min()+c == n) {
00111       assert(y.assigned());
