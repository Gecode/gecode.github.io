00577   void
00578   ViewSelChoose<Choose,Merit>::ties(Space& home, ViewArray<View>& x, int s,
00579                                     int* ties, int& n) {
00580     // Consider x[s] as the so-far best view and record as tie
00581     Val b = m(home,x[s],s);
00582     int j=0; ties[j++]=s;
00583     for (int i=s+1; i<x.size(); i++)
00584       if (!x[i].assigned()) {
00585         Val mxi = m(home,x[i],i);
00586         if (c(mxi,b)) {
00587           // Found a better one, reset all ties and record
00588           j=0; ties[j++]=i; b=mxi;
00589         } else if (mxi == b) {
00590           // Found a tie, record
00591           ties[j++]=i;
00592         }
00593       }
00594     n=j;
00595     // There must be at least one tie, of course!
00596     assert(n > 0);
00597   }

00688   template<class Choose, class Merit>
00689   void
00690   ViewSelChooseTbl<Choose,Merit>::ties(Space& home, ViewArray<View>& x, int s,
00691                                        int* ties, int& n) {
00692     // Find the worst and best merit value
00693     Val w = m(home,x[s],s);
00694     Val b = w;
00695     for (int i=s+1; i<x.size(); i++)
00696       if (!x[i].assigned()) {
00697         Val mxi = m(home,x[i],i);
00698         if (c(mxi,b))
00699           b=mxi;
00700         else if (c(w,mxi))
00701           w=mxi;
00702       }
00703     // Compute tie-break limit

00705     // If the limit is not better than the worst merit, everything is a tie
