00233   SpaceStatus
00234   Space::status(StatusStatistics& stat) {
00235     SpaceStatus s = SS_FAILED;
00236     // Check whether space is failed
00237     if (failed())
00238       goto exit;
00239     assert(pc.p.active <= &pc.p.queue[PropCost::AC_MAX+1]);
00240     // Check whether space is stable but not failed
00241     if (pc.p.active >= &pc.p.queue[0]) {
00242       Propagator* p;
00243       ModEventDelta med_o;
00244       goto unstable;
00245     execute:
00246       stat.propagate++;
00247       if (p->disabled())
00248         goto put_into_idle;
00249       pc.p.ei.propagator(*p);
00250       // Keep old modification event delta
00251       med_o = p->u.med;
00252       // Clear med but leave propagator in queue
00253       p->u.med = 0;
00254       switch (p->propagate(*this,med_o)) {
00255       case ES_FAILED:
00256         // Count failure
00257         if (afc_enabled())
00258           gpi.fail(p->gpi());
00259         // Mark as failed
00260         fail(); s = SS_FAILED; goto exit;
00261       case ES_NOFIX:
00262         // Find next, if possible
00263         if (p->u.med != 0) {
00264         unstable:
00265           // There is at least one propagator in a queue
00266           do {
00267             assert(pc.p.active >= &pc.p.queue[0]);
00268             // First propagator or link back to queue
00269             ActorLink* fst = pc.p.active->next();
00270             if (pc.p.active != fst) {
00271               p = Propagator::cast(fst);
00272               goto execute;
00273             }
00274             pc.p.active--;
00275           } while (true);
00276           GECODE_NEVER;
00277         }
00278         // Fall through
00279       case ES_FIX:
00280       put_into_idle:
00281         // Clear med
00282         p->u.med = 0;
00283         // Put into idle queue
00284         p->unlink(); pl.head(p);
00285       stable_or_unstable:
00286         // There might be a propagator in the queue
00287         do {
00288           assert(pc.p.active >= &pc.p.queue[0]);
00289           // First propagator or link back to queue
00290           ActorLink* fst = pc.p.active->next();
00291           if (pc.p.active != fst) {
00292             p = Propagator::cast(fst);
00293             goto execute;
00294           }
00295         } while (--pc.p.active >= &pc.p.queue[0]);
00296         assert(pc.p.active < &pc.p.queue[0]);
00297         goto stable;
00298       case __ES_SUBSUMED:
00299         p->unlink(); rfree(p,p->u.size);
00300         goto stable_or_unstable;
00301       case __ES_PARTIAL:
00302         // Schedule propagator with specified propagator events
00303         assert(p->u.med != 0);
00304         enqueue(p);
00305         goto unstable;
00306       default:
00307         GECODE_NEVER;
00308       }
00309     }
00310   stable:
00311     pc.p.ei.other();
00312     /*
00313      * Find the next brancher that has still alternatives left
00314      *
00315      * It is important to note that branchers reporting to have no more
00316      * alternatives left cannot be deleted. They cannot be deleted
00317      * as there might be choices to be used in commit
00318      * that refer to one of these branchers. This e.g. happens when
00319      * we combine branch-and-bound search with adaptive recomputation: during
00320      * recomputation, a copy is constrained to be better than the currently
00321      * best solution, then the first half of the choices are posted,
00322      * and a fixpoint computed (for storing in the middle of the path). Then
00323      * the remaining choices are posted, and because of the additional
00324      * constraints that the space must be better than the previous solution,
00325      * the corresponding Branchers may already have no alternatives left.
00326      *
00327      * The same situation may arise due to weakly monotonic propagators.
00328      *
00329      * A brancher reporting that no more alternatives exist is exhausted.
00330      * All exhausted branchers will be left of the current pointer b_status.
00331      * Only when it is known that no more choices
00332      * can be used for commit an exhausted brancher can actually be deleted.
00333      * This becomes known when choice is called.
00334      */
00335     while (b_status != Brancher::cast(&bl))
00336       if (b_status->status(*this)) {
00337         // Brancher still has choices to generate
00338         s = SS_BRANCH; goto exit;
00339       } else {
00340         // Brancher is exhausted
00341         b_status = Brancher::cast(b_status->next());
00342       }
00343     // No brancher with alternatives left, space is solved
00344     s = SS_SOLVED;
00345   exit:
00346     stat.wmp = (wmp() > 0U);
00347     if (wmp() == 1U)
00348       wmp(0U);
00349     return s;
00350   }

00353   const Choice*
00354   Space::choice(void) {
00355     if (!stable())
00356       throw SpaceNotStable("Space::choice");
00357     if (failed() || (b_status == Brancher::cast(&bl))) {
00358       // There are no more choices to be generated
00359       // Delete all branchers
00360       Brancher* b = Brancher::cast(bl.next());
00361       while (b != Brancher::cast(&bl)) {
00362         Brancher* d = b;
00363         b = Brancher::cast(b->next());
00364         rfree(d,d->dispose(*this));
00365       }
00366       bl.init();
00367       b_status = b_commit = Brancher::cast(&bl);
00368       return NULL;
00369     }
00370     /*
00371      * The call to choice() says that no older choices
00372      * can be used. Hence, all branchers that are exhausted can be deleted.
00373      */
00374     Brancher* b = Brancher::cast(bl.next());
00375     while (b != b_status) {
00376       Brancher* d = b;
00377       b = Brancher::cast(b->next());
00378       d->unlink();
00379       rfree(d,d->dispose(*this));
00380     }
00381     // Make sure that b_commit does not point to a deleted brancher!
00382     b_commit = b_status;
00383     return b_status->choice(*this);
00384   }

00475   /*
00476    * Space cloning
00477    *
00478    * Cloning is performed in two steps:
00479    *  - The space itself is copied by the copy constructor. This
00480    *    also copies all propagators, branchers, and variables.
00481    *    The copied variables are recorded.
00482    *  - In the second step the dependency information of the recorded
00483    *    variables is updated and their forwarding information is reset.
00484    *
00485    */
00486   Space::Space(bool share, Space& s)
00487     : sm(s.sm->copy(share)),
00488       mm(sm,s.mm,s.pc.p.n_sub*sizeof(Propagator**)),
00489       gpi(s.gpi),
00490       d_fst(&Actor::sentinel),
00491       _wmp_afc(s._wmp_afc) {
00492 #ifdef GECODE_HAS_VAR_DISPOSE
00493     for (int i=0; i<AllVarConf::idx_d; i++)
00494       _vars_d[i] = NULL;
00495 #endif
00496     for (int i=0; i<AllVarConf::idx_c; i++)
00497       pc.c.vars_u[i] = NULL;
00498     pc.c.vars_noidx = NULL;
00499     pc.c.shared = NULL;
00500     pc.c.local = NULL;
00501     // Copy all propagators
00502     {
00503       ActorLink* p = &pl;
00504       ActorLink* e = &s.pl;
00505       for (ActorLink* a = e->next(); a != e; a = a->next()) {
00506         Actor* c = Actor::cast(a)->copy(*this,share);
00507         // Link copied actor
00508         p->next(ActorLink::cast(c)); ActorLink::cast(c)->prev(p);
00509         // Note that forwarding is done in the constructors
00510         p = c;
00511       }
00512       // Link last actor
00513       p->next(&pl); pl.prev(p);
00514     }
00515     // Copy all branchers
00516     {
00517       ActorLink* p = &bl;
00518       ActorLink* e = &s.bl;
00519       for (ActorLink* a = e->next(); a != e; a = a->next()) {
00520         Actor* c = Actor::cast(a)->copy(*this,share);
00521         // Link copied actor
00522         p->next(ActorLink::cast(c)); ActorLink::cast(c)->prev(p);
00523         // Note that forwarding is done in the constructors
00524         p = c;
00525       }
00526       // Link last actor
00527       p->next(&bl); bl.prev(p);
00528     }
00529     // Setup brancher pointers
00530     if (s.b_status == &s.bl) {
00531       b_status = Brancher::cast(&bl);
00532     } else {
00533       b_status = Brancher::cast(s.b_status->prev());
00534     }
00535     if (s.b_commit == &s.bl) {
00536       b_commit = Brancher::cast(&bl);
00537     } else {
00538       b_commit = Brancher::cast(s.b_commit->prev());

00544     if (failed())
00545       throw SpaceFailed("Space::clone");

00552     if (c->d_fst != &Actor::sentinel)
00553       throw SpaceNotCloned("Space::clone");

00556     {
00557       unsigned int n = static_cast<unsigned int>(d_cur - d_fst);
00558       if (n == 0) {
00559         // No actors
00560         c->d_fst = c->d_cur = c->d_lst = NULL;
00561       } else {
00562         // Leave one entry free
00563         c->d_fst = c->alloc<Actor*>(n+1);
00564         c->d_cur = c->d_fst;
00565         c->d_lst = c->d_fst+n+1;
00566         for (Actor** d_fst_iter = d_fst; d_fst_iter != d_cur; d_fst_iter++) {
00567           if ((*d_fst_iter)->prev())
00568             *(c->d_cur++) = Actor::cast((*d_fst_iter)->prev());
00569         }
00570       }
00571     }
00572 
00573     // Update variables without indexing structure
00574     VarImp<NoIdxVarImpConf>* x =
00575       static_cast<VarImp<NoIdxVarImpConf>*>(c->pc.c.vars_noidx);
00576     while (x != NULL) {
00577       VarImp<NoIdxVarImpConf>* n = x->next();
00578       x->b.base = NULL; x->u.idx[0] = 0;
00579       if (sizeof(ActorLink**) > sizeof(unsigned int))
00580         *(1+&x->u.idx[0]) = 0;
00581       x = n;
00582     }

00584     c->update(static_cast<ActorLink**>(c->mm.subscriptions()));

00607       while (c_a != &bl) {
00608         c_a->prev(p_a); p_a = c_a; c_a = c_a->next();
00609       }
00610     }
00611

00613     for (SharedHandle::Object* s = c->pc.c.shared; s != NULL; s = s->next)
00614       s->fwd = NULL;

00619

00623       c->pc.p.queue[i].init();
