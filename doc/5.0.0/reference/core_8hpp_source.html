03582   forceinline Brancher*
03583   Space::brancher(unsigned int id) {
03584     /*
03585      * Due to weakly monotonic propagators the following scenario might
03586      * occur: a brancher has been committed with all its available
03587      * choices. Then, propagation determines less information
03588      * than before and the brancher now will create new choices.
03589      * Later, during recomputation, all of these choices
03590      * can be used together, possibly interleaved with
03591      * choices for other branchers. That means all branchers
03592      * must be scanned to find the matching brancher for the choice.
03593      *
03594      * b_commit tries to optimize scanning as it is most likely that
03595      * recomputation does not generate new choices during recomputation
03596      * and hence b_commit is moved from newer to older branchers.
03597      */
03598     Brancher* b_old = b_commit;
03599     // Try whether we are lucky
03600     while (b_commit != Brancher::cast(&bl))
03601       if (id != b_commit->id())
03602         b_commit = Brancher::cast(b_commit->next());
03603       else
03604         return b_commit;
03605     if (b_commit == Brancher::cast(&bl)) {
03606       // We did not find the brancher, start at the beginning
03607       b_commit = Brancher::cast(bl.next());
03608       while (b_commit != b_old)
03609         if (id != b_commit->id())
03610           b_commit = Brancher::cast(b_commit->next());
03611         else
03612           return b_commit;
03613     }
03614     return NULL;
03615   }

03843   template<class A>
03844   forceinline void
03845   Council<A>::update(Space& home, bool share, Council<A>& c) {
03846     // Skip all disposed advisors
03847     {
03848       ActorLink* a = c.advisors;
03849       while ((a != NULL) && static_cast<A*>(a)->disposed())
03850         a = a->next();
03851       c.advisors = a;
03852     }
03853     // Are there any advisors to be cloned?
03854     if (c.advisors != NULL) {
03855       // The propagator in from-space
03856       Propagator* p_f = &static_cast<A*>(c.advisors)->propagator();
03857       // The propagator in to-space
03858       Propagator* p_t = Propagator::cast(p_f->prev());
03859       // Advisors in from-space
03860       ActorLink** a_f = &c.advisors;
03861       // Advisors in to-space
03862       A* a_t = NULL;
03863       while (*a_f != NULL) {
03864         if (static_cast<A*>(*a_f)->disposed()) {
03865           *a_f = (*a_f)->next();
03866         } else {
03867           // Run specific copying part
03868           A* a = new (home) A(home,share,*static_cast<A*>(*a_f));
03869           // Set propagator pointer
03870           a->prev(p_t);
03871           // Set forwarding pointer
03872           (*a_f)->prev(a);
03873           // Link
03874           a->next(a_t);
03875           a_t = a;
03876           a_f = (*a_f)->next_ref();
03877         }
03878       }
03879       advisors = a_t;
03880       // Enter advisor link for reset
03881       assert(p_f->u.advisors == NULL);
03882       p_f->u.advisors = c.advisors;
03883     } else {
03884       advisors = NULL;
03885     }
03886   }

04175     home.pc.p.n_sub += 1;

04217   template<class VIC>
04218   void
04219   VarImp<VIC>::resize(Space& home) {
04220     if (b.base == NULL) {
04221       assert((free_and_bits >> free_bits) == 0);
04222       // Create fresh dependency array with four entries
04223       free_and_bits += 4 << free_bits;
04224       b.base = home.alloc<ActorLink*>(4);
04225       for (int i=0; i<pc_max+1; i++)
04226         u.idx[i] = 0;
04227     } else {
04228       // Resize dependency array
04229       unsigned int n = degree();
04230       // Find out whether the area is most likely in the special area
04231       // reserved for subscriptions. If yes, just resize mildly otherwise
04232       // more agressively
04233       ActorLink** s = static_cast<ActorLink**>(home.mm.subscriptions());
04234       unsigned int m =
04235         ((s <= b.base) && (b.base < s+home.pc.p.n_sub)) ?
04236         (n+4) : ((n+1)*3>>1);
04237       ActorLink** prop = home.alloc<ActorLink*>(m);
04238       free_and_bits += (m-n) << free_bits;
04239       // Copy entries
04240       Heap::copy<ActorLink*>(prop, b.base, n);
04241       home.free<ActorLink*>(b.base,n);
04242       b.base = prop;
04243     }
04244   }

04349     unsigned int n_sub = degree();
04350     home.pc.p.n_sub -= n_sub;
04351     unsigned int n = (free_and_bits >> VIC::free_bits) + n_sub;
04352     home.free<ActorLink*>(b.base,n);

04354     b.base = NULL;
04355     // Must be 0 such that degree works
04356     entries = 0;

04359   template<class VIC>
04360   forceinline bool
04361   VarImp<VIC>::advise(Space& home, ModEvent me, Delta& d) {
04362     /*
04363      * An advisor that is executed might remove itself due to subsumption.
04364      * As entries are removed from front to back, the advisors must
04365      * be iterated in forward direction.
04366      */
04367     ActorLink** la = actorNonZero(pc_max+1);
04368     ActorLink** le = b.base+entries;
04369     if (la == le)
04370       return true;
04371     d.me = me;
04372     // An advisor that is run, might be removed during execution.
04373     // As removal is done from the back the advisors have to be executed
04374     // in inverse order.
04375     do {
04376       Advisor* a = Advisor::cast(*la);
04377       assert(!a->disposed());
04378       Propagator& p = a->propagator();
04379       switch (p.advise(home,*a,d)) {
04380       case ES_FIX:
04381         break;
04382       case ES_FAILED:
04383         if (home.afc_enabled())
04384           home.gpi.fail(p.gpi());
04385         return false;
04386       case ES_NOFIX:
04387         schedule(home,p,me);
04388         break;
04389       case ES_NOFIX_FORCE:
04390         schedule(home,p,me,true);
04391         break;
04392       case __ES_SUBSUMED:
04393       default:
04394         GECODE_NEVER;
04395       }
04396     } while (++la < le);
04397     return true;
04398   }

04400   template<class VIC>
04401   forceinline void
04402   VarImp<VIC>::update(VarImp<VIC>* x, ActorLink**& sub) {
04403     // this refers to the variable to be updated (clone)
04404     // x refers to the original
04405     // Recover from copy
04406     x->b.base = b.base;
04407     x->u.idx[0] = u.idx[0];
04408     if (pc_max > 0 && sizeof(ActorLink**) > sizeof(unsigned int))
04409       x->u.idx[1] = u.idx[1];
04410 
04411     ActorLink** f = x->b.base;
04412     unsigned int n = x->degree();
04413     ActorLink** t = sub;
04414     sub += n;
04415     b.base = t;
04416     // Set subscriptions using actor forwarding pointers
04417     while (n >= 4) {
04418       n -= 4;
04419       t[0]=f[0]->prev(); t[1]=f[1]->prev();
04420       t[2]=f[2]->prev(); t[3]=f[3]->prev();
04421       t += 4; f += 4;
04422     }
04423     if (n >= 2) {
04424       n -= 2;
04425       t[0]=f[0]->prev(); t[1]=f[1]->prev();
04426       t += 2; f += 2;
04427     }
04428     if (n > 0) {
04429       t[0]=f[0]->prev();
04430     }
04431   }
