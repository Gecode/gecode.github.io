01819   /*
01820    * \brief Relaxed assignment of variables in \a x from values in \a sx
01821    *
01822    * The variables in \a x are assigned values from the assigned variables
01823    * in the solution \a sx with a relaxation probability \a p. That is,
01824    * if \$fp=0.1\f$ approximately 10% of the variables in \a x will be
01825    * assigned a value from \a sx.
01826    *
01827    * The random numbers are generated from the generator \a r. At least
01828    * one variable will not be assigned: in case the relaxation attempt
01829    * would suggest that all variables should be assigned, a single
01830    * variable will be selected randomly to remain unassigned.
01831    *
01832    * Throws an exception of type Float::ArgumentSizeMismatch, if \a x and
01833    * \a sx are of different size.
01834    *
01835    * Throws an exception of type Float::OutOfLimits, if \a p is not between
01836    * \a 0.0 and \a 1.0.
01837    *
01838    * \ingroup TaskModeFloat
01839    */
01840   GECODE_FLOAT_EXPORT void
01841   relax(Home home, const FloatVarArgs& x, const FloatVarArgs& sx,
01842         Rnd r, double p);
