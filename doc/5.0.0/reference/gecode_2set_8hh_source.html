01602   /*
01603    * \brief Relaxed assignment of variables in \a x from values in \a sx
01604    *
01605    * The variables in \a x are assigned values from the assigned variables
01606    * in the solution \a sx with a relaxation probability \a p. That is,
01607    * if \$fp=0.1\f$ approximately 10% of the variables in \a x will be
01608    * assigned a value from \a sx.
01609    *
01610    * The random numbers are generated from the generator \a r. At least
01611    * one variable will not be assigned: in case the relaxation attempt
01612    * would suggest that all variables should be assigned, a single
01613    * variable will be selected randomly to remain unassigned.
01614    *
01615    * Throws an exception of type Set::ArgumentSizeMismatch, if \a x and
01616    * \a sx are of different size.
01617    *
01618    * Throws an exception of type Set::OutOfLimits, if \a p is not between
01619    * \a 0.0 and \a 1.0.
01620    *
01621    * \ingroup TaskModeSet
01622    */
01623   GECODE_SET_EXPORT void
01624   relax(Home home, const SetVarArgs& x, const SetVarArgs& sx,
