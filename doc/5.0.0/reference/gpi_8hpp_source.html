00192   GPI::Manager::decay(Info& c) {
00193     assert((t >= c.t) && (d != 1.0));
00194     unsigned int n = t - c.t;

00281   forceinline void
00282   GPI::dispose(void) {
00283     Support::FastMutex* m = object()->mutex;
00284     m->acquire();
00285     Object* c = object();
00286     Manager* manager = c->manager;
00287     while ((c != NULL) && (--c->use_cnt == 0)) {
00288       // Delete all blocks for c
00289       Block* b = c->cur;
00290       while (b != NULL) {
00291         Block* d = b; b=b->next;
00292         heap.rfree(d);
00293       }
00294       // Delete object
00295       Object* d = c; c = c->parent;
00296       delete d;
00297     }
00298     m->release();
00299     // All objects are deleted, so also delete mutex and manager
00300     if (c == NULL) {
00301       delete manager;
00302       delete m;
00303     }
00304   }

00356   GPI::allocate(unsigned int gid) {
00357     /*
00358      * If there is no local object, create one.
00359      *
00360      * There is no synchronization needed as only ONE space has access
00361      * to the marked pointer AND the local object.
00362      */

00364       local(new Object(object()->mutex,object()));
00365 
00366     assert(local());

00368     Object* o = object();
00369 
00370     if (o->free == 0) {
