00418   template<class View>
00419   ExecStatus
00420   Activity::Recorder<View>::propagate(Space& home, const ModEventDelta&) {
00421     // Lock activity information
00422     a.acquire();
00423     for (Advisors<Idx> as(c); as(); ++as) {
00424       int i = as.advisor().idx();
00425       if (as.advisor().marked()) {
00426         as.advisor().unmark();
00427         a.update(i);
00428         if (x[i].assigned())
00429           as.advisor().dispose(home,c);
00430       } else {
00431         assert(!x[i].assigned());
00432         a.decay(i);
00433       }
00434     }
00435     a.release();
00436     return c.empty() ? home.ES_SUBSUMED(*this) : ES_FIX;
00437   }
