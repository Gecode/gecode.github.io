00189 class PerfectSquare : public Script {
00190 protected:
00192   IntVarArray x;
00194   IntVarArray y;
00195 public:
00197   enum {
00198     PROP_REIFIED,    
00199     PROP_CUMULATIVES

00202   PerfectSquare(const SizeOptions& opt)
00203     : Script(opt),
00204       x(*this,specs[opt.size()][0],0,specs[opt.size()][1]-1),
00205       y(*this,specs[opt.size()][0],0,specs[opt.size()][1]-1) {
00206 
00207     const int* s = specs[opt.size()];
00208     int n = *s++;
00209     int w = *s++;
00210 
00211     // Restrict position according to square size
00212     for (int i=n; i--; ) {
00213       rel(*this, x[i], IRT_LQ, w-s[i]);
00214       rel(*this, y[i], IRT_LQ, w-s[i]);
00215     }
00216 
00217     IntArgs sa(n,s);
00218 
00219     // Squares do not overlap
00220     nooverlap(*this, x, sa, y, sa);
00221 
00222     /*
00223      * Capacity constraints
00224      *
00225      */
00226     switch (opt.propagation()) {
00227     case PROP_REIFIED:
00228       {
00229         for (int cx=0; cx<w; cx++) {
00230           BoolVarArgs bx(*this,n,0,1);
00231           for (int i=0; i<n; i++)
00232             dom(*this, x[i], cx-s[i]+1, cx, bx[i]);
00233           linear(*this, sa, bx, IRT_EQ, w);
00234         }
00235         for (int cy=0; cy<w; cy++) {
00236           BoolVarArgs by(*this,n,0,1);
00237           for (int i=0; i<n; i++)
00238             dom(*this, y[i], cy-s[i]+1, cy, by[i]);
00239           linear(*this, sa, by, IRT_EQ, w);
00240         }
00241       }
00242       break;
00243     case PROP_CUMULATIVES:
00244       {
00245         IntArgs m(n), dh(n);
00246         for (int i = n; i--; ) {
00247           m[i]=0; dh[i]=s[i];
00248         }
00249         IntArgs limit(1, w);
00250         {
00251           // x-direction
00252           IntVarArgs e(n);
00253           for (int i=n; i--;)
00254             e[i]=expr(*this, x[i]+dh[i]);
00255           cumulatives(*this, m, x, dh, e, dh, limit, true);
00256           cumulatives(*this, m, x, dh, e, dh, limit, false);
00257         }
00258         {
00259           // y-direction
00260           IntVarArgs e(n);
00261           for (int i=n; i--;)
00262             e[i]=expr(*this, y[i]+dh[i]);
00263           cumulatives(*this, m, y, dh, e, dh, limit, true);
00264           cumulatives(*this, m, y, dh, e, dh, limit, false);
00265         }
00266       }
00267       break;
00268     default:
00269       GECODE_NEVER;
00270     }
00271 
00272     branch(*this, x, INT_VAR_MIN_MIN(), INT_VAL_MIN());
00273     branch(*this, y, INT_VAR_MIN_MIN(), INT_VAL_MIN());
00274   }
