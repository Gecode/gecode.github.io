00054   Thread::Run::exec(void) {
00055     while (true) {
00056       // Execute runnable
00057       {
00058         Runnable* e;
00059         m.acquire();
00060         e=r; r=NULL;
00061         m.release();
00062         assert(e != NULL);
00063         e->run();
00064         if (e->todelete())
00065           delete e;
00066       }
00067       // Put into idle stack
00068       Thread::m()->acquire();
00069       n=Thread::idle; Thread::idle=this;
00070       Thread::m()->release();
00071       // Wait for next runnable
00072       e.wait();
00073     }
00074   }
