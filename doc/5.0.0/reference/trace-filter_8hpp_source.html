00281   TraceFilter::TFO::operator ()(const ExecInfo& ei) const {
00282     if (n == 0)
00283       return true;
00284     for (int i=n; i--; )
00285       if (f[i].what & (1 << ei.what())) {
00286         // Group is of the right type
00287         switch (ei.what()) {
00288         case ExecInfo::PROPAGATOR:
00289           if (f[i].g.in(ei.propagator().group()) != f[i].neg)
00290             return true;
00291           break;
00292         case ExecInfo::BRANCHER:
00293           if (f[i].g.in(ei.brancher().group()) != f[i].neg)
00294             return true;
00295           break;
00296         case ExecInfo::POST:
00297           if (f[i].g.in(ei.post()) != f[i].neg)
00298             return true;
00299           break;
00300         case ExecInfo::OTHER:
00301           return true;
00302         default:
00303           GECODE_NEVER;
00304         }
00305       }
00306     return false;
