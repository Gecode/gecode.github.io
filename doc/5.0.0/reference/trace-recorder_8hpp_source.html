00208    * Posting of tracer propagator

00211   template<class View>
00212   forceinline
00213   TraceRecorder<View>::TraceRecorder(Home home, ViewArray<View>& x,
00214                                      TraceFilter tf0, int te0,
00215                                      Tracer<View>& t0)
00216     : Propagator(home), o(home,x.size()), n(x), c(home),
00217       tf(tf0), te(te0), t(t0) {
00218     home.notice(*this, AP_DISPOSE);
00219     for (int i=n.size(); i--; ) {
00220       o[i] = TraceView(home,n[i]);
00221       if (!n[i].assigned())
00222         n[i].subscribe(home,*new (home) Idx(home,*this,c,i));
00223     }
00224     View::schedule(home,*this,ME_GEN_ASSIGNED);
00225     s.i = TraceView::slack(n[n.size()-1]);
00226     for (int i=n.size()-1; i--; )
00227       s.i += TraceView::slack(n[i]);
00228     s.p = s.i;
00229     if ((te & TE_INIT) != 0)
00230       t._init(home,*this);
00231   }

00290   template<class View>
00291   ExecStatus
00292   TraceRecorder<View>::advise(Space& home, Advisor& _a, const Delta& d) {
00293     Idx& a = static_cast<Idx&>(_a);
00294     int i = a.idx();
00295     if (((te & TE_PRUNE) != 0) && !disabled() && tf(a(home)) ) {
00296       TraceDelta td(o[i],n[i],d);
00297       t._prune(home,*this,a(home),i,td);
00298     }
00299     o[i].prune(home,n[i],d);
00300     if (n[a.idx()].assigned())
00301       a.dispose(home,c);
00302     return ES_NOFIX;
00303   }

00305   template<class View>
00306   ExecStatus
00307   TraceRecorder<View>::propagate(Space& home, const ModEventDelta&) {
00308     s.c = TraceView::slack(n[n.size()-1]);
00309     for (int i=n.size()-1; i--; )
00310       s.c += TraceView::slack(n[i]);
00311     if ((te & TE_FIX) != 0)
00312       t._fix(home,*this);
00313     s.p = s.c;
00314     if (c.empty()) {
00315       if ((te & TE_DONE) != 0)
00316         t._done(home,*this);
00317       return home.ES_SUBSUMED(*this);
00318     }
00319     return ES_FIX;
00320   }
