00065   class Bishops : public Space {
00066   public:
00067     const int n;
00068     BoolVarArray k;
00069     bool valid_pos(int i, int j) {
00070       return (i >= 0 && i < n) && (j >= 0 && j < n);
00071     }
00072     Bishops(int size)
00073       : n(size), k(*this,n*n,0,1) {
00074       Matrix<BoolVarArray> kb(k, n, n);
00075       for (int l = n; l--; ) {
00076         const int il = (n-1) - l;
00077         BoolVarArgs d1(l+1), d2(l+1), d3(l+1), d4(l+1);
00078         for (int i = 0; i <= l; ++i) {
00079           d1[i] = kb(i+il, i);
00080           d2[i] = kb(i, i+il);
00081           d3[i] = kb((n-1)-i-il, i);
00082           d4[i] = kb((n-1)-i, i+il);
00083         }
00084 
00085         linear(*this, d1, IRT_LQ, 1);
00086         linear(*this, d2, IRT_LQ, 1);
00087         linear(*this, d3, IRT_LQ, 1);
00088         linear(*this, d4, IRT_LQ, 1);
00089       }
00090 
00091       linear(*this, k, IRT_EQ, 2*n - 2);
00092       // Forced bishop placement from crowded chess model
00093       rel(*this, kb(n-1,   0), IRT_EQ, 1);
00094       rel(*this, kb(n-1, n-1), IRT_EQ, 1);
00095       branch(*this, k, BOOL_VAR_DEGREE_MAX(), BOOL_VAL_MAX());
00096     }
00097     Bishops(bool share, Bishops& s) : Space(share,s), n(s.n) {
00098       k.update(*this, share, s.k);
00099     }
00100     virtual Space* copy(bool share) {
00101       return new Bishops(share,*this);
00102     }
00103   };
00107   void init_bishops(int size) {
00108     Bishops* prob = new Bishops(size);
00109     DFS<Bishops> e(prob); IntArgs ia(size*size);
00110     delete prob;
00111 
00112     while (Bishops* s = e.next()) {
00113       for (int i = size*size; i--; )
00114         ia[i] = s->k[i].val();
00115       bishops.add(ia);
00116       delete s;
00117     }
00118 
00119     bishops.finalize();
00120   }

00213   void knight_constraints(void) {
00214     static const int kmoves[4][2] = {
00215       {-1,2}, {1,2}, {2,-1}, {2,1}
00216     };
00217     Matrix<BoolVarArray> kb(knights, n, n);
00218     for (int x = n; x--; )
00219       for (int y = n; y--; )
00220         for (int i = 4; i--; )
00221           if (valid_pos(x+kmoves[i][0], y+kmoves[i][1]))
00222             rel(*this,
00223                 kb(x, y),
00224                 BOT_AND,
00225                 kb(x+kmoves[i][0], y+kmoves[i][1]),
00226                 0);
00227   }

00237   CrowdedChess(const SizeOptions& opt)
00238     : Script(opt),
00239       n(opt.size()),
00240       s(*this, n*n, 0, PMAX-1),
00241       queens(*this, n, 0, n-1),
00242       rooks(*this, n, 0, n-1),
00243       knights(*this, n*n, 0, 1) {
00244     const int nkval = sizeof(kval)/sizeof(int);
00245     const int nn = n*n, q = n, r = n, b = (2*n)-2,
00246       k = n <= nkval ? kval[n-1] : kval[nkval-1];
00247     const int e = nn - (q + r + b + k);
00248 
00249     assert(nn == (e + q + r + b + k));
00250 
00251     Matrix<IntVarArray> m(s, n);
00252 
00253     // ***********************
00254     // Basic model
00255     // ***********************
00256 
00257     count(*this, s, E, IRT_EQ, e, opt.ipl());
00258     count(*this, s, Q, IRT_EQ, q, opt.ipl());
00259     count(*this, s, R, IRT_EQ, r, opt.ipl());
00260     count(*this, s, B, IRT_EQ, b, opt.ipl());
00261     count(*this, s, K, IRT_EQ, k, opt.ipl());
00262 
00263     // Collect rows and columns for handling rooks and queens
00264     for (int i = 0; i < n; ++i) {
00265       IntVarArgs aa = m.row(i), bb = m.col(i);
00266 
00267       count(*this, aa, Q, IRT_EQ, 1, opt.ipl());
00268       count(*this, bb, Q, IRT_EQ, 1, opt.ipl());
00269       count(*this, aa, R, IRT_EQ, 1, opt.ipl());
00270       count(*this, bb, R, IRT_EQ, 1, opt.ipl());
00271 
00272       // Connect (queens|rooks)[i] to the row it is in
00273       element(*this, aa, queens[i], Q, IPL_DOM);
00274       element(*this, aa,  rooks[i], R, IPL_DOM);
00275     }
00276 
00277     // N-queens constraints
00278     distinct(*this, queens, IPL_DOM);
00279     distinct(*this, IntArgs::create(n,0,1), queens, IPL_DOM);
00280     distinct(*this, IntArgs::create(n,0,-1), queens, IPL_DOM);
00281 
00282     // N-rooks constraints
00283     distinct(*this,  rooks, IPL_DOM);
00284 
00285     // Collect diagonals for handling queens and bishops
00286     for (int l = n; l--; ) {
00287       const int il = (n-1) - l;
00288       IntVarArgs d1(l+1), d2(l+1), d3(l+1), d4(l+1);
00289       for (int i = 0; i <= l; ++i) {
00290         d1[i] = m(i+il, i);
00291         d2[i] = m(i, i+il);
00292         d3[i] = m((n-1)-i-il, i);
00293         d4[i] = m((n-1)-i, i+il);
00294       }
00295 
00296       count(*this, d1, Q, IRT_LQ, 1, opt.ipl());
00297       count(*this, d2, Q, IRT_LQ, 1, opt.ipl());
00298       count(*this, d3, Q, IRT_LQ, 1, opt.ipl());
00299       count(*this, d4, Q, IRT_LQ, 1, opt.ipl());
00300       if (opt.propagation() == PROP_DECOMPOSE) {
00301         count(*this, d1, B, IRT_LQ, 1, opt.ipl());
00302         count(*this, d2, B, IRT_LQ, 1, opt.ipl());
00303         count(*this, d3, B, IRT_LQ, 1, opt.ipl());
00304         count(*this, d4, B, IRT_LQ, 1, opt.ipl());
00305       }
00306     }
00307     if (opt.propagation() == PROP_TUPLE_SET) {
00308       IntVarArgs b(s.size());
00309       for (int i = s.size(); i--; )
00310         b[i] = channel(*this, expr(*this, (s[i] == B)));
00311       extensional(*this, b, bishops, opt.ipl());
00312     }
00313 
00314     // Handle knigths
00315     // Connect knigths to board
00316     for(int i = n*n; i--; )
00317       knights[i] = expr(*this, (s[i] == K));
00318     knight_constraints();
00319 
00320 
00321     // ***********************
00322     // Redundant constraints
00323     // ***********************
00324 
00325     // Queens and rooks not in the same place
00326     // Faster than going through the channelled board-connection
00327     for (int i = n; i--; )
00328       rel(*this, queens[i], IRT_NQ, rooks[i]);
00329 
00330     // Place bishops in two corners (from Schimpf and Hansens solution)
00331     // Avoids some symmetries of the problem
00332     rel(*this, m(n-1,   0), IRT_EQ, B);
00333     rel(*this, m(n-1, n-1), IRT_EQ, B);
00334 
00335 
00336     // ***********************
00337     // Branching
00338     // ***********************
00339     // Place each piece in turn
00340     branch(*this, s, INT_VAR_MIN_MIN(), INT_VAL_MIN());
00341   }
