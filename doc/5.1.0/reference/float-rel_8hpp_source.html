00061   forceinline FloatRelType
00062   LinFloatRel::neg(FloatRelType frt) {
00063     switch (frt) {
00064     case FRT_EQ: return FRT_NQ;
00065     case FRT_NQ: return FRT_EQ;
00066     case FRT_LQ: return FRT_GR;
00067     case FRT_LE: return FRT_GQ;
00068     case FRT_GQ: return FRT_LE;
00069     case FRT_GR: return FRT_LQ;
00070     default: GECODE_NEVER;
00071     }
00072     return FRT_LQ;
00073   }
