00060 class Grocery : public Script {
00061 protected:
00063   IntVarArray abcd;
00065   static const int s = 711;
00067   static const int p = 711 * 100 * 100 * 100;
00068 public:
00070   Grocery(const Options& opt)
00071     : Script(opt), abcd(*this,4,0,s) {
00072     IntVar a(abcd[0]), b(abcd[1]), c(abcd[2]), d(abcd[3]);
00073 
00074     // The sum of all variables is s
00075     rel(*this, a+b+c+d == s, opt.ipl());
00076 
00077     // The product of all variables is s (corrected by scale factor)
00078     rel(*this, (a*b)*(c*d) == p, opt.ipl());
00079 
00080     // Break symmetries: order the variables
00081     rel(*this, abcd, IRT_LQ);
00082 
00083     branch(*this, abcd, INT_VAR_NONE(), INT_VAL_SPLIT_MAX());
00084   }
00085 
00087   Grocery(bool share, Grocery& s) : Script(share,s) {
00088     abcd.update(*this, share, s.abcd);
00089   }
00090 
00092   virtual Space*
00093   copy(bool share) {
00094     return new Grocery(share,*this);
00095   }
00096 
00098   virtual void
00099   print(std::ostream& os) const {
00100     os << "\t" << abcd << std::endl;
00101   }
00102 };
