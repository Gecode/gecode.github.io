00059   forceinline IntRelType
00060   LinIntRel::neg(IntRelType irt) {
00061     switch (irt) {
00062     case IRT_EQ: return IRT_NQ;
00063     case IRT_NQ: return IRT_EQ;
00064     case IRT_LQ: return IRT_GR;
00065     case IRT_LE: return IRT_GQ;
00066     case IRT_GQ: return IRT_LE;
00067     case IRT_GR: return IRT_LQ;
00068     default: GECODE_NEVER;
00069     }
00070     return IRT_LQ;
00071   }
