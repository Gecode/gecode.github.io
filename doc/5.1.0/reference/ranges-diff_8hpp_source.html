00073   template<class I, class J>
00074   forceinline void
00075   Diff<I,J>::operator ++(void) {
00076     // Precondition: mi <= ma
00077     // Task: find next mi greater than ma
00078     while (true) {
00079       if (!i()) break;
00080       mi = ma+1;
00081       ma = i.max();
00082       if (mi > i.max()) {
00083         ++i;
00084         if (!i()) break;
00085         mi = i.min();
00086         ma = i.max();
00087       }
00088       while (j() && (j.max() < mi))
00089         ++j;
00090       if (j() && (j.min() <= ma)) {
00091         // Now the interval [mi ... ma] must be shrunken
00092         // Is [mi ... ma] completely consumed?
00093         if ((mi >= j.min()) && (ma <= j.max()))
00094           continue;
00095         // Does [mi ... ma] overlap on the left?
00096         if (j.min() <= mi) {
00097           mi = j.max()+1;
00098           // Search for max!
00099           ++j;
00100           if (j() && (j.min() <= ma))
00101             ma = j.min()-1;
00102         } else {
00103           ma = j.min()-1;
00104         }
00105       }
00106       return;
00107     }
00108     finish();
00109   }
