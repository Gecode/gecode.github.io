00053   template<class VarArgs, class Post>
00054   forceinline void
00055   relax(Home home, const VarArgs& x, const VarArgs& sx, Rnd r,
00056         double p, Post& post) {
00057     if (home.failed())
00058       return;
00059     Region reg(home);
00060     // Which variables to assign
00061     Support::BitSet<Region> ax(reg, static_cast<unsigned int>(x.size()));
00062     // Select randomly with probability p to relax value
00063     for (int i=x.size(); i--; )
00064       if (r(UINT_MAX) >=
00065           static_cast<unsigned int>(p * static_cast<double>(UINT_MAX)))
00066         ax.set(static_cast<unsigned int>(i));
00067     if (ax.all())
00068       // Choose one variable uniformly and unset it
00069       ax.clear(r(static_cast<unsigned int>(x.size())));
00070     for (Iter::Values::BitSet<Support::BitSet<Region> > i(ax); i(); ++i)
00071       if (post(home, x[i.val()], sx[i.val()]) != ES_OK) {
00072         home.failed();
00073         return;
00074       }
00075   }
