00087 class Schur : public Script {
00088 protected:
00090   IntVarArray box;
00091 public:
00093   Schur(const SchurOptions& opt)
00094     : Script(opt), box(*this,opt.n,1,opt.c) {
00095     int n = opt.n;
00096 
00097     IntVarArgs triple(3);
00098 
00099     // Iterate over balls and find triples
00100     for (int i=1; i<=n; i++) {
00101       triple[0] = box[i-1];
00102       for (int j=1; i+j<=n; j++) {
00103         triple[1] = box[j-1];
00104         triple[2] = box[i+j-1];
00105         rel(*this, triple, IRT_NQ);
00106       }
00107     }
00108 
00109     // Break value symmetries
00110     precede(*this, box, IntArgs::create(opt.c, 1));
00111 
00112     branch(*this, box, INT_VAR_AFC_SIZE_MAX(opt.decay()), INT_VAL_MIN());
00113   }
00115   virtual void
00116   print(std::ostream& os) const {
00117     os << "\t" << box << std::endl;
00118   }
00119 
00121   Schur(bool share, Schur& s) : Script(share,s) {
00122     box.update(*this, share, s.box);
00123   }
00125   virtual Space*
00126   copy(bool share) {
00127     return new Schur(share,*this);
