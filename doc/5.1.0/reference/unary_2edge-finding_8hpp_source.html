00042   template<class TaskView>
00043   forceinline ExecStatus
00044   edgefinding(Space& home, TaskViewArray<TaskView>& t) {
00045     Region r(home);
00046 
00047     OmegaLambdaTree<TaskView> ol(r,t);
00048     TaskViewIter<TaskView,STO_LCT,false> q(r,t);
00049 
00050     int j = q.task();
00051     while (q.left() > 1) {
00052       if (ol.ect() > t[j].lct())
00053         return ES_FAILED;
00054       ol.shift(j);
00055       ++q;
00056       j = q.task();
00057       while (!ol.lempty() && (ol.lect() > t[j].lct())) {
00058         int i = ol.responsible();
00059         GECODE_ME_CHECK(t[i].est(home,ol.ect()));
00060         ol.lremove(i);
00061       }
00062     }
00063 
00064     return ES_OK;
00065   }

00067   template<class Task>
00068   ExecStatus
00069   edgefinding(Space& home, TaskArray<Task>& t) {
00070     TaskViewArray<typename TaskTraits<Task>::TaskViewFwd> f(t);
00071     GECODE_ES_CHECK(edgefinding(home,f));
00072     TaskViewArray<typename TaskTraits<Task>::TaskViewBwd> b(t);
00073     return edgefinding(home,b);
00074   }
