00061   void generate(int seed) {
00062     // The layout consists of 17 piles of 3 cards each
00063     layout = vector<vector<int> >(17, vector<int>(3));
00064     // Deck without the Ace of Spades
00065     vector<int> deck(51);
00066     for (int i = 51; i--; ) deck[i] = i+1;
00067     Support::RandomGenerator rnd(seed+1);
00068     std::random_shuffle(deck.begin(), deck.end(), rnd);
00069 
00070     // Place cards from deck
00071     int pos = 0;
00072     for (int i = 17; i--; )
00073       for (int j = 3; j--; )
00074         layout[i][j] = deck[pos++];
00075 
00076     // Location-information for each card
00077     layer = vector<int>(52);
00078     pile  = vector<int>(52);
00079     for (int i = 17; i--; ) {
00080       for (int j = 3; j--; ) {
00081         layer[layout[i][j]] = j;
00082         pile[ layout[i][j]] = i;
00083       }
00084     }
00085   }
00086 }

00131   BlackHole(const SizeOptions& opt)
00132     : Script(opt), x(*this, 52, 0,51), y(*this, 52, 0,51) {
00133     // Black ace at bottom
00134     rel(*this, x[0], IRT_EQ, 0);
00135 
00136     // x is order and y is placement
00137     channel(*this, x, y, opt.ipl());
00138 
00139     // The placement rules: the absolute value of the difference
00140     // between two consecutive cards is 1 or 12.
00141     if (opt.propagation() == PROPAGATION_REIFIED) {
00142       // Build table for accessing the rank of a card
00143       IntArgs modtable(52);
00144       for (int i = 0; i < 52; ++i) {
00145         modtable[i] = i%13;
00146       }
00147       for (int i = 0; i < 51; ++i) {
00148         IntVar x1(*this, 0, 12), x2(*this, 0, 12);
00149         element(*this, modtable, x[i], x1);
00150         element(*this, modtable, x[i+1], x2);
00151         const int dr[2] = {1, 12};
00152         IntVar diff(*this, IntSet(dr, 2));
00153         rel(*this, abs(x1-x2) == diff, IPL_DOM);
00154       }
00155     } else if (opt.propagation() == PROPAGATION_DFA) {
00156       // Build table for allowed tuples
00157       REG expression;
00158       for (int r = 13; r--; ) {
00159         for (int s1 = 4; s1--; ) {
00160           for (int s2 = 4; s2--; ) {
00161             for (int i = -1; i <= 1; i+=2) {
00162               REG r1 = REG(r+13*s1);
00163               REG r2 = REG((r+i+52+13*s2)%52);
00164               REG r = r1 + r2;
00165               expression |= r;
00166             }
00167           }
00168         }
00169       }
00170       DFA table(expression);
00171 
00172       for (int i = 51; i--; )
00173         extensional(*this, IntVarArgs() << x[i] << x[i+1], table);
00174 
00175     } else { // opt.propagation() == PROPAGATION_TUPLE_SET)
00176       // Build table for allowed tuples
00177       TupleSet ts(2);
00178       for (int r = 13; r--; )
00179         for (int s1 = 4; s1--; )
00180           for (int s2 = 4; s2--; )
00181             for (int i = -1; i <= 1; i+=2)
00182               ts.add(r+13*s1, (r+i+52+13*s2)%52);
00183       ts.finalize();
00184 
00185       for (int i = 51; i--; )
00186         extensional(*this, IntVarArgs() << x[i] << x[i+1], ts);
00187     }
00188 
00189     // A card must be played before the one under it.
00190     for (int i = 17; i--; )
00191       for (int j = 2; j--; )
00192         rel(*this, y[layout[i][j]] < y[layout[i][j+1]]);
00193 
00194     // Compute and break the conditional symmetries that are dependent
00195     // on the current layout.
00196     // Two cards with the same rank but different suits are symmetric
00197     // with respect to their placement in the black hole if changing
00198     // their order does not affect any other card.
00199     if (opt.symmetry() == SYMMETRY_CONDITIONAL) {
00200       // For all ranks
00201       for (int r = 13; r--; ) {
00202         // For all pairs of suits
00203         for (int s1 = 4; s1--; ) {
00204           for (int s2 = s1; s2--; ) {
00205             int c1 = 13*s1 + r,
00206               c2 = 13*s2 + r;
00207             // The ace of spades is already placed
00208             if (c1 == 0 || c2 == 0) continue;
00209             // Piles are handled by the rules of the game
00210             if (pile[c1] == pile[c2]) continue;
00211             // Fix the right order of the cards
00212             int o1 = c1, o2 = c2;
00213             if (pile[c1] > pile[c2] && layer[c2] >= layer[c1])
00214               std::swap(o1, o2);
00215             // cond is the condition for the symmetry
00216             BoolVarArgs ba;
00217             // Both cards played after the ones on top of them
00218             for (int i = 0; i < layer[o1]; ++i)
00219               ba << expr(*this, (y[layout[pile[o1]][i]] < y[o2]));
00220             for (int i = 0; i < layer[o2]; ++i)
00221               ba << expr(*this, (y[layout[pile[o2]][i]] < y[o1]));
00222             // Both cards played before the ones under them
00223             for (int i = layer[o1]+1; i < 3; ++i)
00224               ba << expr(*this, (y[o2] < y[layout[pile[o1]][i]]));
00225             for (int i = layer[o2]+1; i < 3; ++i)
00226               ba << expr(*this, (y[o1] < y[layout[pile[o2]][i]]));
00227             // Cond holds when all the above holds
00228             BoolVar cond(*this, 0, 1);
00229             rel(*this, BOT_AND, ba, cond);
00230 
00231             // If cond is fulfilled, then we can order the cards
00232             // cond -> (y[o1] < y[o2])
00233             rel(*this, !cond || (y[o1] < y[o2]));
00234           }
00235         }
00236       }
00237     }
00238 
00239     // Install custom brancher
00240     branch(*this, x, INT_VAR_NONE(), INT_VAL(&val));
00241   }
00243   static int val(const Space&, IntVar x, int) {
00244     int v = -1;
00245     int w = 4;
00246     for (IntVarValues vals(x); vals(); ++vals)
00247       if (layer[vals.val()] < w) {
00248         v = vals.val();
00249         if ((w = layer[vals.val()]) == 0)
00250           break;
00251       }
00252     assert(v >= 1 && v < 52);
00253     return v;
