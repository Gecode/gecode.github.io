03533   forceinline Brancher*
03534   Space::brancher(unsigned int id) {
03535     /*
03536      * Due to weakly monotonic propagators the following scenario might
03537      * occur: a brancher has been committed with all its available
03538      * choices. Then, propagation determines less information
03539      * than before and the brancher now will create new choices.
03540      * Later, during recomputation, all of these choices
03541      * can be used together, possibly interleaved with
03542      * choices for other branchers. That means all branchers
03543      * must be scanned to find the matching brancher for the choice.
03544      *
03545      * b_commit tries to optimize scanning as it is most likely that
03546      * recomputation does not generate new choices during recomputation
03547      * and hence b_commit is moved from newer to older branchers.
03548      */
03549     Brancher* b_old = b_commit;
03550     // Try whether we are lucky
03551     while (b_commit != Brancher::cast(&bl))
03552       if (id != b_commit->id())
03553         b_commit = Brancher::cast(b_commit->next());
03554       else
03555         return b_commit;
03556     if (b_commit == Brancher::cast(&bl)) {
03557       // We did not find the brancher, start at the beginning
03558       b_commit = Brancher::cast(bl.next());
03559       while (b_commit != b_old)
03560         if (id != b_commit->id())
03561           b_commit = Brancher::cast(b_commit->next());
03562         else
03563           return b_commit;
03564     }
03565     return NULL;
03566   }

03794   template<class A>
03795   forceinline void
03796   Council<A>::update(Space& home, Council<A>& c) {
03797     // Skip all disposed advisors
03798     {
03799       ActorLink* a = c.advisors;
03800       while ((a != NULL) && static_cast<A*>(a)->disposed())
03801         a = a->next();
03802       c.advisors = a;
03803     }
03804     // Are there any advisors to be cloned?
03805     if (c.advisors != NULL) {
03806       // The propagator in from-space
03807       Propagator* p_f = &static_cast<A*>(c.advisors)->propagator();
03808       // The propagator in to-space
03809       Propagator* p_t = Propagator::cast(p_f->prev());
03810       // Advisors in from-space
03811       ActorLink** a_f = &c.advisors;
03812       // Advisors in to-space
03813       A* a_t = NULL;
03814       while (*a_f != NULL) {
03815         if (static_cast<A*>(*a_f)->disposed()) {
03816           *a_f = (*a_f)->next();
03817         } else {
03818           // Run specific copying part
03819           A* a = new (home) A(home,*static_cast<A*>(*a_f));
03820           // Set propagator pointer
03821           a->prev(p_t);
03822           // Set forwarding pointer
03823           (*a_f)->prev(a);
03824           // Link
03825           a->next(a_t);
03826           a_t = a;
03827           a_f = (*a_f)->next_ref();
03828         }
03829       }
03830       advisors = a_t;
03831       // Enter advisor link for reset
03832       assert(p_f->u.advisors == NULL);
03833       p_f->u.advisors = c.advisors;
03834     } else {
03835       advisors = NULL;
03836     }
03837   }

04152   VarImp<VIC>::resize(Space& home) {
04153     if (b.base == NULL) {
04154       assert((free_and_bits >> free_bits) == 0);
04155       // Create fresh dependency array with four entries
04156       free_and_bits += 4 << free_bits;
04157       b.base = home.alloc<ActorLink*>(4);
04158       for (int i=0; i<pc_max+1; i++)
04159         u.idx[i] = 0;
04160     } else {
04161       // Resize dependency array
04162       unsigned int n = degree();
04163       // Find out whether the area is most likely in the special area
04164       // reserved for subscriptions. If yes, just resize mildly otherwise
04165       // more agressively
04166       ActorLink** s = static_cast<ActorLink**>(home.mm.subscriptions());
04167       unsigned int m =
04168         ((s <= b.base) && (b.base < s+home.pc.p.n_sub)) ?
04169         (n+4) : ((n+1)*3>>1);
04170       ActorLink** prop = home.alloc<ActorLink*>(m);
04171       free_and_bits += (m-n) << free_bits;
04172       // Copy entries
04173       Heap::copy<ActorLink*>(prop, b.base, n);
04174       home.free<ActorLink*>(b.base,n);
04175       b.base = prop;
04176     }

04267     // Find actor in dependency array

04269 #ifdef GECODE_AUDIT

04302   VarImp<VIC>::remove(Space& home, Advisor* a) {
04303     // Note that a might be a marked pointer
04304     // Find actor in dependency array
04305     ActorLink** f = actorNonZero(pc_max+1);
04306 #ifdef GECODE_AUDIT
04307     while (f < b.base+entries)
04308       if (*f == a)
04309         goto found;
04310       else

04312     GECODE_NEVER;
04313   found: ;

04315     while (*f != a) f++;

04333   forceinline void
04334   VarImp<VIC>::cancel(Space& home) {
04335     unsigned int n_sub = degree();
04336     home.pc.p.n_sub -= n_sub;
04337     unsigned int n = (free_and_bits >> free_bits) + n_sub;
04338     home.free<ActorLink*>(b.base,n);
04339     // Must be NULL such that cloning works
04340     b.base = NULL;
04341     // Must be 0 such that degree works
04342     entries = 0;
04343     // Must be NULL such that afc works
04344     for (PropCond pc=1; pc<pc_max+2; pc++)
04345       idx(pc) = 0;
04346     free_and_bits &= (1 << free_bits) - 1;
04347   }

04349   template<class VIC>
04350   forceinline bool
04351   VarImp<VIC>::advise(Space& home, ModEvent me, Delta& d) {
04352     /*
04353      * An advisor that is executed might remove itself due to subsumption.
04354      * As entries are removed from front to back, the advisors must
04355      * be iterated in forward direction.
04356      */
04357     ActorLink** la = actorNonZero(pc_max+1);
04358     ActorLink** le = b.base+entries;
04359     if (la == le)
04360       return true;
04361     d.me = me;
04362     // An advisor that is run, might be removed during execution.
04363     // As removal is done from the back the advisors have to be executed
04364     // in inverse order.
04365     do {
04366       Advisor* a = Advisor::cast
04367         (static_cast<ActorLink*>(Support::funmark(*la)));
04368       assert(!a->disposed());
04369       Propagator& p = a->propagator();
04370       switch (p.advise(home,*a,d)) {
04371       case ES_FIX:
04372         break;
04373       case ES_FAILED:
04374         return false;
04375       case ES_NOFIX:
04376         schedule(home,p,me);
04377         break;
04378       case ES_NOFIX_FORCE:
04379         schedule(home,p,me,true);
04380         break;
04381       case __ES_SUBSUMED:
04382       default:
04383         GECODE_NEVER;
04384       }
04385     } while (++la < le);
04386     return true;
04387   }

04390   void
04391   VarImp<VIC>::_fail(Space& home) {
04392     /*
04393      * An advisor that is executed might remove itself due to subsumption.
04394      * As entries are removed from front to back, the advisors must
04395      * be iterated in forward direction.
04396      */
04397     ActorLink** la = actorNonZero(pc_max+1);
04398     ActorLink** le = b.base+entries;
04399     if (la == le)
04400       return;
04401     // An advisor that is run, might be removed during execution.
04402     // As removal is done from the back the advisors have to be executed
04403     // in inverse order.
04404     do {
04405       if (Support::marked(*la)) {
04406         Advisor* a = Advisor::cast(static_cast<ActorLink*>
04407                                    (Support::unmark(*la)));
04408         assert(!a->disposed());
04409         Propagator& p = a->propagator();
04410         p.advise(home,*a);
04411       }
04412     } while (++la < le);
04413   }

04422   template<class VIC>
04423   forceinline void
04424   VarImp<VIC>::update(VarImp<VIC>* x, ActorLink**& sub) {
04425     // this refers to the variable to be updated (clone)
04426     // x refers to the original
04427     // Recover from copy
04428     x->b.base = b.base;
04429     x->u.idx[0] = u.idx[0];
04430     if (pc_max > 0 && sizeof(ActorLink**) > sizeof(unsigned int))
04431       x->u.idx[1] = u.idx[1];
04432 
04433     unsigned int np =
04434       static_cast<unsigned int>(x->actorNonZero(pc_max+1) - x->actor(0));
04435     unsigned int na = 
04436       static_cast<unsigned int >(x->b.base + x->entries - 
04437                                  x->actorNonZero(pc_max+1));
04438     unsigned int n  = na + np;
04439     assert(n == x->degree());
04440 
04441     ActorLink** f = x->b.base;
04442     ActorLink** t = sub;
04443 
04444     sub += n;
04445     b.base = t;
04446     // Process propagator subscriptions
04447     while (np >= 4) {
04448       ActorLink* p3 = f[3]->prev();
04449       ActorLink* p0 = f[0]->prev();
04450       ActorLink* p1 = f[1]->prev();
04451       ActorLink* p2 = f[2]->prev(); 
04452       t[0] = p0; t[1] = p1; t[2] = p2; t[3] = p3;
04453       np -= 4; t += 4; f += 4;
04454     }
04455     if (np >= 2) {
04456       ActorLink* p0 = f[0]->prev();
04457       ActorLink* p1 = f[1]->prev();
04458       t[0] = p0; t[1] = p1;
04459       np -= 2; t += 2; f += 2;
04460     }
04461     if (np > 0) {
04462       ActorLink* p0 = f[0]->prev();
04463       t[0] = p0;
04464       t += 1; f += 1;
04465     }
04466     // Process advisor subscriptions
04467     while (na >= 4) {
04468       ptrdiff_t m0, m1, m2, m3;
04469       ActorLink* p3 =
04470         static_cast<ActorLink*>(Support::ptrsplit(f[3],m3))->prev();
04471       ActorLink* p0 = 
04472         static_cast<ActorLink*>(Support::ptrsplit(f[0],m0))->prev();
04473       ActorLink* p1 =
04474         static_cast<ActorLink*>(Support::ptrsplit(f[1],m1))->prev();
04475       ActorLink* p2 =
04476         static_cast<ActorLink*>(Support::ptrsplit(f[2],m2))->prev();
04477       t[0] = static_cast<ActorLink*>(Support::ptrjoin(p0,m0));
04478       t[1] = static_cast<ActorLink*>(Support::ptrjoin(p1,m1));
04479       t[2] = static_cast<ActorLink*>(Support::ptrjoin(p2,m2));
04480       t[3] = static_cast<ActorLink*>(Support::ptrjoin(p3,m3));
04481       na -= 4; t += 4; f += 4;
04482     }
04483     if (na >= 2) {
04484       ptrdiff_t m0, m1;
04485       ActorLink* p0 = 
04486         static_cast<ActorLink*>(Support::ptrsplit(f[0],m0))->prev();
04487       ActorLink* p1 =
04488         static_cast<ActorLink*>(Support::ptrsplit(f[1],m1))->prev();
04489       t[0] = static_cast<ActorLink*>(Support::ptrjoin(p0,m0));
04490       t[1] = static_cast<ActorLink*>(Support::ptrjoin(p1,m1));
04491       na -= 2; t += 2; f += 2;
04492     }
04493     if (na > 0) {
04494       ptrdiff_t m0;
04495       ActorLink* p0 = 
04496         static_cast<ActorLink*>(Support::ptrsplit(f[0],m0))->prev();
04497       t[0] = static_cast<ActorLink*>(Support::ptrjoin(p0,m0));
04498     }
04499   }
