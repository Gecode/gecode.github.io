00058   template<class OptTask, class Cap, class PL>
00059   ExecStatus
00060   OptProp<OptTask,Cap,PL>::post(Home home, Cap c, TaskArray<OptTask>& t) {
00061     // Capacity must be nonnegative
00062     GECODE_ME_CHECK(c.gq(home, 0));
00063     // Check for overload by single task and remove excluded tasks
00064     int n=t.size(), m=0;
00065     for (int i=n; i--; ) {
00066       if (t[i].c() > c.max())
00067         GECODE_ME_CHECK(t[i].excluded(home));
00068       if (t[i].excluded())
00069         t[i]=t[--n];
00070       else if (t[i].mandatory())
00071         m++;
00072     }
00073     t.size(n);
00074     if (t.size() < 2) {
00075       if (t.size() == 1) {
00076         if (t[0].mandatory()) {
00077           GECODE_ME_CHECK(c.gq(home, t[0].c()));
00078           return ES_OK;
00079         } else if (c.min() >= t[0].c()) {
00080           return ES_OK;
00081         }
00082       } else {
00083         return ES_OK;
00084       }
00085     }
00086     if (c.assigned() && (c.val() == 1)) {
00087       TaskArray<typename TaskTraits<OptTask>::UnaryTask> mt(home,t.size());
00088       for (int i=t.size(); i--; )
00089         mt[i]=t[i];
00090       return Unary::OptProp<typename TaskTraits<OptTask>::UnaryTask,PL>
00091         ::post(home,mt);
00092     }
00093     if (m == t.size()) {
00094       TaskArray<typename TaskTraits<OptTask>::ManTask> mt(home,m);
00095       for (int i=m; i--; )
00096         mt[i].init(t[i]);
00097       return ManProp<typename TaskTraits<OptTask>::ManTask,Cap,PL>
00098         ::post(home,c,mt);
00099     }
00100     (void) new (home) OptProp<OptTask,Cap,PL>(home,c,t);
00101     return ES_OK;
00102   }

00118   template<class OptTask, class Cap, class PL>
00119   ExecStatus
00120   OptProp<OptTask,Cap,PL>::propagate(Space& home, const ModEventDelta& med) {
00121     // Did one of the Boolean views change?
00122     if (BoolView::me(med) == ME_BOOL_VAL)
00123       GECODE_ES_CHECK((purge<OptTask,PL>(home,*this,t,c)));
00124 
00125     // Only bounds changes?
00126     if (IntView::me(med) != ME_INT_DOM)
00127       GECODE_ES_CHECK(overload(home,c.max(),t));
00128 
00129     if (PL::basic)
00130       GECODE_ES_CHECK(timetabling(home,*this,c,t));
00131 
00132     if (PL::advanced) {
00133       // Partition into mandatory and optional activities
00134       int n = t.size();
00135       int i=0, j=n-1;
00136       while (true) {
00137         while ((i < n) && t[i].mandatory()) i++;
00138         while ((j >= 0) && !t[j].mandatory()) j--;
00139         if (i >= j) break;
00140         std::swap(t[i],t[j]);
00141       }
00142 
00143       if (i > 1) {
00144         // Truncate array to only contain mandatory tasks
00145         t.size(i);
00146         GECODE_ES_CHECK(edgefinding(home,c.max(),t));
00147         // Restore to also include optional tasks
00148         t.size(n);
00149       }
00150     }
00151 
00152     if (Cap::varderived() && c.assigned() && c.val()==1) {
00153       // Check that tasks do not overload resource
00154       for (int i=t.size(); i--; )
00155         if (t[i].c() > 1)
00156           GECODE_ME_CHECK(t[i].excluded(home));
00157       // Rewrite to unary resource constraint
00158       TaskArray<typename TaskTraits<OptTask>::UnaryTask> ut(home,t.size());
00159       for (int i=t.size(); i--;)
00160         ut[i]=t[i];
00161       GECODE_REWRITE(*this,
00162                      (Unary::OptProp<typename TaskTraits<OptTask>::UnaryTask,PL>
00163                       ::post(home(*this),ut)));
00164     }
00165 
00166     if (!PL::basic && c.assigned())
00167       GECODE_ES_CHECK(subsumed(home,*this,c.val(),t));
00168 
00169     return ES_NOFIX;
00170   }
