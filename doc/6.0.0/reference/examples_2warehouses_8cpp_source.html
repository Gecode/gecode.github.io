00103 template<class Script>
00104 class Warehouses : public Script {
00105 protected:
00107   IntVarArray supplier;
00109   BoolVarArray open;
00111   IntVarArray c_store;
00112 public:
00114   Warehouses(const Options& opt)
00115     : Script(opt),
00116       supplier(*this, n_stores, 0, n_warehouses-1),
00117       open(*this, n_warehouses, 0, 1),
00118       c_store(*this, n_stores) {
00119 
00120     // A warehouse is open, if it supplies to a store
00121     for (int s=0; s<n_stores; s++)
00122       element(*this, open, supplier[s], 1);
00123 
00124     // Compute cost for each warehouse
00125     for (int s=0; s<n_stores; s++) {
00126       IntArgs c(n_warehouses, c_supply[s]);
00127       c_store[s] = expr(*this, element(c, supplier[s]));
00128     }
00129 
00130     // Do not exceed capacity
00131     {
00132       IntSetArgs c(n_warehouses);
00133       for (int w=0; w<n_warehouses; w++)
00134         c[w] = IntSet(0,capacity[w]);
00135       count(*this, supplier, c, IPL_DOM);
00136     }
00137 
00138     // Branch with largest minimum regret on store cost
00139     branch(*this, c_store, INT_VAR_REGRET_MIN_MAX(), INT_VAL_MIN());
00140     // Branch by assigning a supplier to each store
00141     branch(*this, supplier, INT_VAR_NONE(), INT_VAL_MIN());
00142   }
00144   Warehouses(Warehouses& s) : Script(s) {
00145     supplier.update(*this, s.supplier);
00146     open.update(*this, s.open);
00147     c_store.update(*this, s.c_store);
00148   }
00149 };

00154 protected:
00156   IntVar c_total;
00157 public:
00159   SumCostWarehouses(const Options& opt)
00160     : Warehouses<IntMinimizeScript>(opt) {
00161     // Compute total cost
00162     c_total = expr(*this, c_fixed*sum(open) + sum(c_store));
