00209       if (neg(x1) || neg(x2)) goto rewrite_pnn;

00213       if (neg(x1) || pos(x2)) goto rewrite_nnp;
00214       if (pos(x1) || neg(x2)) goto rewrite_npn;
00215       goto prop_nxx;

00217     if (pos(x1)) {
00218       if (pos(x2)) goto rewrite_ppp;
00219       if (neg(x2)) goto rewrite_npn;
00220       goto prop_xpx;

00222     if (neg(x1)) {
00223       if (pos(x2)) goto rewrite_nnp;
00224       if (neg(x2)) goto rewrite_pnn;
00225       goto prop_xnx;
00226     }

00229     GECODE_ME_CHECK(x2.lq(home,std::max(r.mul_up(x0.max(),x1.max()),
00230                                         r.mul_up(x0.min(),x1.min()))));
00231     GECODE_ME_CHECK(x2.gq(home,std::min(r.mul_down(x0.min(),x1.max()),
00232                                         r.mul_down(x0.max(),x1.min()))));
00233 
00234     if (pos(x2)) {
00235       if (r.div_up(x2.min(),x1.min()) < x0.min())
00236         GECODE_ME_CHECK(x0.gq(home,0));

00248       assert((x0.val() == 0.0) && (x2.val() == 0.0));
00249       return home.ES_SUBSUMED(*this);
00250     }
00251 
00252     if (x1.assigned()) {
00253       assert((x1.val() == 0.0) && (x2.val() == 0.0));
00254       return home.ES_SUBSUMED(*this);

00257     return ES_NOFIX;

00265     GECODE_ME_CHECK(x2.gq(home,r.mul_down(x0.max(),x1.min())));

00319     if (same(x0,x1))
00320       return Sqr<View>::post(home,x0,x2);
00321     if (same(x0,x2))
00322       return MultZeroOne<View>::post(home,x0,x1);
00323     if (same(x1,x2))
00324       return MultZeroOne<View>::post(home,x1,x0);
00325     if (pos(x0)) {
00326       if (pos(x1) || pos(x2)) goto post_ppp;
00327       if (neg(x1) || neg(x2)) goto post_pnn;
00328     } else if (neg(x0)) {
00329       if (neg(x1) || pos(x2)) goto post_nnp;
00330       if (pos(x1) || neg(x2)) goto post_npn;
00331     } else if (pos(x1)) {
00332       if (pos(x2)) goto post_ppp;
00333       if (neg(x2)) goto post_npn;
00334     } else if (neg(x1)) {
00335       if (pos(x2)) goto post_nnp;
00336       if (neg(x2)) goto post_pnn;
00337     }
00338     {
00339       GECODE_ME_CHECK(x2.eq(home,x0.val()*x1.val()));
00340       (void) new (home) Mult<View>(home,x0,x1,x2);
00341     }
00342     return ES_OK;
