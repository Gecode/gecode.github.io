00050   bool
00051   IntSet::IntSetObject::in(int n) const {
00052     int l = 0;
00053     int r = this->n - 1;
00054 
00055     while (l <= r) {
00056       int m = l + (r - l) / 2;
00057       if ((this->r[m].min <= n) && (n <= this->r[m].max)) {
00058         return true;
00059       } else if (l == r) {
00060         return false;
00061       } else if (n < this->r[m].min) {
00062         r=m-1;
00063       } else {
00064         l=m+1;
00065       }
00066     }
00067     return false;
00068   }

00085   void
00086   IntSet::normalize(Range* r, int n) {
00087     if (n > 0) {
00088       // Sort ranges
00089       {
00090         MinInc lt_mi;
00091         Support::quicksort<Range>(r, n, lt_mi);
00092       }
00093       // Conjoin continuous ranges
00094       {
00095         int min = r[0].min;
00096         int max = r[0].max;
00097         int i = 1;
00098         int j = 0;
00099         while (i < n) {
00100           if (max+1 < r[i].min) {
00101             r[j].min = min; r[j].max = max; j++;
00102             min = r[i].min; max = r[i].max; i++;
00103           } else {
00104             max = std::max(max,r[i].max); i++;
00105           }
00106         }
00107         r[j].min = min; r[j].max = max;
00108         n=j+1;
00109       }
00110       IntSetObject* o = IntSetObject::allocate(n);
00111       unsigned int s = 0;
00112       for (int i=n; i--; ) {
00113         s += static_cast<unsigned int>(r[i].max-r[i].min+1);
00114         o->r[i]=r[i];
00115       }
00116       o->size = s;
00117       object(o);
00118     }
00119   }
