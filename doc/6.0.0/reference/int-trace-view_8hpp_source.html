00074   forceinline void
00075   IntTraceView::prune(Space& home, IntView y, const Delta& d) {
00076     if (y.range() && (dom->next() == NULL)) {
00077       dom->min(y.min()); dom->max(y.max());
00078     } else if (!y.any(d) && (y.max(d)+1 == y.min())) {
00079       // The lower bound has been adjusted
00080       if (y.min() > dom->max()) {
00081         RangeList* p = dom;
00082         RangeList* l = p->next();
00083         while ((l != NULL) && (l->max() < y.min())) {
00084           p=l; l=l->next();
00085         }
00086         dom->dispose(home,p);
00087         dom = l;
00088       }
00089       dom->min(y.min());
00090     } else if (!y.any(d) && (y.max()+1 == y.min(d))) {
00091       // upper bound has been adjusted
00092       if ((y.max() <= dom->max()) && (dom->next() == NULL)) {
00093         dom->max(y.max());
00094       } else {
00095         RangeList* p = dom;
00096         RangeList* l = p->next();
00097         while ((l != NULL) && (l->min() <= y.max())) {
00098           p=l; l=l->next();
00099         }
00100         p->max(y.max());
00101         if (p->next() != NULL)
00102           p->next()->dispose(home);
00103         p->next(NULL);
00104       }
00105     } else {
00106       // Just copy the domain
00107       ViewRanges<IntView> yr(y);
00108       RangeList::overwrite(home,dom,yr);
00109     }
00110   }
