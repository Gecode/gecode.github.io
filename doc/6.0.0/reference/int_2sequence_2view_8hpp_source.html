00307   forceinline bool
00308   ViewValSupport<View,Val,iss>::pushup(ViewArray<View>& a,Val s,int i,int q,int idx, int v) {
00309     if ( !retired() ) {
00310       int n = a.size() + 1;
00311 
00312       set(idx,y[idx]+v,q,n);
00313 
00314       if ( y[idx] > idx ) {
00315         return false;
00316       }
00317 
00318       int t = idx;
00319 
00320       // repair y on the left
00321       while ( idx > 0 && ((y[idx]-y[idx-1]>1) || ((y[idx]-y[idx-1]==1 && s_not_possible(a,s,i,idx)))) )  {
00322         if ( s_not_possible(a,s,i,idx) ) {
00323           set(idx-1,y[idx],q,n);
00324         } else {
00325           set(idx-1,y[idx]-1,q,n);
00326         }
00327         if ( y[idx-1]>idx-1 ) {
00328           return false;
00329         }
00330         idx -= 1;
00331       }
00332 
00333       idx = t;
00334 
00335       // repair y on the right
00336       while ( idx < a.size() && ((y[idx]-y[idx+1]>0) || ((y[idx]-y[idx+1]==0) && alternative_not_possible(a,s,i,idx+1))) ) {
00337         if ( alternative_not_possible(a,s,i,idx+1) ) {
00338           set(idx+1,y[idx]+1,q,n);
00339         } else {
00340           set(idx+1,y[idx],q,n);
00341         }
00342         idx += 1;
00343       }
00344     }

00346     return true;

00350   forceinline ExecStatus
00351   ViewValSupport<View,Val,iss>::advise(Space&, ViewArray<View>& a,
00352                                        Val s,int i,int q, int j,
00353                                        const Delta&) {
00354     ExecStatus status = ES_FIX;
00355     if (!retired()) {
00356       if ((j == i) && shaved(a[j],s,j)) {
00357         retire();
00358       } else {
00359         switch (takes(a[j],s)) {
00360         case TS_YES:
00361           if (y[j+1]-y[j] == 0) {
00362             if (!pushup(a,s,i,q,j+1,1)) {
00363               GECODE_ES_CHECK(schedule_conclusion(a,s,i));
00364             }
00365           }
00366           break;
00367         case TS_NO:
00368           if (y[j+1]-y[j] > 0) {
00369             if (!pushup(a,s,i,q,j,y[j+1]-y[j])) {
00370               GECODE_ES_CHECK(schedule_conclusion(a,s,i));
00371             }
00372           }
00373           break;
00374         case TS_MAYBE:
00375           break;
00376         default:
00377           GECODE_NEVER;
00378         }
00379 
00380         if ( has_potential_violation() )
00381           status = ES_NOFIX;
00382       }
00383     }
00384 
00385     return status;
00386   }

00388   template<class View,class Val,bool iss>
00389   forceinline ExecStatus
00390   ViewValSupport<View,Val,iss>::propagate(Space& home,ViewArray<View>& a,Val s,int i,int q,int l,int u) {
00391     if ( !retired() ) {
00392       if ( conlusion_scheduled() ) {
00393         return conclude(home,a,s,i);
00394       }
00395 
00396       while (has_potential_violation()) {
00397         int j = next_potential_violation();
00398         if (violated(j,q,l,u)) {
00399           int forced_to_s = values(j,q);
00400           if (forced_to_s < l) {
00401             if (!pushup(a,s,i,q,j+q,l-forced_to_s))
00402               return conclude(home,a,s,i);
00403           } else {
00404             if (!pushup(a,s,i,q,j,forced_to_s-u))
00405               return conclude(home,a,s,i);
00406           }
00407           if (violated(j,q,l,u))
00408             return conclude(home,a,s,i);
00409         }
00410       }
00411     }
00412     return ES_OK;
00413   }
