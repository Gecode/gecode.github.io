00281     // Is new domain empty?
00282     if (!i())
00283       return fail(home);
00284     assert((i.min() == 0) || (i.min() == 1));
00285     assert((i.max() == 0) || (i.max() == 1));

00292       // Assign domain to be one (domain cannot be zero)
