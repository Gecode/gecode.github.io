00201   forceinline HeapChunk*
00202   SharedMemory::alloc(size_t s, size_t l) {
00203     m.acquire();
00204     while ((heap.hc != NULL) && (heap.hc->size < l)) {
00205       heap.n_hc--;
00206       HeapChunk* hc = heap.hc;
00207       heap.hc = static_cast<HeapChunk*>(hc->next);
00208       Gecode::heap.rfree(hc);
00209     }
00210     HeapChunk* hc;
00211     if (heap.hc == NULL) {
00212       assert(heap.n_hc == 0);
00213       hc = static_cast<HeapChunk*>(Gecode::heap.ralloc(s));
00214       hc->size = s;
00215     } else {
00216       heap.n_hc--;
00217       hc = heap.hc;
00218       heap.hc = static_cast<HeapChunk*>(hc->next);
00219     }
00220     m.release();
00221     return hc;

00224   SharedMemory::free(HeapChunk* hc) {
00225     m.acquire();
00226     if (heap.n_hc == MemoryConfig::n_hc_cache) {
00227       Gecode::heap.rfree(hc);
00228     } else {
00229       heap.n_hc++;
00230       hc->next = heap.hc; heap.hc = hc;

00306   forceinline void
00307   MemoryManager::alloc_fill(SharedMemory& sm, size_t sz, bool first) {
00308     // Adjust current heap chunk size
00309     if (((requested > MemoryConfig::hcsz_inc_ratio*cur_hcsz) ||
00310          (sz > cur_hcsz)) &&
00311         (cur_hcsz < MemoryConfig::hcsz_max) &&
00312         !first) {
00313       cur_hcsz <<= 1;
00314     }
00315     // Increment the size that it caters for the initial overhead
00316     size_t overhead = sizeof(HeapChunk) - sizeof(double);
00317     sz += overhead;
00318     // Round size to next multiple of current heap chunk size
00319     size_t allocate = ((sz > cur_hcsz) ?
00320                        (((size_t) (sz / cur_hcsz)) + 1) * cur_hcsz : cur_hcsz);
00321     // Request a chunk of preferably size allocate, but at least size sz
00322     HeapChunk* hc = sm.alloc(allocate,sz);
00323     start = ptr_cast<char*>(&hc->area[0]);
00324     lsz   = hc->size - overhead;
00325     // Link heap chunk, where the first heap chunk is kept in place
00326     if (first) {
00327       requested = hc->size;
00328       hc->next = NULL; cur_hc = hc;
00329     } else {
00330       requested += hc->size;
00331       hc->next = cur_hc->next; cur_hc->next = hc;
00332     }
00333 #ifdef GECODE_MEMORY_CHECK
00334     for (char* c = start; c < (start+lsz); c++)
00335       *c = 0;
00336 #endif
00337   }

00440   template<size_t sz>
00441   void
00442   MemoryManager::fl_refill(SharedMemory& sm) {
00443     // Try to acquire memory from slack
00444     if (slack != NULL) {
00445       MemoryChunk* m = slack;
00446       slack = NULL;
00447       do {
00448         char*  block = ptr_cast<char*>(m);
00449         size_t s     = m->size;
00450         assert(s >= sz);
00451         m = m->next;
00452         fl[sz2i(sz)] = ptr_cast<FreeList*>(block);
00453         while (s >= 2*sz) {
00454           ptr_cast<FreeList*>(block)->next(ptr_cast<FreeList*>(block+sz));
00455           block += sz;
00456           s     -= sz;
00457         }
00458         ptr_cast<FreeList*>(block)->next(NULL);
00459       } while (m != NULL);
00460     } else {
00461       char* block = static_cast<char*>(alloc(sm,MemoryConfig::fl_refill*sz));
00462       fl[sz2i(sz)] = ptr_cast<FreeList*>(block);
00463       int i = MemoryConfig::fl_refill-2;
00464       do {
00465         ptr_cast<FreeList*>(block+i*sz)->next(ptr_cast<FreeList*>(block+(i+1)*sz));
00466       } while (--i >= 0);
00467       ptr_cast<FreeList*>(block+
00468                           (MemoryConfig::fl_refill-1)*sz)->next
00469         (ptr_cast<FreeList*>(NULL));
00470     }
00471   }
