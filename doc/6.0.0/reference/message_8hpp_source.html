00269       auto b1 = static_cast<uint32_t>(reinterpret_cast<uint8_t&>(*it++));
00270       auto b2 = static_cast<uint32_t>(reinterpret_cast<uint8_t&>(*it++));

00272       auto b4 = static_cast<uint32_t>(reinterpret_cast<uint8_t&>(*it++));

00274       return static_cast<int32_t>(b1 << 24 | b2 << 16 | b3 << 8 | b4);

00330     std::vector<char> serialize(void) const {
00331       std::vector<char> data;
00332       size_t dataSize = 1 + (msg.isNode() ? 4 * 8 + 1 : 0) +
00333         (msg.has_label() ? 1 + 4 + msg.label().size() : 0) +
00334         (msg.has_nogood() ? 1 + 4 + msg.nogood().size() : 0) +
00335         (msg.has_info() ? 1 + 4 + msg.info().size() : 0);
00336       data.reserve(dataSize);
00337       
00338       serializeType(data, msg.type());
00339       if (msg.isNode()) {
00340         // serialize NodeId node
00341         auto n_uid = msg.nodeUID();
00342         serialize(data, n_uid.nid);
00343         serialize(data, n_uid.rid);
00344         serialize(data, n_uid.tid);
00345         // serialize NodeId parent
00346         auto p_uid = msg.parentUID();
00347         serialize(data, p_uid.nid);
00348         serialize(data, p_uid.rid);
00349         serialize(data, p_uid.tid);
00350         // Other Data
00351         serialize(data, msg.alt());
00352         serialize(data, msg.kids());
00353         serialize(data, msg.status());
00354       }
00355       
00356       if(msg.has_version()) {
00357         serializeField(data, VERSION);
00358         serialize(data, msg.version());
00359       }
00360       if (msg.has_label()) {
00361         serializeField(data, LABEL);
00362         serialize(data, msg.label());
00363       }
00364       if (msg.has_nogood()) {
00365         serializeField(data, NOGOOD);
00366         serialize(data, msg.nogood());
00367       }
00368       if (msg.has_info()) {
00369         serializeField(data, INFO);

00371       }
00372       return data;

00375     void deserialize(char* data, size_t size) {
00376       char *end = data + size;
00377       msg.set_type(deserializeMsgType(data));
00378       if (msg.isNode()) {
00379         int32_t nid = deserializeInt(data);
00380         int32_t rid = deserializeInt(data);
00381         int32_t tid = deserializeInt(data);
00382         
00383         msg.set_nodeUID({nid, rid, tid});
00384         
00385         nid = deserializeInt(data);
00386         rid = deserializeInt(data);
00387         tid = deserializeInt(data);
00388         
00389         msg.set_parentUID({nid, rid, tid});
00390         
00391         msg.set_alt(deserializeInt(data));
00392         msg.set_kids(deserializeInt(data));
00393         msg.set_status(deserializeStatus(data));
00394       }
00395       
00396       msg.reset();
00397       
00398       while (data != end) {
00399         MessageMarshalling::Field f = deserializeField(data);
00400         switch (f) {
00401         case VERSION:
00402           msg.set_version(deserializeInt(data)); break;
00403         case LABEL:
00404           msg.set_label(deserializeString(data)); break;
00405         case NOGOOD:
00406           msg.set_nogood(deserializeString(data)); break;
00407         case INFO:
00408           msg.set_info(deserializeString(data)); break;
00409         default:
00410           break;
00411         }
00412       }
00413     }
