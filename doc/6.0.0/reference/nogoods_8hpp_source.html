00054   NoGoodsProp::NoGoodsProp(Space& home, NGL* root0)
00055     : Propagator(Home(home)), root(root0), n(0U) {
00056     // Create subscriptions
00057     root->subscribe(home,*this); n++;
00058     bool notice = root->notice();
00059     NGL* l = root->next();
00060     while ((l != NULL) && l->leaf()) {
00061       l->subscribe(home,*this); n++;
00062       notice = notice || l->notice();
00063       l = l->next();
00064     }
00065     if (l != NULL) {
00066       l->subscribe(home,*this); n++;
00067     }
00068     while (!notice && (l != NULL)) {
00069       notice = notice || l->notice();
00070       l = l->next();
00071     }
00072     if (notice)
00073       home.notice(*this,AP_DISPOSE);
00074   }

00078     : Propagator(home,p), n(p.n) {
00079     assert(p.root != NULL);
00080     NoNGL s;
00081     NGL* c = &s;
00082     for (NGL* pc = p.root; pc != NULL; pc = pc->next()) {
00083       NGL* n = pc->copy(home);
00084       n->leaf(pc->leaf());
00085       c->next(n); c=n;

00087     root = s.next();

00093   forceinline ExecStatus
00094   NoGoodsProp::post(Space& home, const Path& p) {
00095     int s = 0;
00096     int n = std::min(p.ds.entries(),static_cast<int>(p.ngdl()));
00097 
00098     unsigned long int n_nogood = 0;
00099 
00100     // Eliminate the alternatives which are not no-goods at the end
00101     while ((n > s) && (p.ds[n-1].truealt() == 0U))
00102       n--;
00103 
00104     // A sentinel element
00105     NoNGL nn;
00106     // Current no-good literal
00107     NGL* c = &nn;
00108 
00109     // Commit no-goods at the beginning
00110     while ((s < n) && (p.ds[s].truealt() > 0U))
00111       // Try whether this is a rightmost alternative
00112       if (p.ds[s].rightmost()) {
00113         // No literal needed, directly try to commit
00114         home.trycommit(*p.ds[s].choice(),p.ds[s].truealt());
00115         s++;
00116       } else {
00117         // Prune using no-good literals
00118         for (unsigned int a=0U; a<p.ds[s].truealt(); a++) {
00119           NGL* l = home.ngl(*p.ds[s].choice(),a);
00120           // Does the brancher support no-good literals?
00121           if (l == NULL)
00122             return ES_OK;
00123           GECODE_ES_CHECK(l->prune(home));
00124         }
00125         // Add literal as root if needed and stop
00126         if (NGL* l = home.ngl(*p.ds[s].choice(),p.ds[s].truealt())) {
00127           c = c->add(l,false);
00128           s++; break;
00129         }
00130       }
00131 
00132     // There are no literals
00133     if (home.failed())
00134       return ES_FAILED;
00135     if (s >= n)
00136       return ES_OK;
00137 
00138     // There must be at least two literals
00139     assert((n-s > 1) ||
00140            ((n-s == 1) && (c != &nn)));
00141 
00142     // Remember the last leaf
00143     NGL* ll = NULL;
00144 
00145     // Create literals
00146     for (int i=s; i<n; i++) {
00147       // Add leaves
00148       for (unsigned int a=0U; a<p.ds[i].truealt(); a++) {
00149         NGL* l = home.ngl(*p.ds[i].choice(),a);
00150         if (l == NULL) {
00151           // The brancher does not support no-goods
00152           if (ll == NULL)
00153             return ES_OK;
00154           ll->next(NULL);
00155           break;
00156         }
00157         c = c->add(l,true); ll = c;
00158         n_nogood++;
00159       }
00160       // Check whether to add an additional subtree
00161       if (NGL* l = home.ngl(*p.ds[i].choice(),p.ds[i].truealt())) {
00162         c = c->add(l,false);
00163       } else if (!p.ds[i].rightmost()) {
00164         // The brancher does not support no-goods
00165         if (ll == NULL)
00166           return ES_OK;
00167         ll->next(NULL);
00168         break;
00169       }
00170     }
00171 
00172     const_cast<Path&>(p).ng(n_nogood);
00173 
00174     (void) new (home) NoGoodsProp(home,nn.next());
00175     return ES_OK;
00176   }
