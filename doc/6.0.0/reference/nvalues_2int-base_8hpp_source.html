00085     // Compute positions of disjoint views
00086     int n=x.size();

00088 
00089     int i=0;
00090     while (i < n)
00091       switch (vs.compare(x[i])) {
00092       case Iter::Ranges::CS_SUBSET:
00093         // All values are already contained in vs, eliminate x[i]
00094         x[i].cancel(home, *this, PC_INT_DOM);
00095         x[i] = x[--n];

00097       case Iter::Ranges::CS_DISJOINT:
00098         dis[n_dis++] = i++;
00099         break;
00100       case Iter::Ranges::CS_NONE:
00101         i++;
00102         break;

00132   template<class VY>
00133   ExecStatus
00134   IntBase<VY>::prune_lower(Space& home, int* dis, int n_dis) {
00135     assert(n_dis > 0);
00136 
00137     // At least one more value will be needed
00138     GECODE_ME_CHECK(y.gq(home,vs.size() + 1));
00139 
00140     Region r;
00141 
00142     // Only one additional value is allowed
00143     if (y.max() == vs.size() + 1) {
00144       // Compute possible values
00145       ViewRanges<IntView>* r_dis = r.alloc<ViewRanges<IntView> >(n_dis);
00146       for (int i=n_dis; i--; )
00147         r_dis[i] = ViewRanges<IntView>(x[dis[i]]);
00148       Iter::Ranges::NaryInter iv(r, r_dis, n_dis);
00149       // Is there a common value at all?
00150       if (!iv())
00151         return ES_FAILED;
00152       ValSet::Ranges vsr(vs);
00153       Iter::Ranges::NaryUnion pv(r,iv,vsr);
00154       // Enforce common values
00155       for (int i=x.size(); i--; ) {
00156         pv.reset();
00157         GECODE_ME_CHECK(x[i].inter_r(home, pv, false));
00158       }
00159       return ES_OK;
00160     }
00161 
00162     // Compute independent set for lower bound
00163     // ovl is a bit-matrix defining whether two views overlap
00164     SymBitMatrix ovl(r,x.size());
00165     // deg[i] is the degree of x[i]
00166     int* deg = r.alloc<int>(x.size());
00167     // ovl_i[i] is an array of indices j such that x[j] overlaps with x[i]
00168     int** ovl_i = r.alloc<int*>(x.size());
00169     // n_ovl_i[i] defines how many integers are stored for ovl_i[i]
00170     int* n_ovl_i = r.alloc<int>(x.size());
00171     {
00172 #ifndef NDEBUG
00173       // Initialize all to null pointers so that things crash ;-)
00174       for (int i=x.size(); i--; )
00175         ovl_i[i] = NULL;
00176 #endif
00177       // For each i there can be at most n_dis-1 entries in ovl_i[i]
00178       int* m = r.alloc<int>(n_dis*(n_dis-1));
00179       for (int i=n_dis; i--; ) {
00180         deg[dis[i]] = 0;
00181         ovl_i[dis[i]] = m; m += n_dis-1;
00182       }
00183     }
00184 
00185     // Initialize overlap matrix by analyzing the view ranges
00186     {
00187       // Compute how many events are needed
00188       // One event for the end marker
00189       int n_re = 1;
00190       // Two events for each range
00191       for (int i=n_dis; i--; )
00192         for (ViewRanges<IntView> rx(x[dis[i]]); rx(); ++rx)
00193           n_re += 2;
00194 
00195       // Allocate and initialize events
00196       RangeEvent* re = r.alloc<RangeEvent>(n_re);
00197       {
00198         int j=0;
00199         for (int i=n_dis; i--; )
00200           for (ViewRanges<IntView> rx(x[dis[i]]); rx(); ++rx) {
00201             // Event when a range starts
00202             re[j].ret=RET_FST; re[j].val=rx.min(); re[j].view=dis[i]; j++;
00203             // Event when a range ends
00204             re[j].ret=RET_LST; re[j].val=rx.max(); re[j].view=dis[i]; j++;
00205           }
00206         // Make this the last event
00207         re[j].ret=RET_END; re[j].val=Int::Limits::infinity;
00208         assert(j+1 == n_re);
00209       }
00210       // Sort and process events
00211       Support::quicksort(re,n_re);
00212 
00213       // Current views with a range being active
00214       Support::BitSet<Region> cur(r,static_cast<unsigned int>(x.size()));
00215       // Process all events
00216       for (int i=0; true; i++)
00217         switch (re[i].ret) {
00218         case RET_FST:
00219           // Process all overlapping views
00220           for (Iter::Values::BitSet<Support::BitSet<Region> > j(cur);
00221                j(); ++j) {
00222             int di = re[i].view, dj = j.val();
00223             if (!ovl.get(di,dj))  {
00224               ovl.set(di,dj);
00225               ovl_i[di][deg[di]++] = dj;
00226               ovl_i[dj][deg[dj]++] = di;
00227             }
00228           }
00229           cur.set(static_cast<unsigned int>(re[i].view));
00230           break;
00231         case RET_LST:
00232           cur.clear(static_cast<unsigned int>(re[i].view));
00233           break;
00234         case RET_END:
00235           goto done;
00236         default:
00237           GECODE_NEVER;
00238         }
00239     done:
00240       r.free<RangeEvent>(re,n_re);
00241     }
00242 
00243     // While deg changes, n_ovl_i remains unchanged and is needed, so copy it
00244     for (int i=n_dis; i--; ) {
00245       assert(deg[dis[i]] < n_dis);
00246       n_ovl_i[dis[i]] = deg[dis[i]];
00247     }
00248 
00249     // Views in the independent set
00250     int* ind = r.alloc<int>(n_dis);
00251     int n_ind = 0;
00252 
00253     while (n_dis > 0) {
00254       int i_min = n_dis-1;
00255       int d_min = deg[dis[i_min]];
00256       unsigned int s_min = x[dis[i_min]].size();
00257 
00258       // Find view with smallest (degree,size)
00259       for (int i=n_dis-1; i--; )
00260         if ((d_min > deg[dis[i]]) ||
00261             ((d_min == deg[dis[i]]) && (s_min > x[dis[i]].size()))) {
00262           i_min = i;
00263           d_min = deg[dis[i]];
00264           s_min = x[dis[i]].size();
00265         }
00266 
00267       // i_min refers to view with smallest (degree,size)
00268       ind[n_ind++] = dis[i_min]; dis[i_min] = dis[--n_dis];
00269 
00270       // Filter out non disjoint views
00271       for (int i=n_dis; i--; )
00272         if (ovl.get(dis[i],ind[n_ind-1])) {
00273           // Update degree information
00274           for (int j=n_ovl_i[dis[i]]; j--; )
00275             deg[ovl_i[dis[i]][j]]--;
00276           // Eliminate view
00277           dis[i] = dis[--n_dis];
00278         }
00279     }
00280     // Enforce lower bound
00281     GECODE_ME_CHECK(y.gq(home,vs.size() + n_ind));
00282 
00283     // Prune, if possible
00284     if (vs.size() + n_ind == y.max()) {
00285       // Only values from the indepent set a can be taken
00286       ViewRanges<IntView>* r_ind = r.alloc<ViewRanges<IntView> >(n_ind);
00287       for (int i=n_ind; i--; )
00288         r_ind[i] = ViewRanges<IntView>(x[ind[i]]);
00289       Iter::Ranges::NaryUnion v_ind(r, r_ind, n_ind);
00290       ValSet::Ranges vsr(vs);
00291       v_ind |= vsr;
00292       for (int i=x.size(); i--; ) {
00293         v_ind.reset();
00294         GECODE_ME_CHECK(x[i].inter_r(home,v_ind,false));
00295       }
00296     }
00297     return ES_OK;
00298   }

00302   IntBase<VY>::prune_upper(Space& home, Graph& g) {
00303     if (!g) {
00304       g.init(home,vs,x);
00305     } else {
00306       g.purge();
00307       g.sync();
00308     }
00309     GECODE_ME_CHECK(y.lq(home, g.size()));
00310     if (y.min() == g.size()) {
00311       // All values must be taken on
00312       if (vs.size() + x.size() == g.size()) {
00313         // This is in fact a distinct, simplify and rewrite
00314         for (int i=x.size(); i--; ) {
00315           ValSet::Ranges vsr(vs);
00316           GECODE_ME_CHECK(x[i].minus_r(home, vsr, false));
00317         }
00318         GECODE_REWRITE(*this,Distinct::Dom<IntView>::post(home(*this),x));
00319       }
00320       if (g.mark())
