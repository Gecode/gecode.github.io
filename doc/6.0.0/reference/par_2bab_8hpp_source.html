00083   template<class Tracer>
00084   forceinline
00085   BAB<Tracer>::BAB(Space* s, const Options& o)
00086     : Engine<Tracer>(o), best(NULL) {
00087     WrapTraceRecorder::engine(o.tracer, SearchTracer::EngineType::DFS,
00088                               workers());
00089     // Create workers
00090     _worker = static_cast<Worker**>
00091       (heap.ralloc(workers() * sizeof(Worker*)));
00092     // The first worker gets the entire search tree
00093     _worker[0] = new Worker(s,*this);
00094     // All other workers start with no work
00095     for (unsigned int i=1; i<workers(); i++)
00096       _worker[i] = new Worker(NULL,*this);
00097     // Block all workers
00098     block();
00099     // Create and start threads
00100     for (unsigned int i=0; i<workers(); i++)
00101       Support::Thread::run(_worker[i]);
00102   }

00121   BAB<Tracer>::solution(Space* s) {
00122     m_search.acquire();
00123     if (best != NULL) {
00124       s->constrain(*best);
00125       if (s->status() == SS_FAILED) {
00126         delete s;
00127         m_search.release();
00128         return;
00129       } else {
00130         delete best;
00131         best = s->clone();
00132       }
00133     } else {
00134       best = s->clone();
00135     }
00136     // Announce better solutions
00137     for (unsigned int i=0; i<workers(); i++)
00138       worker(i)->better(best);
00139     bool bs = signal();
00140     solutions.push(s);
00141     if (bs)
00142       e_search.signal();
00143     m_search.release();

00148    * Worker: finding and stealing working
00149    */
00150   template<class Tracer>
00151   forceinline void
00152   BAB<Tracer>::Worker::find(void) {
00153     // Try to find new work (even if there is none)
00154     for (unsigned int i=0; i<engine().workers(); i++) {
00155       unsigned long int r_d = 0ul;
00156       typename Engine<Tracer>::Worker* wi = engine().worker(i);
00157       if (Space* s = wi->steal(r_d,wi->tracer,tracer)) {
00158         // Reset this guy
00159         m.acquire();
00160         idle = false;
00161         // Not idle but also does not have the root of the tree
00162         path.ngdl(0);
00163         d = 0;
00164         cur = s;
00165         mark = 0;
00166         if (best != NULL)
00167           cur->constrain(*best);
00168         Search::Worker::reset(r_d);
00169         m.release();
00170         return;
00171       }
00172     }
00173   }

00188   void
00189   BAB<Tracer>::constrain(const Space& b) {
00190     m_search.acquire();
00191     if (best != NULL) {
00192       best->constrain(b);
00193       if (best->status() != SS_FAILED) {
00194         m_search.release();
00195         return;
00196       }
00197       delete best;
00198     }
00199     best = b.clone();
00200     // Announce better solutions
00201     for (unsigned int i=0; i<workers(); i++)
00202       worker(i)->better(best);
00203     m_search.release();

00206   /*
00207    * Actual work
00208    */
00209   template<class Tracer>
00210   void
00211   BAB<Tracer>::Worker::run(void) {
00212     /*
00213      * The engine maintains the following invariant:
00214      *  - If the current space (cur) is not NULL, the path always points
00215      *    to exactly that space.
00216      *  - If the current space (cur) is NULL, the path always points
00217      *    to the next space (if there is any).
00218      *
00219      * This invariant is needed so that no-goods can be extracted properly
00220      * when the engine is stopped or has found a solution.
00221      *
00222      * An additional invariant maintained by the engine is:
00223      *   For all nodes stored at a depth less than mark, there
00224      *   is no guarantee of betterness. For those above the mark,
00225      *   betterness is guaranteed.
00226      *
00227      */
00228     // Peform initial delay, if not first worker
00229     if (this != engine().worker(0))
00230       Support::Thread::sleep(Config::initial_delay);
00231     // Okay, we are in business, start working
00232     while (true) {
00233       switch (engine().cmd()) {
00234       case C_WAIT:
00235         // Wait
00236         engine().wait();
00237         break;
00238       case C_TERMINATE:
00239         // Acknowledge termination request
00240         engine().ack_terminate();
00241         // Wait until termination can proceed
00242         engine().wait_terminate();
00243         // Terminate thread
00244         engine().terminated();
00245         return;
00246       case C_RESET:
00247         // Acknowledge reset request
00248         engine().ack_reset_start();
00249         // Wait until reset has been performed
00250         engine().wait_reset();
00251         // Acknowledge that reset cycle is over
00252         engine().ack_reset_stop();
00253         break;
00254       case C_WORK:
00255         // Perform exploration work
00256         {
00257           m.acquire();
00258           if (idle) {
00259             m.release();
00260             // Try to find new work
00261             find();
00262           } else if (cur != NULL) {
00263             start();
00264             if (stop(engine().opt())) {
00265               // Report stop
00266               m.release();
00267               engine().stop();
00268             } else {
00269               node++;
00270               SearchTracer::EdgeInfo ei;
00271               if (tracer) {
00272                 if (path.entries() > 0) {
00273                   typename Path<Tracer>::Edge& top = path.top();
00274                   ei.init(tracer.wid(), top.nid(), top.truealt(),
00275                           *cur, *top.choice());
00276                 } else if (*tracer.ei()) {
00277                   ei = *tracer.ei();
00278                   tracer.invalidate();
00279                 }
00280               }
00281               unsigned int nid = tracer.nid();
00282               switch (cur->status(*this)) {
00283               case SS_FAILED:
00284                 if (tracer) {
00285                   SearchTracer::NodeInfo ni(SearchTracer::NodeType::FAILED,
00286                                             tracer.wid(), nid, *cur);
00287                   tracer.node(ei,ni);
00288                 }
00289                 fail++;
00290                 delete cur;
00291                 cur = NULL;
00292                 path.next();
00293                 m.release();
00294                 break;
00295               case SS_SOLVED:
00296                 {
00297                   if (tracer) {
00298                     SearchTracer::NodeInfo ni(SearchTracer::NodeType::SOLVED,
00299                                               tracer.wid(), nid, *cur);
00300                     tracer.node(ei,ni);
00301                   }
00302                   // Deletes all pending branchers
00303                   (void) cur->choice();
00304                   Space* s = cur->clone();
00305                   delete cur;
00306                   cur = NULL;
00307                   path.next();
00308                   m.release();
00309                   engine().solution(s);
00310                 }
00311                 break;
00312               case SS_BRANCH:
00313                 {
00314                   Space* c;
00315                   if ((d == 0) || (d >= engine().opt().c_d)) {
00316                     c = cur->clone();
00317                     d = 1;
00318                   } else {
00319                     c = NULL;
00320                     d++;
00321                   }
00322                   const Choice* ch = path.push(*this,cur,c,nid);
00323                   if (tracer) {
00324                     SearchTracer::NodeInfo ni(SearchTracer::NodeType::BRANCH,
00325                                               tracer.wid(), nid, *cur, ch);
00326                     tracer.node(ei,ni);
00327                   }
00328                   cur->commit(*ch,0);
00329                   m.release();
00330                 }
00331                 break;
00332               default:
00333                 GECODE_NEVER;
00334               }
00335             }
00336           } else if (!path.empty()) {
00337             cur = path.recompute(d,engine().opt().a_d,*this,*best,mark,tracer);
00338             if (cur == NULL)
00339               path.next();
00340             m.release();
00341           } else {
00342             idle = true;
00343             path.ngdl(0);
00344             m.release();
00345             // Report that worker is idle
00346             engine().idle();
00347           }
00348         }
00349         break;
00350       default:
00351         GECODE_NEVER;
00352       }
00353     }
00354   }

00358    * Perform reset
00359    *

00361   template<class Tracer>
00362   void
00363   BAB<Tracer>::reset(Space* s) {
00364     // Grab wait lock for reset
00365     m_wait_reset.acquire();
00366     // Release workers for reset
00367     release(C_RESET);
00368     // Wait for reset cycle started
00369     e_reset_ack_start.wait();
00370     // All workers are marked as busy again
00371     delete best;
00372     best = NULL;
00373     n_busy = workers();
00374     for (unsigned int i=1; i<workers(); i++)
00375       worker(i)->reset(NULL,0);
00376     worker(0)->reset(s,opt().nogoods_limit);
00377     // Block workers again to ensure invariant
00378     block();
00379     // Release reset lock
00380     m_wait_reset.release();
00381     // Wait for reset cycle stopped
00382     e_reset_ack_stop.wait();
00383   }
