00214   Path<Tracer>::unwind(int l, Tracer& t) {
00215     assert((ds[l].space() == NULL) || ds[l].space()->failed());
00216     int n = ds.entries();
00217     if (t) {
00218       for (int i=l; i<n; i++) {
00219         Path<Tracer>::Edge& top = ds.top();
00220         unsigned int fa = (i != l) ? top.alt() + 1 : top.alt();
00221         for (unsigned int a = fa; a < top.choice()->alternatives(); a++) {
00222           SearchTracer::EdgeInfo ei(t.wid(), top.nid(), a);
00223           t.skip(ei);
00224         }
00225         if (ds.top().work())
00226           n_work--;
00227         ds.pop().dispose();
00228       }
00229     } else {
00230       for (int i=l; i<n; i++) {
00231         if (ds.top().work())
00232           n_work--;
00233         ds.pop().dispose();
00234       }
00235     }
00236     assert(ds.entries() == l);

00254   template<class Tracer>
00255   forceinline Space*
00256   Path<Tracer>::steal(Worker& stat, unsigned long int& d,
00257                       Tracer& myt, Tracer& ot) {
00258     // Find position to steal: leave sufficient work
00259     int n = ds.entries()-1;
00260     unsigned int w = 0;
00261     while (n >= 0) {
00262       if (ds[n].work())
00263         w++;
00264       if (w > Config::steal_limit) {
00265         // Okay, there is sufficient work left
00266         int l=n;
00267         // Find last copy
00268         while (ds[l].space() == NULL)
00269           l--;
00270         Space* c = ds[l].space()->clone();
00271         // Recompute, if necessary
00272         for (int i=l; i<n; i++)
00273           commit(c,i);
00274         unsigned int a = ds[n].steal();
00275         c->commit(*ds[n].choice(),a);
00276         if (!ds[n].work())
00277           n_work--;
00278         // No no-goods can be extracted above n
00279         ngdl(std::min(ngdl(),static_cast<unsigned int>(n)));
00280         d = stat.steal_depth(static_cast<unsigned long int>(n+1));
00281         if (myt && ot) {
00282           ot.ei()->init(myt.wid(),ds[n].nid(), a, *c, *ds[n].choice());
00283         }
00284         return c;
00285       }
00286       n--;
00287     }
00288     return NULL;
00289   }

00291   template<class Tracer>
00292   forceinline Space*
00293   Path<Tracer>::recompute(unsigned int& d, unsigned int a_d, Worker& stat,
00294                           Tracer& t) {
00295     assert(!ds.empty());
00296     // Recompute space according to path
00297     // Also say distance to copy (d == 0) requires immediate copying
00298 
00299     // Check for LAO
00300     if ((ds.top().space() != NULL) && ds.top().rightmost()) {
00301       Space* s = ds.top().space();
00302       s->commit(*ds.top().choice(),ds.top().alt());
00303       assert(ds.entries()-1 == lc());
00304       ds.top().space(NULL);
00305       // Mark as reusable
00306       if (static_cast<unsigned int>(ds.entries()) > ngdl())
00307         ds.top().next();
00308       d = 0;
00309       return s;
00310     }
00311     // General case for recomputation
00312     int l = lc();             // Position of last clone
00313     int n = ds.entries();     // Number of stack entries
00314     // New distance, if no adaptive recomputation
00315     d = static_cast<unsigned int>(n - l);
00316 
00317     Space* s = ds[l].space()->clone(); // Last clone
00318 
00319     if (d < a_d) {
00320       // No adaptive recomputation
00321       for (int i=l; i<n; i++)
00322         commit(s,i);
00323     } else {
00324       int m = l + static_cast<int>(d >> 1); // Middle between copy and top
00325       int i = l; // To iterate over all entries
00326       // Recompute up to middle
00327       for (; i<m; i++ )
00328         commit(s,i);
00329       // Skip over all rightmost branches
00330       for (; (i<n) && ds[i].rightmost(); i++)
00331         commit(s,i);
00332       // Is there any point to make a copy?
00333       if (i<n-1) {
00334         // Propagate to fixpoint
00335         SpaceStatus ss = s->status(stat);
00336         /*
00337          * Again, the space might already propagate to failure (due to
00338          * weakly monotonic propagators).
00339          */
00340         if (ss == SS_FAILED) {
00341           // s must be deleted as it is not on the stack
00342           delete s;
00343           stat.fail++;
00344           unwind(i,t);
00345           return NULL;
00346         }
00347         ds[i].space(s->clone());
00348         d = static_cast<unsigned int>(n-i);
00349       }
00350       // Finally do the remaining commits
00351       for (; i<n; i++)
00352         commit(s,i);
00353     }
00354     return s;
00355   }

00358   forceinline Space*
00359   Path<Tracer>::recompute(unsigned int& d, unsigned int a_d, Worker& stat,
00360                           const Space& best, int& mark,
00361                           Tracer& t) {
00362     assert(!ds.empty());

00364     // Also say distance to copy (d == 0) requires immediate copying
00365 
00366     // Check for LAO
00367     if ((ds.top().space() != NULL) && ds.top().rightmost()) {
00368       Space* s = ds.top().space();
00369       s->commit(*ds.top().choice(),ds.top().alt());
00370       assert(ds.entries()-1 == lc());
00371       if (mark > ds.entries()-1) {
00372         mark = ds.entries()-1;
00373         s->constrain(best);
00374       }
00375       ds.top().space(NULL);
00376       // Mark as reusable
00377       if (static_cast<unsigned int>(ds.entries()) > ngdl())
00378         ds.top().next();
00379       d = 0;
00380       return s;
00381     }
00382     // General case for recomputation
00383     int l = lc();             // Position of last clone
00384     int n = ds.entries();     // Number of stack entries
00385     // New distance, if no adaptive recomputation
00386     d = static_cast<unsigned int>(n - l);
00387 
00388     Space* s = ds[l].space(); // Last clone
00389 
00390     if (l < mark) {
00391       mark = l;
00392       s->constrain(best);
00393       // The space on the stack could be failed now as an additional
00394       // constraint might have been added.
00395       if (s->status(stat) == SS_FAILED) {
00396         // s does not need deletion as it is on the stack (unwind does this)
00397         stat.fail++;
00398         unwind(l,t);
00399         return NULL;
00400       }
00401       // It is important to replace the space on the stack with the
00402       // copy: a copy might be much smaller due to flushed caches
00403       // of propagators
00404       Space* c = s->clone();
00405       ds[l].space(c);
00406     } else {
00407       s = s->clone();
00408     }
00409 
00410     if (d < a_d) {
00411       // No adaptive recomputation
00412       for (int i=l; i<n; i++)
00413         commit(s,i);
00414     } else {
00415       int m = l + static_cast<int>(d >> 1); // Middle between copy and top
00416       int i = l;            // To iterate over all entries
00417       // Recompute up to middle
00418       for (; i<m; i++ )
00419         commit(s,i);
00420       // Skip over all rightmost branches
00421       for (; (i<n) && ds[i].rightmost(); i++)
00422         commit(s,i);
00423       // Is there any point to make a copy?
00424       if (i<n-1) {
00425         // Propagate to fixpoint
00426         SpaceStatus ss = s->status(stat);
00427         /*
00428          * Again, the space might already propagate to failure
00429          *
00430          * This can be for two reasons:
00431          *  - constrain is true, so we fail
00432          *  - the space has weakly monotonic propagators
00433          */

00435           // s must be deleted as it is not on the stack
00436           delete s;
00437           stat.fail++;
00438           unwind(i,t);
00439           return NULL;
