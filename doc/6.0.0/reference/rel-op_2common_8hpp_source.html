00312   partitionNCard(Space& home,

00314                  GLBndSet& unionOfDets) {

00316     unsigned int cardMaxSum=unionOfDets.size();
00317     int xsize = x.size();
00318     for (int i=xsize; i--; ) {
00319       cardMinSum+=x[i].cardMin();
00320       if (cardMinSum < x[i].cardMin()) {
00321         //sum of mins overflows: fail the space.
00322         GECODE_ME_CHECK(ME_SET_FAILED);
00323       }
00324     }
00325     GECODE_ME_CHECK_MODIFIED(modified, y.cardMin(home,cardMinSum));
00326     for (int i=xsize; i--; ) {
00327       cardMaxSum+=x[i].cardMax();
00328       if (cardMaxSum < x[i].cardMax()) {
00329         //sum of maxes overflows: no useful information to tell.
00330         goto overflow;
00331       }
00332     }
00333     GECODE_ME_CHECK_MODIFIED(modified, y.cardMax(home,cardMaxSum));
00334 
00335     if (x.size() == 0)
00336       return ES_NOFIX;
00337 
00338   overflow:
00339 
00340     //Cardinality of each x[i] limited by cardinality of y minus all x[j]s:
00341 
00342     {
00343       Region r;
00344       unsigned int* rightMinSum = r.alloc<unsigned int>(xsize);
00345       unsigned int* rightMaxSum = r.alloc<unsigned int>(xsize);
00346       rightMinSum[xsize-1]=0;
00347       rightMaxSum[xsize-1]=0;
00348 
00349       for (int i=x.size()-1;i--;) {
00350         rightMaxSum[i] = rightMaxSum[i+1] + x[i+1].cardMax();
00351         if (rightMaxSum[i] < rightMaxSum[i+1]) {
00352           //overflow, fill the rest of the array.
00353           for (int j=i; j>0;j--) {
00354             rightMaxSum[j]=Limits::card;
00355           }
00356           break;
00357         }
00358       }
00359       for (int i=x.size()-1;i--;) {
00360         rightMinSum[i] = rightMinSum[i+1] + x[i+1].cardMin();
00361         if (rightMinSum[i] < rightMinSum[i+1]) {
00362           //overflow, fail the space
00363           GECODE_ME_CHECK(ME_SET_FAILED);
00364         }
00365       }
00366       unsigned int leftMinAcc=unionOfDets.size();
00367       unsigned int leftMaxAcc=unionOfDets.size();
00368 
00369       for (int i=0; i<xsize;i++) {
00370         unsigned int maxSum = leftMaxAcc+rightMaxSum[i];
00371         unsigned int minSum = leftMinAcc+rightMinSum[i];
00372         //If maxSum did not overflow and is less than y.cardMin:
00373         if (maxSum >= leftMaxAcc &&  maxSum < y.cardMin()) {
00374           GECODE_ME_CHECK_MODIFIED(modified, x[i].cardMin(home,y.cardMin()-maxSum));

00379           GECODE_ME_CHECK(ME_SET_FAILED);
00380         }
