00144   template<class Tracer>
00145   forceinline Space*
00146   Probe<Tracer>::next(const Options& opt) {
00147     start();
00148     while (true) {
00149       if (cur == NULL) {
00150       backtrack:
00151         if (ds.empty())
00152           return NULL;
00153         if (stop(opt))
00154           return NULL;
00155         unsigned int a = ds.top().alt();
00156         const Choice* ch = ds.top().choice();
00157         unsigned int nid = ds.top().nid();
00158         if (a == 0) {
00159           cur = ds.pop().space();
00160           if (tracer)
00161             tracer.ei()->init(tracer.wid(), nid, 0, *cur, *ch);
00162           cur->commit(*ch,0);
00163           delete ch;
00164         } else {
00165           ds.top().next();
00166           cur = ds.top().space()->clone();
00167           if (tracer)
00168             tracer.ei()->init(tracer.wid(), nid, a, *cur, *ch);
00169           cur->commit(*ch,a);
00170         }
00171         node++;
00172         d++;
00173       }
00174     check_discrepancy:
00175       if (d == 0) {
00176         Space* s = cur;
00177         while (s->status(*this) == SS_BRANCH) {
00178           if (stop(opt)) {
00179             cur = s;
00180             return NULL;
00181           }
00182           const Choice* ch = s->choice();
00183           if (ch->alternatives() > 1)
00184             exhausted = false;
00185           if (tracer) {
00186             unsigned int nid = tracer.nid();
00187             SearchTracer::NodeInfo ni(SearchTracer::NodeType::BRANCH,
00188                                       tracer.wid(), nid, *s, ch);
00189             tracer.node(*tracer.ei(),ni);
00190             if (tracer)
00191               tracer.ei()->init(tracer.wid(), nid, 0, *cur, *ch);
00192           }
00193           s->commit(*ch,0);
00194           node++;
00195           delete ch;
00196         }
00197         cur = NULL;
00198         if (s->failed()) {
00199           if (tracer) {
00200             SearchTracer::NodeInfo ni(SearchTracer::NodeType::FAILED,
00201                                       tracer.wid(), tracer.nid(), *s);
00202             tracer.node(*tracer.ei(),ni);
00203           }
00204           fail++;
00205           delete s;
00206           goto backtrack;
00207         } else {
00208           if (tracer) {
00209             SearchTracer::NodeInfo ni(SearchTracer::NodeType::SOLVED,
00210                                       tracer.wid(), tracer.nid(), *s);
00211             tracer.node(*tracer.ei(),ni);
00212           }
00213           // Deletes all pending branchings
00214           (void) s->choice();
00215           return s;
00216         }
00217       } else {
00218         node++;
00219         switch (cur->status(*this)) {
00220         case SS_FAILED:
00221           if (tracer) {
00222             SearchTracer::NodeInfo ni(SearchTracer::NodeType::FAILED,
00223                                       tracer.wid(), tracer.nid(), *cur);
00224             tracer.node(*tracer.ei(),ni);
00225           }
00226           fail++;
00227           delete cur;
00228           cur = NULL;
00229           goto backtrack;
00230         case SS_SOLVED:
00231           if (tracer) {
00232             tracer.skip(*tracer.ei());
00233           }
00234           delete cur;
00235           cur = NULL;
00236           goto backtrack;
00237         case SS_BRANCH:
00238           {
00239             const Choice* ch = cur->choice();
00240             unsigned int alt = ch->alternatives();
00241             unsigned int nid = tracer.nid();
00242             if (tracer) {
00243               SearchTracer::NodeInfo ni(SearchTracer::NodeType::BRANCH,
00244                                         tracer.wid(), nid, *cur, ch);
00245               tracer.node(*tracer.ei(),ni);
00246             }
00247             if (alt > 1) {
00248               if (d < alt-1)
00249                 exhausted = false;
00250               unsigned int d_a = (d >= alt-1) ? alt-1 : d;
00251               Space* cc = cur->clone();
00252               Node sn(cc,ch,d_a-1,nid);
00253               ds.push(sn);
00254               stack_depth(static_cast<unsigned long int>(ds.entries()));
00255               if (tracer)
00256                 tracer.ei()->init(tracer.wid(), nid, d_a, *cur, *ch);
00257               cur->commit(*ch,d_a);
00258               d -= d_a;
00259             } else {
00260               if (tracer)
00261                 tracer.ei()->init(tracer.wid(), nid, 0, *cur, *ch);
00262               cur->commit(*ch,0);
00263               node++;
00264               delete ch;
00265             }
00266             goto check_discrepancy;
00267           }
00268         default: GECODE_NEVER;
00269         }
00270       }
00271     }
00272   }

00276   LDS<Tracer>::LDS(Space* s, const Options& o)
00277     : opt(o), e(opt), root(NULL), d(0) {

00291   template<class Tracer>
00292   Space*
00293   LDS<Tracer>::next(void) {
00294     while (true) {
00295       Space* s = e.next(opt);
00296       if (s != NULL)
00297         return s;
00298       if (((s == NULL) && e.stopped()) || (++d > opt.d_l) || e.done())
00299         break;
00300       if (d == opt.d_l) {
00301         if (root != NULL)
00302           e.reset(root,d);
00303         root = NULL;
00304       } else if (root != NULL) {
00305         e.reset(root->clone(),d);
00306       }
00307     }
00308     return NULL;
00309   }
