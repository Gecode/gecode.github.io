00199   forceinline void
00200   Path<Tracer>::unwind(int l, Tracer& t) {
00201     assert((ds[l].space() == NULL) || ds[l].space()->failed());
00202     int n = ds.entries();
00203     if (t) {
00204       for (int i=l; i<n; i++) {
00205         Path<Tracer>::Edge& top = ds.top();
00206         unsigned int fa = (i != l) ? top.alt() + 1 : top.alt();
00207         for (unsigned int a = fa; a < top.choice()->alternatives(); a++) {
00208           SearchTracer::EdgeInfo ei(t.wid(),top.nid(),a);
00209           t.skip(ei);
00210         }
00211         ds.pop().dispose();
00212       }
00213     } else {
00214       for (int i=l; i<n; i++)
00215         ds.pop().dispose();
00216     }
00217     assert(ds.entries() == l);
00218   }

00227   template<class Tracer>
00228   forceinline Space*
00229   Path<Tracer>::recompute(unsigned int& d, unsigned int a_d, Worker& stat,
00230                           Tracer& t) {
00231     assert(!ds.empty());
00232     // Recompute space according to path
00233     // Also say distance to copy (d == 0) requires immediate copying
00234 
00235     // Check for LAO
00236     if ((ds.top().space() != NULL) && ds.top().rightmost()) {
00237       Space* s = ds.top().space();
00238       s->commit(*ds.top().choice(),ds.top().alt());
00239       assert(ds.entries()-1 == lc());
00240       ds.top().space(NULL);
00241       // Mark as reusable
00242       if (static_cast<unsigned int>(ds.entries()) > ngdl())
00243         ds.top().next();
00244       d = 0;
00245       return s;
00246     }
00247     // General case for recomputation
00248     int l = lc();             // Position of last clone
00249     int n = ds.entries();     // Number of stack entries
00250     // New distance, if no adaptive recomputation
00251     d = static_cast<unsigned int>(n - l);
00252 
00253     Space* s = ds[l].space()->clone(); // Last clone
00254 
00255     if (d < a_d) {
00256       // No adaptive recomputation
00257       for (int i=l; i<n; i++)
00258         commit(s,i);
00259     } else {
00260       int m = l + static_cast<int>(d >> 1); // Middle between copy and top
00261       int i = l; // To iterate over all entries
00262       // Recompute up to middle
00263       for (; i<m; i++ )
00264         commit(s,i);
00265       // Skip over all rightmost branches
00266       for (; (i<n) && ds[i].rightmost(); i++)
00267         commit(s,i);
00268       // Is there any point to make a copy?
00269       if (i<n-1) {
00270         // Propagate to fixpoint
00271         SpaceStatus ss = s->status(stat);
00272         /*
00273          * Again, the space might already propagate to failure (due to
00274          * weakly monotonic propagators).
00275          */
00276         if (ss == SS_FAILED) {
00277           // s must be deleted as it is not on the stack
00278           delete s;
00279           stat.fail++;
00280           unwind(i,t);
00281           return NULL;
00282         }
00283         ds[i].space(s->clone());
00284         d = static_cast<unsigned int>(n-i);
00285       }
00286       // Finally do the remaining commits
00287       for (; i<n; i++)
00288         commit(s,i);
00289     }
00290     return s;
00291   }

00295   Path<Tracer>::recompute(unsigned int& d, unsigned int a_d, Worker& stat,
00296                           const Space& best, int& mark,

00310       }
00311       ds.top().space(NULL);
00312       // Mark as reusable
00313       if (static_cast<unsigned int>(ds.entries()) > ngdl())
00314         ds.top().next();

00316       return s;
00317     }

00319     int l = lc();             // Position of last clone
00320     int n = ds.entries();     // Number of stack entries
00321     // New distance, if no adaptive recomputation
00322     d = static_cast<unsigned int>(n - l);
00323 
00324     Space* s = ds[l].space(); // Last clone
00325 
00326     if (l < mark) {
00327       mark = l;
00328       s->constrain(best);
00329       // The space on the stack could be failed now as an additional
00330       // constraint might have been added.
00331       if (s->status(stat) == SS_FAILED) {
00332         // s does not need deletion as it is on the stack (unwind does this)
00333         stat.fail++;
00334         unwind(l,t);
00335         return NULL;
00336       }
00337       // It is important to replace the space on the stack with the
00338       // copy: a copy might be much smaller due to flushed caches
00339       // of propagators
00340       Space* c = s->clone();
00341       ds[l].space(c);
00342     } else {
00343       s = s->clone();
00344     }
00345 
00346     if (d < a_d) {
00347       // No adaptive recomputation
00348       for (int i=l; i<n; i++)
00349         commit(s,i);
00350     } else {

00353       // Recompute up to middle

00355         commit(s,i);
00356       // Skip over all rightmost branches
00357       for (; (i<n) && ds[i].rightmost(); i++)

00359       // Is there any point to make a copy?
00360       if (i<n-1) {

00363         /*

00366          * This can be for two reasons:
00367          *  - constrain is true, so we fail
