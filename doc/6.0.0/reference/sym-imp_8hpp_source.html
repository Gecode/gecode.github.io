00069   VariableSymmetryImp<View>
00070   ::VariableSymmetryImp(Space& home, int* _indices, unsigned int n)
00071     : indices(home, 0, 0) {
00072     // Find minimum and maximum value in _indices: the minimum is the
00073     // offset, and the maximum dictates how large the bitset needs to
00074     // be.
00075     int maximum = _indices[0];
00076     int minimum = _indices[0];
00077     for (unsigned int i = 1 ; i < n ; i++) {
00078       if (_indices[i] > maximum) maximum = _indices[i];
00079       if (_indices[i] < minimum) minimum = _indices[i];
00080     }
00081     indices.resize(home, maximum-minimum+1, minimum);
00082 
00083     // Set the bits for the included indices.
00084     for (unsigned int i = 0 ; i < n ; i++) {
00085       indices.set(_indices[i]);
00086     }
00087   }

00183   VariableSequenceSymmetryImp<View>
00184   ::VariableSequenceSymmetryImp(Space& home, int* _indices, unsigned int n,
00185                                 unsigned int seqsize)
00186     : n_indices(n), seq_size(seqsize), n_seqs(n/seqsize) {
00187     indices = home.alloc<unsigned int>(n_indices);
00188     unsigned int max_index = _indices[0];
00189     for (unsigned int i = 0 ; i < n_indices ; i++) {
00190       indices[i] = _indices[i];
00191       if (indices[i] > max_index)
00192         max_index = indices[i];
00193     }
00194 
00195     lookup_size = max_index+1;
00196     lookup = home.alloc<int>(lookup_size);
00197     for (unsigned int i = 0 ; i < lookup_size ; i++)
00198       lookup[i] = -1;
00199     for (unsigned int i = 0 ; i < n_indices ; i++) {
00200       if (lookup[indices[i]] == -1)
00201         lookup[indices[i]] = i;
00202     }
00203   }

00227   template <class View>
00228   ArgArray<Literal>
00229   VariableSequenceSymmetryImp<View>
00230   ::symmetric(Literal l, const ViewArray<View>& x) const {
00231     Support::DynamicStack<Literal, Heap> s(heap);
00232     if (l._variable < (int)lookup_size) {
00233       int posIt = lookup[l._variable];
00234       if (posIt == -1) {
00235         return dynamicStackToArgArray(s);
00236       }
00237       unsigned int seqNum = posIt / seq_size;
00238       unsigned int seqPos = posIt % seq_size;
00239       for (unsigned int seq = 0 ; seq < n_seqs ; seq++) {
00240         if (seq == seqNum) {
00241           continue;
00242         }
00243         if (x[getVal(seq, seqPos)].assigned()) {
00244           continue;
00245         }
00246         bool active = true;
00247         const unsigned int *firstSeq = &indices[seqNum*seq_size];
00248         const unsigned int *secondSeq = &indices[seq*seq_size];
00249         for (unsigned int i = 0 ; i < seq_size ; i++) {
00250           const View& xv = x[firstSeq[i]];
00251           const View& yv = x[secondSeq[i]];
00252           if ((!xv.assigned() && !yv.assigned())
00253               || (xv.assigned() && yv.assigned() && xv.val() == yv.val())) {
00254             continue;
00255           } else {
00256             active = false;
00257             break;
00258           }
00259         }
00260 
00261         if (active) {
00262           s.push(Literal(secondSeq[seqPos], l._value));
00263         }
00264       }
00265     }
00266     return dynamicStackToArgArray(s);
00267   }
