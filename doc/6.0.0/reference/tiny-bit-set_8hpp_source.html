00049    forceinline
00050    TinyBitSet<sz>::TinyBitSet(Space&, unsigned int n) {
00051      assert(n <= sz);
00053      for (unsigned int i=0U; i < n; i++)
00054        bits[i].init(true);
00056      for (unsigned int i=n; i < sz; i++)
00057        bits[i].init(false);

00072    template<class IndexType>
00073    forceinline
00074    TinyBitSet<sz>::TinyBitSet(Space&, const BitSet<IndexType>& sbs) {
00075      assert(sz == sbs.width());
00076      assert(!sbs.empty());
00077      for (unsigned int i = sz; i--; )
00078        bits[i].init(false);
00079      for (unsigned int i = sbs.words(); i--; ) {
00080        bits[sbs.index[i]] = sbs.bits[i];
00081      }
00082      assert(!empty());
00083    }
