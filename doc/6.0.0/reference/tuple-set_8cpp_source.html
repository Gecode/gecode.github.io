00109     assert(!finalized());
00110     // Mark as finalized
00111     n_free = -1;
00112 
00113     // Initialization
00114     if (n_tuples == 0) {
00115       delete td; td=nullptr;
00116       return;
00117     }
00118 
00119     // Compact and copy data
00120     Region r;
00121     // Set up tuple pointers
00122     Tuple* tuple = r.alloc<Tuple>(n_tuples);
00123     {
00124       for (int t=n_tuples; t--; )
00125         tuple[t] = td + t*arity;
00126       TupleCompare tc(arity);
00127       Support::quicksort(tuple, n_tuples, tc);
00128       // Remove duplicates
00129       int j=1;
00130       for (int t=1; t<n_tuples; t++) {
00131         for (int a=arity; a--; )
00132           if (tuple[t-1][a] != tuple[t][a])
00133             goto notsame;
00134         goto same;
00135       notsame: ;
00136         tuple[j++] = tuple[t];
00137       same: ;
00138       }
00139       assert(j <= n_tuples);
00140       n_tuples=j;
00141       // Initialize hash key
00142       key = static_cast<std::size_t>(n_tuples);
00143       cmb_hash(key, arity);
00144       // Copy into now possibly smaller area
00145       int* new_td = heap.alloc<int>(n_tuples*arity);
00146       for (int t=n_tuples; t--; ) {
00147         for (int a=arity; a--; ) {
00148           new_td[t*arity+a] = tuple[t][a];
00149           cmb_hash(key,tuple[t][a]);
00150         }
00151         tuple[t] = new_td + t*arity;
00152       }
00153       heap.rfree(td);
00154       td = new_td;
00155     }
00156     
00157     // Only now compute how many tuples are needed!
00158     n_words = BitSetData::data(n_tuples);
00159 
00160     // Compute range information
00161     {
00162       /*
00163        * Pass one: compute how many values and ranges are needed
00164        */
00165       // How many values
00166       unsigned int n_vals = 0U;
00167       // How many ranges
00168       unsigned int n_ranges = 0U;
00169       for (int a=arity; a--; ) {
00170         // Sort tuple according to position
00171         PosCompare pc(a);
00172         Support::quicksort(tuple, n_tuples, pc);
00173         // Scan values
00174         {
00175           int max=tuple[0][a];
00176           n_vals++; n_ranges++;
00177           for (int i=1; i<n_tuples; i++) {
00178             assert(tuple[i-1][a] <= tuple[i][a]);
00179             if (max+1 == tuple[i][a]) {
00180               n_vals++;
00181               max=tuple[i][a];
00182             } else if (max+1 < tuple[i][a]) {
00183               n_vals++; n_ranges++;
00184               max=tuple[i][a];
00185             } else {
00186               assert(max == tuple[i][a]);
00187             }
00188           }
00189         }
00190       }
00191       /*
00192        * Pass 2: allocate memory and fill data structures
00193        */
00194       // Allocate memory for ranges
00195       Range* cr = range = heap.alloc<Range>(n_ranges);
00196       // Allocate and initialize memory for supports
00197       BitSetData* cs = support = heap.alloc<BitSetData>(n_words * n_vals);
00198       for (unsigned int i=n_vals * n_words; i--; )
00199         cs[i].init();
00200       for (int a=arity; a--; ) {
00201         // Set range pointer
00202         vd[a].r = cr;
00203         // Sort tuple according to position
00204         PosCompare pc(a);
00205         Support::quicksort(tuple, n_tuples, pc);
00206         // Update min and max
00207         min = std::min(min,tuple[0][a]);
00208         max = std::max(max,tuple[n_tuples-1][a]);
00209         // Compress into non-overlapping ranges
00210         {
00211           unsigned int j=0U;
00212           vd[a].r[0].max=vd[a].r[0].min=tuple[0][a];
00213           for (int i=1; i<n_tuples; i++) {
00214             assert(tuple[i-1][a] <= tuple[i][a]);
00215             if (vd[a].r[j].max+1 == tuple[i][a]) {
00216               vd[a].r[j].max=tuple[i][a];
00217             } else if (vd[a].r[j].max+1 < tuple[i][a]) {
00218               j++; vd[a].r[j].min=vd[a].r[j].max=tuple[i][a];
00219             } else {
00220               assert(vd[a].r[j].max == tuple[i][a]);
00221             }
00222           }
00223           vd[a].n = j+1U;
00224           cr += j+1U;
00225         }
00226         // Set support pointer and set bits
00227         for (unsigned int i=0U; i<vd[a].n; i++) {
00228           vd[a].r[i].s = cs;
00229           cs += n_words * vd[a].r[i].width();
00230         }
00231         {
00232           int j=0;
00233           for (int i=0; i<n_tuples; i++) {
00234             while (tuple[i][a] > vd[a].r[j].max)
00235               j++;
00236             set(const_cast<BitSetData*>
00237                 (vd[a].r[j].supports(n_words,tuple[i][a])),
00238                 tuple2idx(tuple[i]));
00239           }
00240         }
00241       }
00242       assert(cs == support + n_words * n_vals);
00243       assert(cr == range + n_ranges);
00244     }
00245     if ((min < Int::Limits::min) || (max > Int::Limits::max))
00246       throw Int::OutOfLimits("TupleSet::finalize()");
00247     assert(finalized());
00248   }

00292       unsigned int i_deg; 
00293       unsigned int o_deg; 
00294       unsigned int n_tuples;

00296     };
00298     struct Support {
00299       int val; 
00300       unsigned int n_edges; 
00301       Edge* edges; 
00302     };
00304     struct Layer {
00305       State* states; 
00306       Support* supports; 
00307       unsigned int n_supports; 
00308     };
00309     // Initialize
00310     object(new Data(a));
00311 
00312     Region r;
00313     // Number of states
00314     int max_states = dfa.n_states();
00315     // Allocate memory for all layers and states
00316     Layer* layers = r.alloc<Layer>(a+1);
00317     State* states = r.alloc<State>(max_states*(a+1));
00318 
00319     for (int i=max_states*(a+1); i--; ) {
00320       states[i].i_deg = 0U; states[i].o_deg = 0U;
00321       states[i].n_tuples = 0U;
00322       states[i].tuples = nullptr;
00323     }
00324     for (int i=a+1; i--; ) {
00325       layers[i].states = states + i*max_states;
00326       layers[i].n_supports = 0U;
00327     }
00328 
00329     // Mark initial state as being reachable
00330     layers[0].states[0].i_deg = 1U;
00331     layers[0].states[0].n_tuples = 1U;
00332     layers[0].states[0].tuples = r.alloc<int>(1U);
00333     assert(layers[0].states[0].tuples != nullptr);
00334   
00335     // Allocate temporary memory for edges and supports
00336     Edge* edges = r.alloc<Edge>(dfa.max_degree());
00337     Support* supports = r.alloc<Support>(dfa.n_symbols());
00338   
00339     // Forward pass: accumulate
00340     for (int i=0; i<a; i++) {
00341       unsigned int n_supports=0;
00342       for (DFA::Symbols s(dfa); s(); ++s) {
00343         unsigned int n_edges=0;
00344         for (DFA::Transitions t(dfa,s.val()); t(); ++t) {
00345           if (layers[i].states[t.i_state()].i_deg != 0) {
00346             // Create edge
00347             edges[n_edges].i_state = t.i_state();
00348             edges[n_edges].o_state = t.o_state();
00349             n_edges++;
00350             // Adjust degrees
00351             layers[i].states[t.i_state()].o_deg++;
00352             layers[i+1].states[t.o_state()].i_deg++;
00353             // Adjust number of tuples
00354             layers[i+1].states[t.o_state()].n_tuples
00355               += layers[i].states[t.i_state()].n_tuples;
00356           }
00357           assert(n_edges <= dfa.max_degree());
00358         }
00359         // Found a support for the value
00360         if (n_edges > 0U) {
00361           Support& support = supports[n_supports++];
00362           support.val = s.val();
00363           support.n_edges = n_edges;
00364           support.edges = Heap::copy(r.alloc<Edge>(n_edges),edges,n_edges);
00365         }
00366       }
00367       // Create supports
00368       if (n_supports > 0U) {
00369         layers[i].supports =
00370           Heap::copy(r.alloc<Support>(n_supports),supports,n_supports);
00371         layers[i].n_supports = n_supports;
00372       } else {
00373         finalize();
00374         return;
00375       }
00376     }
00377 
00378     // Mark final states as being reachable
00379     for (int s = dfa.final_fst(); s < dfa.final_lst(); s++) {
00380       if (layers[a].states[s].i_deg != 0U)
00381         layers[a].states[s].o_deg = 1U;
00382     }
00383 
00384     // Backward pass: validate
00385     for (int i=a; i--; ) {
00386       for (unsigned int j = layers[i].n_supports; j--; ) {
00387         Support& s = layers[i].supports[j];
00388         for (unsigned int k = s.n_edges; k--; ) {
00389           unsigned int i_state = s.edges[k].i_state;
00390           unsigned int o_state = s.edges[k].o_state;
00391           // State is unreachable
00392           if (layers[i+1].states[o_state].o_deg == 0U) {
00393             // Adjust degree
00394             --layers[i+1].states[o_state].i_deg;
00395             --layers[i].states[i_state].o_deg;
00396             // Remove edge
00397             assert(s.n_edges > 0U);
00398             s.edges[k] = s.edges[--s.n_edges];
00399           }
00400         }
00401         // Lost support
00402         if (s.n_edges == 0U)
00403           layers[i].supports[j] = layers[i].supports[--layers[i].n_supports];
00404       }
00405       if (layers[i].n_supports == 0U) {
00406         finalize();
00407         return;
00408       }
00409     }
00410 
00411     // Generate tuples
00412     for (int i=0; i<a; i++) {
00413       for (unsigned int j = layers[i].n_supports; j--; ) {
00414         Support& s = layers[i].supports[j];
00415         for (unsigned int k = s.n_edges; k--; ) {
00416           unsigned int i_state = s.edges[k].i_state;
00417           unsigned int o_state = s.edges[k].o_state;
00418           // Allocate memory for tuples if not done
00419           if (layers[i+1].states[o_state].tuples == nullptr) {
00420             unsigned int n_tuples = layers[i+1].states[o_state].n_tuples;
00421             layers[i+1].states[o_state].tuples = r.alloc<int>((i+1)*n_tuples);
00422             layers[i+1].states[o_state].n_tuples = 0;
00423           }
00424           unsigned int n = layers[i+1].states[o_state].n_tuples;
00425           // Write tuples
00426           for (unsigned int t=0; t < layers[i].states[i_state].n_tuples; t++) {
00427             // Copy the first i number of digits from the previous layer
00428             Heap::copy(&layers[i+1].states[o_state].tuples[n*(i+1)+t*(i+1)],
00429                        &layers[i].states[i_state].tuples[t*i], i);
00430             // Write the last digit
00431             layers[i+1].states[o_state].tuples[n*(i+1)+t*(i+1)+i] = s.val;
00432           }
00433           layers[i+1].states[o_state].n_tuples
00434             += layers[i].states[i_state].n_tuples;
00435         }
00436       }
00437     }
00438 
00439     // Add tuples to tuple set
00440     for (int s = dfa.final_fst(); s < dfa.final_lst(); s++) {
00441       for (unsigned int i=0; i<layers[a].states[s].n_tuples; i++) {
00442         int* tuple = &layers[a].states[s].tuples[i*a];
00443         add(IntArgs(a,tuple));
00444       }
00445     }
00446   
00447     finalize();
00448   }
