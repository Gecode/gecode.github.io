00096   forceinline unsigned int
00097   TupleSet::ValueData::start(int k) const {
00098     if (n > 1U) {
00099       unsigned int l=0U, h=n-1U;
00100       while (true) {
00101         assert(l<=h);
00102         unsigned int m = l + ((h-l) >> 1);
00103         if (k < r[m].min)
00104           h=m-1U;
00105         else if (k > r[m].max)
00106           l=m+1U;
00107         else
00108           return m;
00109       }
00110       GECODE_NEVER;
