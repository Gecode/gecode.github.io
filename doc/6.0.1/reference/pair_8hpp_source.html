00043   Pair::post(Home home, IntView x0, IntView x1, IntView x2,
00044              int w, int h) {
00045     GECODE_ME_CHECK(x0.gq(home,0)); GECODE_ME_CHECK(x0.le(home,w));
00046     GECODE_ME_CHECK(x1.gq(home,0)); GECODE_ME_CHECK(x1.le(home,h));
00047     GECODE_ME_CHECK(x2.gq(home,0)); GECODE_ME_CHECK(x2.le(home,w*h));
00048     if (x0.assigned() && x1.assigned()) {
00049       GECODE_ME_CHECK(x2.eq(home,x0.val()+w*x1.val()));
00050     } else if (x1.assigned()) {
00051       OffsetView x0x1w(x0,x1.val()*w);
00052       return Rel::EqDom<OffsetView,IntView>::post(home,x0x1w,x2);
00053     } else if (x2.assigned()) {
00054       GECODE_ME_CHECK(x0.eq(home,x2.val() % w));
00055       GECODE_ME_CHECK(x1.eq(home,static_cast<int>(x2.val() / w)));
00056     } else {
00057       assert(!shared(x0,x2) && !shared(x1,x2));
00058       (void) new (home) Pair(home,x0,x1,x2,w);
00059     }
00060     return ES_OK;
00061   }
