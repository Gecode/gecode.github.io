00226   forceinline void
00227   SearchTracer::engine(EngineType t, unsigned int n) {
00228     assert(pending > 0);
00229     pending += n-1;
00230     switch (t) {
00231     case EngineType::PBS: case EngineType::RBS:
00232       es[n_e]=EngineInfo(t,n_e+1,n_e+1+n);
00233       break;
00234     case EngineType::DFS: case EngineType::BAB:
00235     case EngineType::LDS: case EngineType::AOE:
00236       es[n_e]=EngineInfo(t,n_w,n_w+n);
00237       break;
00238     default: GECODE_NEVER;
00239     }
00240     n_e++;
00241     assert(pending > 0);
00242   }

00244   forceinline void
00245   SearchTracer::worker(unsigned int& wid, unsigned int& eid) {
00246     assert(pending > 0);
00247     pending--;
00248     w2e[n_w]=eid=n_e-1;
00249     wid=n_w++;
00250     if (pending == 0) {
00251       n_active = n_w;
00252       init();
00253     }
00254   }
