03590   forceinline Brancher*
03591   Space::brancher(unsigned int id) {
03592     /*
03593      * Due to weakly monotonic propagators the following scenario might
03594      * occur: a brancher has been committed with all its available
03595      * choices. Then, propagation determines less information
03596      * than before and the brancher now will create new choices.
03597      * Later, during recomputation, all of these choices
03598      * can be used together, possibly interleaved with
03599      * choices for other branchers. That means all branchers
03600      * must be scanned to find the matching brancher for the choice.
03601      *
03602      * b_commit tries to optimize scanning as it is most likely that
03603      * recomputation does not generate new choices during recomputation
03604      * and hence b_commit is moved from newer to older branchers.
03605      */
03606     Brancher* b_old = b_commit;
03607     // Try whether we are lucky
03608     while (b_commit != Brancher::cast(&bl))
03609       if (id != b_commit->id())
03610         b_commit = Brancher::cast(b_commit->next());
03611       else
03612         return b_commit;
03613     if (b_commit == Brancher::cast(&bl)) {
03614       // We did not find the brancher, start at the beginning
03615       b_commit = Brancher::cast(bl.next());
03616       while (b_commit != b_old)
03617         if (id != b_commit->id())
03618           b_commit = Brancher::cast(b_commit->next());
03619         else
03620           return b_commit;

03851   template<class A>
03852   forceinline void
03853   Council<A>::update(Space& home, Council<A>& c) {
03854     // Skip all disposed advisors
03855     {
03856       ActorLink* a = c.advisors;
03857       while ((a != NULL) && static_cast<A*>(a)->disposed())
03858         a = a->next();
03859       c.advisors = a;
03860     }
03861     // Are there any advisors to be cloned?
03862     if (c.advisors != NULL) {
03863       // The propagator in from-space
03864       Propagator* p_f = &static_cast<A*>(c.advisors)->propagator();
03865       // The propagator in to-space
03866       Propagator* p_t = Propagator::cast(p_f->prev());
03867       // Advisors in from-space
03868       ActorLink** a_f = &c.advisors;
03869       // Advisors in to-space
03870       A* a_t = NULL;
03871       while (*a_f != NULL) {
03872         if (static_cast<A*>(*a_f)->disposed()) {
03873           *a_f = (*a_f)->next();
03874         } else {
03875           // Run specific copying part
03876           A* a = new (home) A(home,*static_cast<A*>(*a_f));
03877           // Set propagator pointer
03878           a->prev(p_t);
03879           // Set forwarding pointer
03880           (*a_f)->prev(a);
03881           // Link
03882           a->next(a_t);
03883           a_t = a;
03884           a_f = (*a_f)->next_ref();
03885         }
03886       }
03887       advisors = a_t;
03888       // Enter advisor link for reset
03889       assert(p_f->u.advisors == NULL);
03890       p_f->u.advisors = c.advisors;
03891     } else {
03892       advisors = NULL;
03893     }
03894   }

04233     if (b.base == NULL) {
04234       assert((free_and_bits >> free_bits) == 0);
04235       // Create fresh dependency array with four entries
04236       free_and_bits += 4 << free_bits;
04237       b.base = home.alloc<ActorLink*>(4);

04244       // reserved for subscriptions. If yes, just resize mildly otherwise
04245       // more agressively
04246       ActorLink** s = static_cast<ActorLink**>(home.mm.subscriptions());
04247       unsigned int m =
04248         ((s <= b.base) && (b.base < s+home.pc.p.n_sub)) ?
04249         (n+4) : ((n+1)*3>>1);

04251       free_and_bits += (m-n) << free_bits;
04252       // Copy entries
04253       Heap::copy<ActorLink*>(prop, b.base, n);
04254       home.free<ActorLink*>(b.base,n);

04344   VarImp<VIC>::remove(Space& home, Propagator* p, PropCond pc) {
04345     assert(pc <= pc_max);
04346     ActorLink* a = ActorLink::cast(p);
04347     // Find actor in dependency array
04348     ActorLink** f = actor(pc);
04349 #ifdef GECODE_AUDIT

04351       if (*f == a)
04352         goto found;
04353       else

04355     GECODE_NEVER;
04356   found: ;
04357 #else
04358     while (*f != a) f++;

04429   template<class VIC>
04430   forceinline bool
04431   VarImp<VIC>::advise(Space& home, ModEvent me, Delta& d) {
04432     /*
04433      * An advisor that is executed might remove itself due to subsumption.
04434      * As entries are removed from front to back, the advisors must
04435      * be iterated in forward direction.
04436      */
04437     ActorLink** la = actorNonZero(pc_max+1);
04438     ActorLink** le = b.base+entries;
04439     if (la == le)
04440       return true;
04441     d.me = me;
04442     // An advisor that is run, might be removed during execution.
04443     // As removal is done from the back the advisors have to be executed
04444     // in inverse order.
04445     do {
04446       Advisor* a = Advisor::cast
04447         (static_cast<ActorLink*>(Support::funmark(*la)));
04448       assert(!a->disposed());
04449       Propagator& p = a->propagator();
04450       switch (p.advise(home,*a,d)) {
04451       case ES_FIX:
04452         break;
04453       case ES_FAILED:
04454         return false;
04455       case ES_NOFIX:
04456         schedule(home,p,me);
04457         break;
04458       case ES_NOFIX_FORCE:
04459         schedule(home,p,me,true);
04460         break;
04461       case __ES_SUBSUMED:
04462       default:
04463         GECODE_NEVER;
04464       }
04465     } while (++la < le);
04466     return true;
04467   }

04473      * An advisor that is executed might remove itself due to subsumption.

04502   template<class VIC>
04503   forceinline void
04504   VarImp<VIC>::update(VarImp<VIC>* x, ActorLink**& sub) {
04505     // this refers to the variable to be updated (clone)
04506     // x refers to the original
04507     // Recover from copy
04508     x->b.base = b.base;
04509     x->u.idx[0] = u.idx[0];
04510     if (pc_max > 0 && sizeof(ActorLink**) > sizeof(unsigned int))
04511       x->u.idx[1] = u.idx[1];
04512 
04513     unsigned int np =
04514       static_cast<unsigned int>(x->actorNonZero(pc_max+1) - x->actor(0));
04515     unsigned int na = 
04516       static_cast<unsigned int >(x->b.base + x->entries - 
04517                                  x->actorNonZero(pc_max+1));
04518     unsigned int n  = na + np;
04519     assert(n == x->degree());
04520 
04521     ActorLink** f = x->b.base;
04522     ActorLink** t = sub;
04523 
04524     sub += n;
04525     b.base = t;
04526     // Process propagator subscriptions
04527     while (np >= 4) {
04528       ActorLink* p3 = f[3]->prev();
04529       ActorLink* p0 = f[0]->prev();
04530       ActorLink* p1 = f[1]->prev();
04531       ActorLink* p2 = f[2]->prev(); 
04532       t[0] = p0; t[1] = p1; t[2] = p2; t[3] = p3;
04533       np -= 4; t += 4; f += 4;
04534     }
04535     if (np >= 2) {
04536       ActorLink* p0 = f[0]->prev();
04537       ActorLink* p1 = f[1]->prev();
04538       t[0] = p0; t[1] = p1;
04539       np -= 2; t += 2; f += 2;
04540     }
04541     if (np > 0) {
04542       ActorLink* p0 = f[0]->prev();
04543       t[0] = p0;
04544       t += 1; f += 1;
04545     }
04546     // Process advisor subscriptions
04547     while (na >= 4) {
04548       ptrdiff_t m0, m1, m2, m3;
04549       ActorLink* p3 =
04550         static_cast<ActorLink*>(Support::ptrsplit(f[3],m3))->prev();
04551       ActorLink* p0 = 
04552         static_cast<ActorLink*>(Support::ptrsplit(f[0],m0))->prev();
04553       ActorLink* p1 =
04554         static_cast<ActorLink*>(Support::ptrsplit(f[1],m1))->prev();
04555       ActorLink* p2 =
04556         static_cast<ActorLink*>(Support::ptrsplit(f[2],m2))->prev();
04557       t[0] = static_cast<ActorLink*>(Support::ptrjoin(p0,m0));
04558       t[1] = static_cast<ActorLink*>(Support::ptrjoin(p1,m1));
04559       t[2] = static_cast<ActorLink*>(Support::ptrjoin(p2,m2));
04560       t[3] = static_cast<ActorLink*>(Support::ptrjoin(p3,m3));
04561       na -= 4; t += 4; f += 4;
04562     }
04563     if (na >= 2) {
04564       ptrdiff_t m0, m1;
04565       ActorLink* p0 = 
04566         static_cast<ActorLink*>(Support::ptrsplit(f[0],m0))->prev();
04567       ActorLink* p1 =
04568         static_cast<ActorLink*>(Support::ptrsplit(f[1],m1))->prev();
04569       t[0] = static_cast<ActorLink*>(Support::ptrjoin(p0,m0));
04570       t[1] = static_cast<ActorLink*>(Support::ptrjoin(p1,m1));
04571       na -= 2; t += 2; f += 2;
04572     }
04573     if (na > 0) {
04574       ptrdiff_t m0;
04575       ActorLink* p0 = 
04576         static_cast<ActorLink*>(Support::ptrsplit(f[0],m0))->prev();
04577       t[0] = static_cast<ActorLink*>(Support::ptrjoin(p0,m0));
04578     }
04579   }
