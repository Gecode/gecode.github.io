00093   void
00094   DynamicQueue<T,A>::resize(void) {
00095     assert(fst == lst);
00096     T* nq = a.template alloc<T>(limit << 1);
00097     int j=0;
00098     for (int i = fst; i<limit; i++)
00099       nq[j++] = q[i];
00100     for (int i = 0; i<lst; i++)
00101       nq[j++] = q[i];
00102     a.template free<T>(q,limit);
00103     q = nq;
00104     fst = 0;
00105     lst = limit;
00106     limit <<= 1;
00107   }
