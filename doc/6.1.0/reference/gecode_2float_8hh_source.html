01974   /*
01975    * \brief Relaxed assignment of variables in \a x from values in \a sx
01976    *
01977    * The variables in \a x are assigned values from the assigned variables
01978    * in the solution \a sx with a relaxation probability \a p. That is,
01979    * if \$fp=0.1\f$ approximately 10% of the variables in \a x will be
01980    * assigned a value from \a sx.
01981    *
01982    * The random numbers are generated from the generator \a r. At least
01983    * one variable will not be assigned: in case the relaxation attempt
01984    * would suggest that all variables should be assigned, a single
01985    * variable will be selected randomly to remain unassigned.
01986    *
01987    * Throws an exception of type Float::ArgumentSizeMismatch, if \a x and
01988    * \a sx are of different size.
01989    *
01990    * Throws an exception of type Float::OutOfLimits, if \a p is not between
01991    * \a 0.0 and \a 1.0.
01992    *
01993    * \ingroup TaskModeFloat
01994    */
01995   GECODE_FLOAT_EXPORT void
01996   relax(Home home, const FloatVarArgs& x, const FloatVarArgs& sx,
