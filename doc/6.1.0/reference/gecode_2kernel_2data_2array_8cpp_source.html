00049   duplicates(void** p, int n, void** q, int m) {
00050     assert((n > 0) && (m > 0));
00051     Support::quicksort<void*>(p,n);
00052     Support::quicksort<void*>(q,m);
00053     int i=0, j=0;
00054     do {
00055       if (p[i] == q[j])
00056         return true;
00057       else if (p[i] < q[j])
00058         i++;
00059       else
00060         j++;
00061     } while ((i < n) && (j < m));
00062     return false;
