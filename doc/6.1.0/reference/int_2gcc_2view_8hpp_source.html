00044   forceinline bool
00045   lookupValue(T& a, int v, int& i) {
00046     int l = 0;
00047     int r = a.size() - 1;
00048 
00049     while (l <= r) {
00050       int m = l + (r - l) / 2;
00051       if (v == a[m].card()) {
00052         i=m; return true;
00053       } else if (l == r) {
00054         return false;
00055       } else if (v < a[m].card()) {
00056         r=m-1;
00057       } else {
00058         l=m+1;
00059       }
00060     }
00061     return false;
00062   }
