00047   forceinline void
00048   LayoutCursor::processCurrentNode(void) {
00049     VisualNode* currentNode = node();
00050     if (currentNode->isDirty()) {
00051       if (currentNode->isHidden()) {
00052         // do nothing
00053       } else if (false && currentNode->getNumberOfChildren() < 1) {
00054         currentNode->setShape(Shape::leaf);
00055       } else {
00056         currentNode->computeShape(na);
00057       }
00058       currentNode->setDirty(false);
00059     }
00060     if (currentNode->getNumberOfChildren() >= 1)
00061       currentNode->setChildrenLayoutDone(true);
00062   }
