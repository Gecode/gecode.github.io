00041   template<class Char, class Traits, class I>
00042   void
00043   printBound(std::basic_ostream<Char,Traits>& s, I& r) {
00044     s << '{';
00045     while (r()) {
00046       if (r.min() == r.max()) {
00047         s << r.min();
00048       } else if (r.min()+1 == r.max()) {
00049         s << r.min() << "," << r.max();
00050       } else {
00051         s << r.min() << ".." << r.max();
00052       }
00053       ++r;
00054       if (!r()) break;
00055       s << ',';
00056     }
00057     s << '}';
00058   }
