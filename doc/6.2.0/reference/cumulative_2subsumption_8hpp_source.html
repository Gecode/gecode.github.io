00036   template<class Task>
00037   forceinline ExecStatus
00038   subsumed(Space& home, Propagator& p, int c, TaskArray<Task>& t) {
00039     Region r;
00040 
00041     if (Event* e = Event::events(r,t)) {
00042       // Process events, use ccur as the capacity that is still free
00043       do {
00044         // Current time
00045         int time = e->time();
00046         // Process events for completion of required part
00047         for ( ; (e->type() == Event::LRT) && (e->time() == time); e++)
00048           c += t[e->idx()].c();
00049         // Process events for zero-length task
00050         for ( ; (e->type() == Event::ZRO) && (e->time() == time); e++)
00051           if (c < t[e->idx()].c())
00052             return ES_FAILED;
00053         // Process events for start of required part
00054         for ( ;  (e->type() == Event::ERT) && (e->time() == time); e++) {
00055           c -= t[e->idx()].c();
00056           if (c < 0)
00057             return ES_FAILED;
00058         }
00059       } while (e->type() != Event::END);
00060     } else {
00061       return ES_OK;
00062     }
00063 
00064     return home.ES_SUBSUMED(p);
00065   }
