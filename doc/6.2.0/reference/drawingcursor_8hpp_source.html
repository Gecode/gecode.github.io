00043   forceinline bool
00044   DrawingCursor::isClipped(void) {
00045     if (clippingRect.width() == 0 && clippingRect.x() == 0
00046         && clippingRect.height() == 0 && clippingRect.y() == 0)
00047       return false;
00048     BoundingBox b = node()->getBoundingBox();
00049     return (x + b.left > clippingRect.x() + clippingRect.width() ||
00050             x + b.right < clippingRect.x() ||
00051             y > clippingRect.y() + clippingRect.height() ||
00052             y + (node()->getShape()->depth()+1) * Layout::dist_y <
00053             clippingRect.y());
00054   }
