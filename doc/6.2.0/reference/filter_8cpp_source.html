00043   TFE::Node::decrement(void) {
00044     if (--use == 0) {
00045       if ((l != NULL) && l->decrement())
00046         delete l;

00158   void
00159   TraceFilter::TFO::fill(TFE::Node* n) {
00160     Region region;
00161     Support::DynamicStack<StackFrame,Region> next(region);
00162     int i=0;
00163     next.push(StackFrame(n,false));
00164     do {
00165       StackFrame s = next.pop();
00166       switch (s.n->t) {
00167       case TFE::NT_GROUP:
00168         f[i].g = s.n->g; f[i].neg = s.neg; f[i].what=s.n->w;
00169         i++;
00170         break;
00171       case TFE::NT_NEGATE:
00172         next.push(StackFrame(s.n->l,!s.neg));
00173         break;
00174       case TFE::NT_ADD:
00175         next.push(StackFrame(s.n->l,s.neg));
00176         next.push(StackFrame(s.n->r,s.neg));
00177         break;
00178       default: GECODE_NEVER;
00179       }
00180     } while (!next.empty());
00181   }
