00068   ExecStatus
00069   Div<A,B,C>::propagate(Space& home, const ModEventDelta&) {
00070     if (x1.assigned() && (x1.val() == 0)) return ES_FAILED;
00071     GECODE_ME_CHECK(x2.eq(home,x0.domain() / x1.domain()));
00072     GECODE_ME_CHECK(x0.eq(home,x2.domain() * x1.domain()));
00073     if (!x2.assigned() || (x2.val() != 0.0))
00074       GECODE_ME_CHECK(x1.eq(home,x0.domain() / x2.domain()));
00075     return (x0.assigned() && x1.assigned()) ? home.ES_SUBSUMED(*this) : ES_NOFIX;
