00058       if (vti.brancher().group().in())

00061       break;
00062     case ViewTraceInfo::POST:
00063       s << "post(";
00064       if (vti.post().in())
00065         s << "g:" << vti.post().id();
00066       s << ')';
00067       break;
00068     case ViewTraceInfo::OTHER:
00069       s << '-';
00070       break;
00071     default:
00072       GECODE_NEVER;
00073     }
00074     return os << s.str();
