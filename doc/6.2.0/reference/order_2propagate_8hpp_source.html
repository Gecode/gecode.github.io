00047   forceinline ExecStatus
00048   OrderProp::post(Home home, IntView s0, int p0, IntView s1, int p1,
00049                   BoolView b) {
00050     if (s0.max() + p0 <= s1.min()) {
00051       GECODE_ME_CHECK(b.zero(home));
00052     } else if (s1.max() + p1 <= s0.min()) {
00053       GECODE_ME_CHECK(b.one(home));
00054     } else if ((s1.min() + p1 > s0.max()) || b.zero()) {
00055       GECODE_ME_CHECK(b.zero(home));
00056       return Rel::Lq<OffsetView,IntView>::post(home,OffsetView(s0,p0),s1);
00057     } else if ((s0.min() + p0 > s1.max()) || b.one()) {
00058       GECODE_ME_CHECK(b.one(home));
00059       return Rel::Lq<OffsetView,IntView>::post(home,OffsetView(s1,p1),s0);
00060     } else {
00061       (void) new (home) OrderProp(home,s0,p0,s1,p1,b);
00062     }
00063     return ES_OK;
00064   }
