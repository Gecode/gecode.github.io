00092   forceinline unsigned int
00093   TupleSet::ValueData::start(int k) const {
00094     if (n > 1U) {
00095       unsigned int l=0U, h=n-1U;
00096       while (true) {
00097         assert(l<=h);
00098         unsigned int m = l + ((h-l) >> 1);
00099         if (k < r[m].min)
00100           h=m-1U;
00101         else if (k > r[m].max)
00102           l=m+1U;
00103         else
00104           return m;
00105       }
00106       GECODE_NEVER;
00107     } else {
00108       return 0U;
00109     }
00110   }
