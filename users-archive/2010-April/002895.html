If I understood you correctly, this should do the trick:

BoolVarArray rtrue(*this, m.height(), 0,1);
or (int i=0; i < m.height() ;  i++)  
      rel(*this, BOT_AND,  m.row(i), rtrue[i] );
rel(*this, BOT_OR, rtrue);
