<p class=MsoNormal><span style='font-size:11.0pt;font-family:"Calibri","sans-serif";
color:#1F497D'>No, you can't delete. The idea is: the i-th view (that is, x[i])
is assigned the last view in the array (x[n-1]) and the size of the array is shrunk
by one (--n). Unfortunately, the propagator Tias sent is non-sensical in that
it does not implement a constraint: its semantics is random!<o:p></o:p></span></p>

<p class=MsoNormal><span style='font-size:11.0pt;font-family:"Calibri","sans-serif";
color:#1F497D'><o:p>&nbsp;</o:p></span></p>

<p class=MsoNormal><span style='font-size:11.0pt;font-family:"Calibri","sans-serif";
color:#1F497D'>The reason why you can't delete it here is that it is also used
for detecting subsumption: any propagator must at very latest report
subsumption if it is not failed and all of its views are assigned!<o:p></o:p></span></p>

#include &lt;gecode/kernel.hh&gt;<br>
#include &lt;gecode/int.hh&gt;<br>
<br>
namespace myprop {<br>
&nbsp; using namespace ::Gecode;<br>
&nbsp; using namespace ::Gecode::Int;<br>
<br>
&nbsp; /**<br>
&nbsp;&nbsp; * \brief Base-class for simple cleaner<br>
&nbsp;&nbsp; */<br>
&nbsp; template &lt;class V&gt;<br>
&nbsp; class CleanerBool : public Propagator {<br>
&nbsp; protected:<br>
&nbsp;&nbsp;&nbsp; /// Boolean views<br>
&nbsp;&nbsp;&nbsp; ViewArray&lt;V&gt; x;<br>
<br>
&nbsp;&nbsp;&nbsp; /// Constructor for creation<br>
&nbsp;&nbsp;&nbsp; CleanerBool(Space&amp; home, ViewArray&lt;V&gt;&amp; x);<br>
&nbsp;&nbsp;&nbsp; /// Constructor for cloning \a p<br>
&nbsp;&nbsp;&nbsp; CleanerBool(Space&amp; home, bool share, CleanerBool&amp;
p);<br>
&nbsp; public:<br>
&nbsp;&nbsp;&nbsp; /// Cost function (defined as low linear)<br>
&nbsp;&nbsp;&nbsp; virtual PropCost cost(const Space&amp; home, const
ModEventDelta&amp; med) const;<br>
&nbsp;&nbsp;&nbsp; /// Delete propagator and return its size<br>
//&nbsp;&nbsp; virtual size_t dispose(Space&amp; home);<br>
&nbsp;&nbsp;&nbsp; /// Create copy during cloning<br>
&nbsp;&nbsp;&nbsp; virtual Actor* copy(Space&amp; home, bool share);<br>
&nbsp;&nbsp;&nbsp; /// Perform propagation<br>
&nbsp;&nbsp;&nbsp; virtual ExecStatus propagate(Space&amp; home, const
ModEventDelta&amp; med);<br>
&nbsp;&nbsp;&nbsp; /// Post propagator<br>
&nbsp;&nbsp;&nbsp; static ExecStatus post(Space&amp; home,
ViewArray&lt;V&gt;&amp; x);<br>
&nbsp; };<br>
<br>
&nbsp; /*<br>
&nbsp;&nbsp; * propagator implementation<br>
&nbsp;&nbsp; */<br>
&nbsp; template &lt;class V&gt;<br>
&nbsp; CleanerBool&lt;V&gt;::CleanerBool(Space&amp; home,
ViewArray&lt;V&gt;&amp; x0)<br>
&nbsp;&nbsp;&nbsp; :&nbsp; Propagator(home), x(x0) {<br>
&nbsp;&nbsp;&nbsp; x.subscribe(home,*this,PC_INT_VAL);<br>
&nbsp; }<br>
<br>
&nbsp; template &lt;class V&gt;<br>
&nbsp; forceinline size_t<br>
&nbsp; CleanerBool&lt;V&gt;::dispose(Space&amp; home) {<br>
&nbsp;&nbsp;&nbsp; assert(!home-&gt;failed());<br>
&nbsp;&nbsp;&nbsp; x.cancel(home,*this,PC_INT_VAL);<br>
&nbsp;&nbsp;&nbsp; (void) Propagator::dispose(home);<br>
&nbsp;&nbsp;&nbsp; return sizeof(*this);<br>
&nbsp; }<br>
<br>
&nbsp; template &lt;class V&gt;<br>
&nbsp; forceinline<br>
&nbsp; CleanerBool&lt;V&gt;::CleanerBool(Space&amp; home, bool share,
CleanerBool&amp; p)<br>
&nbsp;&nbsp;&nbsp; : Propagator(home,share,p) {<br>
&nbsp;&nbsp;&nbsp; x.update(home,share,p.x);<br>
&nbsp; }<br>
<br>
&nbsp; template &lt;class V&gt;<br>
&nbsp; PropCost<br>
&nbsp; CleanerBool&lt;V&gt;::cost(const Space&amp;, const ModEventDelta&amp;)
const {<br>
&nbsp;&nbsp;&nbsp; // make sure this propagator is run as last before the
fixpoint<br>
&nbsp;&nbsp;&nbsp; return PropCost::crazy(PropCost::HI, 100000);<br>
&nbsp;&nbsp;&nbsp; // in a real propagator, this would be something more
sensible,<br>
&nbsp;&nbsp;&nbsp; // for example: PropCost::linear(PropCost::LO, x.size());<br>
&nbsp; }<br>
<br>
&nbsp; template &lt;class V&gt;<br>
&nbsp; ExecStatus<br>
&nbsp; CleanerBool&lt;V&gt;::post(Space&amp; home, ViewArray&lt;V&gt;&amp; x) {<br>
&nbsp;&nbsp;&nbsp; (void) new (home) CleanerBool&lt;V&gt;(home,x);<br>
&nbsp;&nbsp;&nbsp; return ES_OK;<br>
&nbsp; }<br>
<br>
&nbsp; template &lt;class V&gt;<br>
&nbsp; Actor*<br>
&nbsp; CleanerBool&lt;V&gt;::copy(Space&amp; home, bool share) {<br>
&nbsp;&nbsp;&nbsp; return new (home) CleanerBool&lt;V&gt;(home,share,*this);<br>
&nbsp; }<br>
<br>
&nbsp; template &lt;class V&gt;<br>
&nbsp; ExecStatus<br>
&nbsp; CleanerBool&lt;V&gt;::propagate(Space&amp; home, const
ModEventDelta&amp;) {<br>
&nbsp;&nbsp;&nbsp; int n = x.size();<br>
&nbsp;&nbsp;&nbsp; for (int i = n; i--; ) {<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; if (x[i].zero()) {<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; <span style='color:red'>x[i]=x[--n];</span><br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; } else if (x[i].one()) {<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; <span style='color:red'>x[i]=x[--n];</span><br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; } else {<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; GECODE_ME_CHECK( x[i].one(home) );<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;<span style='color:red'> x[i]=x[--n];</span><br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; } // all cases covered, so every var assigned
and removed<br>
&nbsp;&nbsp;&nbsp; }<br>
&nbsp;&nbsp;&nbsp; x.size(n);<br>
<br>
&nbsp;&nbsp;&nbsp; if (n == 0) {<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; return ES_SUBSUMED(*this,home);<br>
&nbsp;&nbsp;&nbsp; }<br>
&nbsp;&nbsp;&nbsp; return ES_FIX;<br>
&nbsp; }<br>
<br>
<br>
&nbsp; /** \brief Post propagator for cleanup of unassigned variables<br>
&nbsp;&nbsp; * Is run as last propagator (crazy_hi), sets unassigned variables
to 1<br>
&nbsp;&nbsp; */<br>
&nbsp; void cleaner(Space&amp; home, const BoolVarArgs&amp; x) {<br>
&nbsp;&nbsp;&nbsp; if (home.failed()) return;<br>
<br>
&nbsp;&nbsp;&nbsp; ViewArray&lt;BoolView&gt; Vx(home, x);<br>
<br>
&nbsp;&nbsp;&nbsp; GECODE_ES_FAIL(home,(CleanerBool&lt;BoolView&gt;::post(home,
Vx)));<br>
&nbsp; }<br>
<br>
}<br>
