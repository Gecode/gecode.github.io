      <br>
      <br>
      While I was add it, I added another feature.&nbsp; The patch includes a
      file "gecode/iter/wrap-val.hpp" (and a patch to "gecode/iter.hh"
      to include this file) which contains an iterator class "WrapVal"
      that wraps another iterator and upon dereferencing returns the the
      result of calling ".val()" on the dereferenced wrapped iterator,
      as well as a function "wrap_val" that is a convenience function
      for constructing a "WrapVal" (since this causes the template
      parameters to automatically be inferred).
      <br>
      <br>
      That might sound complicated, but the point of it is to allow
      someone to do the following:
      <br>
      <br>
      &nbsp;&nbsp;&nbsp; #include &lt;algorithm&gt;
      <br>
      &nbsp;&nbsp;&nbsp; #include &lt;gecode/iter.hh&gt;
      <br>
      &nbsp;&nbsp;&nbsp; #include &lt;iterator&gt;
      <br>
      &nbsp;&nbsp;&nbsp; #include &lt;vector&gt;
      <br>
      &nbsp;&nbsp;&nbsp; ...
      <br>
      &nbsp;&nbsp;&nbsp; using namespace Gecode;
      <br>
      &nbsp;&nbsp;&nbsp; using namespace Gecode::Iter;
      <br>
      &nbsp;&nbsp;&nbsp; using namespace std;
      <br>
      &nbsp;&nbsp;&nbsp; ...
      <br>
      &nbsp;&nbsp;&nbsp; IntVarArray a;
      <br>
      &nbsp;&nbsp;&nbsp; ...
      <br>
      &nbsp;&nbsp;&nbsp; vector&lt;int&gt; v;
      <br>
      &nbsp;&nbsp;&nbsp; copy(wrap_val(a.begin()),wrap_val(a.end()),back_inserter(x));
      <br>
      <br>
      which has the effect of copying all of the values from the array a
      into the vector v, since a.begin() returns an iterator that
      dereferences to an IntVar and wrap_val(a.begin()) returns a
      wrapped iterator that returns the result of calling .val() on the
      IntVar.
      <br>
      <br>
      The basic use case of the iterators is as follows:
      <br>
      <br>
      &nbsp;&nbsp;&nbsp; IntVarArgs a;
      <br>
      &nbsp;&nbsp;&nbsp; ...
      <br>
      &nbsp;&nbsp;&nbsp; for(IntVarArgs::iterator i = a.begin(); i != a.end(); ++i) {
      rel(space,*i,IRT_EQ,0); }
      <br>
      <br>
      Of course, you could more easily do this with
      rel(space,a,IRT_EQ,0).&nbsp; A more interesting example is:
      <br>
      <br>
      &nbsp;&nbsp;&nbsp; IntVarArgs a;
      <br>
      &nbsp;&nbsp;&nbsp; ...
      <br>
      &nbsp;&nbsp;&nbsp; for(IntVarArgs::iterator i = a.begin(); i != a.end()-1; ++i) {
      rel(space,*i &gt; *(i+1)+2); }
      <br>
      <br>
      More powerfully, iterators give you the ability to work with STL
      algorithms:
      <br>
      <br>
      &nbsp;&nbsp;&nbsp; #include &lt;algorithm&gt;
      <br>
      &nbsp;&nbsp;&nbsp; ...
      <br>
      &nbsp;&nbsp;&nbsp; void f(IntVar&amp; v) { ... }
      <br>
      &nbsp;&nbsp;&nbsp; ...
      <br>
      &nbsp;&nbsp;&nbsp; IntVarArgs a;
      <br>
      &nbsp;&nbsp;&nbsp; ...
      <br>
      &nbsp;&nbsp;&nbsp; std::for_each(a.begin(),a.end(),f);
      <br>
      <br>
      The iterators returned by a.begin() and a.end() dereference to a
      reference to an IntVar/BoolVar (or a view in the case of
      ViewArray).&nbsp; If you are interested in iterating over the *values*
      of the array --- that is, the value currently assigned to to each
      variable then you can use the function wrap_val to wrap these
      iterators so that dereferencing them returns the value assigned to
      the variable.&nbsp; For example, the last two lines in the following
      have exactly the same result:
      <br>
      <br>
      &nbsp;&nbsp;&nbsp; #include &lt;algorithm&gt;
      <br>
      &nbsp;&nbsp;&nbsp; #include &lt;iostream&gt;
      <br>
      &nbsp;&nbsp;&nbsp; ...
      <br>
      &nbsp;&nbsp;&nbsp; void f(IntVar&amp; v) { std::cout &lt;&lt; v.val() &lt;&lt; "
      "; }
      <br>
      &nbsp;&nbsp;&nbsp; void g(int i) { std::cout &lt;&lt; i &lt;&lt; " "; }
      <br>
      &nbsp;&nbsp;&nbsp; ...
      <br>
      &nbsp;&nbsp;&nbsp; IntVarArgs a;
      <br>
      &nbsp;&nbsp;&nbsp; ...
      <br>
      &nbsp;&nbsp;&nbsp; std::for_each(a.begin(),a.end(),f); std::cout &lt;&lt;
      std::endl;
      <br>
      &nbsp;&nbsp;&nbsp; std::for_each(wrap_val(a.begin()),wrap_val(a.end()),g);
      std::cout &lt;&lt; std::endl;
      <br>
      <br>
      Or if you want to copy the values assigned to the variables into a
      std::vector:
      <br>
      <br>
      &nbsp;&nbsp;&nbsp; #include &lt;algorithm&gt;
      <br>
      &nbsp;&nbsp;&nbsp; #include &lt;iterator&gt;
      <br>
      &nbsp;&nbsp;&nbsp; #include &lt;vector&gt;
      <br>
      &nbsp;&nbsp;&nbsp; ...
      <br>
      &nbsp;&nbsp;&nbsp; IntVarArray a;
      <br>
      &nbsp;&nbsp;&nbsp; ...
      <br>
      &nbsp;&nbsp;&nbsp; std::vector&lt;int&gt; v;
      <br>
      &nbsp;&nbsp;&nbsp;
      std::copy(wrap_val(a.begin()),wrap_val(a.end()),std::back_inserter(x));
      <br>
      <br>
      Or if you are interested in computing the sum of the values in the
      variables:
      <br>
      <br>
      &nbsp;&nbsp;&nbsp; #include &lt;numeric&gt;
      <br>
      &nbsp;&nbsp;&nbsp; ...
      <br>
      &nbsp;&nbsp;&nbsp; IntVarArray a;
      <br>
      &nbsp;&nbsp;&nbsp; ...
      <br>
      &nbsp;&nbsp;&nbsp; int sum =
      std::accumulate(wrap_val(a.begin()),wrap_val(a.end()),0);
      <br>
      <br>
      Or alternatively if you are interested in counting the number of
      zeros in the variables:
      <br>
      <br>
      &nbsp;&nbsp;&nbsp; #include &lt;algorithm&gt;
      <br>
      &nbsp;&nbsp;&nbsp; ...
      <br>
      &nbsp;&nbsp;&nbsp; IntVarArray a;
      <br>
      &nbsp;&nbsp;&nbsp; ...
      <br>
      &nbsp;&nbsp;&nbsp; unsigned int number_of_zeros =
      std::count(wrap_val(a.begin()),wrap_val(a.end()),0);
      <br>
